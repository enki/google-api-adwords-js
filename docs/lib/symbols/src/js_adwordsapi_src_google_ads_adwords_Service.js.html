 20 /**
 21  * @fileoverview Defines AdWords Service, the class that defines SOAP services
 22  * specific to AdWords.
 23  */
 24 
 25 goog.provide('google.ads.adwords.Service');
 26 
 27 goog.require('google.ads.adwords.ApiException');
 28 goog.require('google.ads.common.Service');
 29 
 30 /**
 31  * The class that defines SOAP services specific to AdWords.
 32  *
 33  * @extends google.ads.common.Service
 34  * @constructor
 35  */
 36 google.ads.adwords.Service = function() {
 37   google.ads.common.Service.call(this);
 38 };
 39 goog.inherits(google.ads.adwords.Service, google.ads.common.Service);
 40 
 41 /**
 42  * Gets a custom exception that wraps the SOAP exception thrown
 43  * by the server.
 44  *
 45  * @param {google.system.soap.Exception} ex The SOAP exception.
 46  * @return {google.system.core.Exception} A custom exception object
 47  * that wraps the SOAP exception.
 48  * @private
 49  */
 50 google.ads.adwords.Service.prototype.getCustomException_ =
 51     function(ex) {
 52   var doc = goog.dom.xml.loadXml(ex.getDetail());
 53   var rootNode = doc.documentElement.childNodes[0];
 54   var apiException = goog.getObjectByName(this.getType().getNamespace() +
 55                                           '.' + 'ApiException');
 56   apiException = new apiException();
 57   apiException.deserialize(rootNode, this.getNametable());
 58 
 59   return new google.ads.adwords.ApiException(
 60       apiException, ex.getMessage(), ex);
 61 };
