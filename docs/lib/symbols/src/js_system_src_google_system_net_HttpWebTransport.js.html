 19 /**
 20  * @fileoverview Defines HttpWebTransport, a wrapper for making HTTP calls.
 21  */
 22 
 23 goog.provide('google.system.net.HttpWebTransport');
 24 
 25 /**
 26  * This class defines a wrapper for making HTTP calls.
 27  *
 28  * @param {google.system.net.HttpWebTransportSettings} settings Settings to be
 29  * applied.
 30  * @constructor
 31  */
 32 google.system.net.HttpWebTransport = function(settings) {
 33   this.settings_ = settings;
 34 };
 35 
 36 /**
 37  * Sends a request to the requested server.
 38  *
 39  * @param {string} url The url to make the request to.
 40  * @param {string} method The HTTP method.
 41  * @param {Object} content The data to send to servers.
 42  * @param {Object} headers The HTTP headers.
 43  * @param {number} timeoutInterval The timeout interval.
 44  * @param {function} onSuccess The callback for success.
 45  * @param {function} onFailure The callback for failure.
 46  */
 47 google.system.net.HttpWebTransport.prototype.send = function(url, method,
 48     content, headers, timeoutInterval, onSuccess, onFailure) {
 49   // default class does nothing.
 50 };
 51 
 52 /**
 53  * Retrieves the transport settings.
 54  *
 55  * @return {google.system.net.HttpWebTransportSettings} Settings of the
 56  * transport.
 57  */
 58 google.system.net.HttpWebTransport.prototype.getSettings = function() {
 59   return this.settings_;
 60 };
