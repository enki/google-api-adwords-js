 19 /**
 20  * @fileoverview Defines HttpWebTransport, a wrapper that uses Google Gadgets
 21  *     API for making for making HTTP calls.
 22  */
 23 
 24 goog.provide('google.system.net.HttpWebTransportForGadgetsApi');
 25 
 26 goog.require('google.system.net.HttpWebTransport');
 27 
 28 /**
 29  * This class defines a wrapper that uses Google Gadgets API for making for
 30  * making HTTP calls.
 31  *
 32  * @param {google.system.net.HttpWebTransportSettings} settings Settings to be
 33  *     applied.
 34  * @extends google.system.net.HttpWebTransport
 35  * @constructor
 36  */
 37 google.system.net.HttpWebTransportForGadgetsApi = function(settings) {
 38   google.system.net.HttpWebTransport.call(this, settings);
 39 };
 40 goog.inherits(google.system.net.HttpWebTransportForGadgetsApi,
 41     google.system.net.HttpWebTransport);
 42 
 43 /**
 44  * Sends a request to the requested server.
 45  *
 46  * @param {string} url The url to make the request to.
 47  * @param {string} method The HTTP method.
 48  * @param {Object} content The data to send to servers.
 49  * @param {Object} headers The HTTP headers.
 50  * @param {number} timeoutInterval The timeout interval.
 51  * @param {function} onSuccess The callback for success.
 52  * @param {function} onFailure The callback for failure.
 53  */
 54 google.system.net.HttpWebTransportForGadgetsApi.prototype.send = function(url,
 55     method, content, headers, timeoutInterval, onSuccess, onFailure) {
 56   var params = {};
 57   params[gadgets.io.RequestParameters.METHOD] = method;
 58   params[gadgets.io.RequestParameters.POST_DATA] = content;
 59   params[gadgets.io.RequestParameters.HEADERS] = headers;
 60 
 61   gadgets.io.makeRequest(url, goog.bind(function(response) {
 62     if (!response.errors.length) {
 63       onSuccess(response.text);
 64     } else {
 65       onFailure(response.text);
 66     }
 67   }, this), params);
 68 };
