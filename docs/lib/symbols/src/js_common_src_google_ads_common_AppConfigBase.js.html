 19 /**
 20  * @fileoverview Defines AppConfigBase, the base class for all configuration
 21  * readers.
 22  */
 23 
 24 goog.provide('google.ads.common.AppConfigBase');
 25 
 26 goog.require('google.system.net.HttpWebTransportSettings');
 27 
 28 /**
 29  * Reads the configuration keys from the configuration files.
 30  *
 31  * @constructor
 32  */
 33 google.ads.common.AppConfigBase = function() {
 34   this.PROXY_URL_ = 'ProxyUrl';
 35   this.shortName_ = null;
 36   this.HTTP_WEBTRANSPORT_ = 'HttpWebTransport';
 37   this.HTTPWebTransport_ = null;
 38   this.HTTP_WEBTRANSPORT_SETTINGS_ = 'HttpWebTransportSettings';
 39   this.HTTPWebTransportSettings_ = null;
 40 };
 41 
 42 /**
 43  * Gets the HTTP web transport.
 44  *
 45  * @return {string} The HTTP web transport provider name.
 46  */
 47 google.ads.common.AppConfigBase.prototype.getHttpWebTransport = function() {
 48   return this.HTTPWebTransport_;
 49 };
 50 
 51 /**
 52  * Sets the HTTP web transport.
 53  *
 54  * @param {string} HTTPWebTransport The HTTP web transport provider name.
 55  */
 56 google.ads.common.AppConfigBase.prototype.setHttpWebTransport =
 57     function(HTTPWebTransport) {
 58   this.HTTPWebTransport_ = HTTPWebTransport;
 59 };
 60 
 61 /**
 62  * Gets the HTTP web transport settings.
 63  *
 64  * @return {string} The HTTP web transport settings.
 65  */
 66 google.ads.common.AppConfigBase.prototype.getHttpWebTransportSettings =
 67     function() {
 68   return this.HTTPWebTransportSettings_;
 69 };
 70 
 71 /**
 72  * Sets the HTTP web transport.
 73  *
 74  * @param {string} HTTPWebTransportSettings The HTTP web transport settings.
 75  */
 76 google.ads.common.AppConfigBase.prototype.setHttpWebTransportSettings =
 77     function(HTTPWebTransportSettings) {
 78   this.HTTPWebTransportSettings_ = HTTPWebTransportSettings;
 79 };
 80 
 81 /**
 82  * Gets the short name for the library.
 83  *
 84  * @return {string} The short name for the library.
 85  */
 86 google.ads.common.AppConfigBase.prototype.getShortName = function() {
 87   return this.shortName_;
 88 };
 89 
 90 /**
 91  * Gets a signature string for the library.
 92  *
 93  * @return {string} The signature string for the library.
 94  */
 95 google.ads.common.AppConfigBase.prototype.getSignature = function() {
 96   return this.getShortName() + '-' + this.getVersion();
 97 };
 98 
 99 /**
100  * Gets the library version.
101  *
102  * @return {string} The library version.
103  */
104 google.ads.common.AppConfigBase.prototype.getVersion = function() {
105   return this.getShortName() + '-' +
106       google.system.core.Runtime.getInstance().getVersion();
107 };
108 
109 /**
110  * Reads the settings from configuration file.
111  *
112  * @param {Array} settings Settings to be read and set.
113  */
114 google.ads.common.AppConfigBase.prototype.readSettings = function(
115     settings) {
116   this.setHttpWebTransport(settings[this.HTTP_WEBTRANSPORT_] ||
117       this.getHttpWebTransport());
118   this.setHttpWebTransportSettings(
119       new google.system.net.HttpWebTransportSettings(
120           settings[this.HTTP_WEBTRANSPORT_SETTINGS_] || {}));
121 };
122 
123 /**
124  * Retrieves a configuration value based on its name.
125  * @param {string} key The name of the configuration property.
126  * @return {object|null} The value or null if does not exists.
127  */
128 google.ads.common.AppConfigBase.prototype.getConfigValue = function(key) {
129   var getter = 'get' + key.substr(0, 1).toUpperCase() + key.substr(1);
130   if (this[getter] != null) {
131     return this[getter]();
132   }
133   return null;
134 };
135 
136 /**
137  * Sets a configuration property based on its name.
138  * @param {string} key The name of the configuration property.
139  * @param {string} newValue New value to set.
140  * @return {object} The old value or null if didn't exists.
141  */
142 google.ads.common.AppConfigBase.prototype.setConfigValue = function(
143     key, newValue) {
144   var oldValue;
145   var getter = 'get' + key.substr(0, 1).toUpperCase() + key.substr(1);
146   var setter = 'set' + key.substr(0, 1).toUpperCase() + key.substr(1);
147   if (this[getter] != null && this[setter] != null) {
148     oldValue = this[getter]();
149     this[setter](newValue);
150   }
151   return oldValue;
152 };
