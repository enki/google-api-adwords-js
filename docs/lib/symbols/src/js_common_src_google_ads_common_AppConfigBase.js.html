 19 /**
 20  * @fileoverview Defines AppConfigBase, the base class for all configuration
 21  *     readers.
 22  */
 23 
 24 goog.provide('google.ads.common.AppConfigBase');
 25 
 26 goog.require('google.system.net.HttpWebTransportSettings');
 27 
 28 /**
 29  * Reads the configuration keys from the configuration files.
 30  *
 31  * @constructor
 32  */
 33 google.ads.common.AppConfigBase = function() {
 34 
 35   /**
 36    * Proxy URL key constant.
 37    * @type {string}
 38    * @private
 39    */
 40   this.PROXY_URL_ = 'ProxyUrl';
 41 
 42   /**
 43    * Library short name to be attached to the user agent.
 44    * @type {string}
 45    * @private
 46    */
 47   this.shortName_ = null;
 48 
 49   /**
 50    * Web Transport key constant.
 51    * @type {string}
 52    * @private
 53    */
 54   this.HTTP_WEBTRANSPORT_ = 'HttpWebTransport';
 55 
 56   /**
 57    * Web transport class name.
 58    * @type {string}
 59    * @private
 60    */
 61   this.HTTPWebTransport_ = null;
 62 
 63   /**
 64    * Web Transport settings key constant.
 65    * @type {string}
 66    * @private
 67    */
 68   this.HTTP_WEBTRANSPORT_SETTINGS_ = 'HttpWebTransportSettings';
 69 
 70   /**
 71    * Web transport settings map.
 72    * @type {Object<String, String>}
 73    * @private
 74    */
 75   this.HTTPWebTransportSettings_ = null;
 76 };
 77 
 78 /**
 79  * Gets the HTTP web transport.
 80  *
 81  * @return {string} The HTTP web transport provider name.
 82  */
 83 google.ads.common.AppConfigBase.prototype.getHttpWebTransport = function() {
 84   return this.HTTPWebTransport_;
 85 };
 86 
 87 /**
 88  * Sets the HTTP web transport.
 89  *
 90  * @param {string} HTTPWebTransport The HTTP web transport provider name.
 91  */
 92 google.ads.common.AppConfigBase.prototype.setHttpWebTransport =
 93     function(HTTPWebTransport) {
 94   this.HTTPWebTransport_ = HTTPWebTransport;
 95 };
 96 
 97 /**
 98  * Gets the HTTP web transport settings.
 99  *
100  * @return {string} The HTTP web transport settings.
101  */
102 google.ads.common.AppConfigBase.prototype.getHttpWebTransportSettings =
103     function() {
104   return this.HTTPWebTransportSettings_;
105 };
106 
107 /**
108  * Sets the HTTP web transport.
109  *
110  * @param {string} HTTPWebTransportSettings The HTTP web transport settings.
111  */
112 google.ads.common.AppConfigBase.prototype.setHttpWebTransportSettings =
113     function(HTTPWebTransportSettings) {
114   this.HTTPWebTransportSettings_ = HTTPWebTransportSettings;
115 };
116 
117 /**
118  * Gets the short name for the library.
119  *
120  * @return {string} The short name for the library.
121  */
122 google.ads.common.AppConfigBase.prototype.getShortName = function() {
123   return this.shortName_;
124 };
125 
126 /**
127  * Gets a signature string for the library.
128  *
129  * @return {string} The signature string for the library.
130  */
131 google.ads.common.AppConfigBase.prototype.getSignature = function() {
132   return this.getShortName() + '-' + this.getVersion();
133 };
134 
135 /**
136  * Gets the library version.
137  *
138  * @return {string} The library version.
139  */
140 google.ads.common.AppConfigBase.prototype.getVersion = function() {
141   return this.getShortName() + '-' +
142       google.system.core.Runtime.getInstance().getVersion();
143 };
144 
145 /**
146  * Reads the settings from configuration file.
147  *
148  * @param {Array} settings Settings to be read and set.
149  */
150 google.ads.common.AppConfigBase.prototype.readSettings = function(
151     settings) {
152   this.setHttpWebTransport(settings[this.HTTP_WEBTRANSPORT_] ||
153       this.getHttpWebTransport());
154   this.setHttpWebTransportSettings(
155       new google.system.net.HttpWebTransportSettings(
156           settings[this.HTTP_WEBTRANSPORT_SETTINGS_] || {}));
157 };
158 
159 /**
160  * Retrieves a configuration value based on its name.
161  * @param {string} key The name of the configuration property.
162  * @return {?object} The value or null if does not exists.
163  */
164 google.ads.common.AppConfigBase.prototype.getConfigValue = function(key) {
165   var getter = 'get' + key.substr(0, 1).toUpperCase() + key.substr(1);
166   if (this[getter] != null) {
167     return this[getter]();
168   }
169   return null;
170 };
171 
172 /**
173  * Sets a configuration property based on its name.
174  * @param {string} key The name of the configuration property.
175  * @param {string} newValue New value to set.
176  * @return {object} The old value or null if didn't exists.
177  */
178 google.ads.common.AppConfigBase.prototype.setConfigValue = function(
179     key, newValue) {
180   var oldValue;
181   var getter = 'get' + key.substr(0, 1).toUpperCase() + key.substr(1);
182   var setter = 'set' + key.substr(0, 1).toUpperCase() + key.substr(1);
183   if (this[getter] != null && this[setter] != null) {
184     oldValue = this[getter]();
185     this[setter](newValue);
186   }
187   return oldValue;
188 };
