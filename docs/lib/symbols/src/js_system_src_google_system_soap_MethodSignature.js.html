 20 /**
 21  * @fileoverview Defines a SOAP method signature class.
 22  */
 23 
 24 goog.provide('google.system.soap.MethodSignature');
 25 
 26 /**
 27  * Represents a SOAP Service method.
 28  * Contains the metadata associated to a service method, like its XML namespace,
 29  * parameters, etc.
 30  *
 31  * @param {string} requestNamespace Xml namespace of the request of this
 32  * service method.
 33  * @param {string} responseNamespace Xml namespace of the response of this
 34  * service method.
 35  * @param {string} methodUse Use either literal or encoded.
 36  * @param {string} parameterStyle Either wrapped or non-wrapped.
 37  * @param {string} returnXmlElement Xml namespace of the response of this
 38  * service method.
 39  * @param {Object} returnTypeDefinition A definition of the return type as
 40  * follows {className: className, isArray: [true|false],
 41  * isSystem: [true|false]}.
 42  * @param {Array.<Object>} headers An array of header
 43  * definitions, every headers is defined as follows {direction: [in|out],
 44  * propertyName: propertyName}.
 45  * @param {Array.<Object>} parameters An array of parameter
 46  * definitions, every parameter is defined as follows {className: className,
 47  * isArray: [true|false], isSystem: [true|false], propertyName: name,
 48  * xmlElementName: xmlName}.
 49  * @constructor
 50  */
 51 google.system.soap.MethodSignature = function(requestNamespace,
 52     responseNamespace, methodUse, parameterStyle, returnXmlElement,
 53     returnTypeDefinition, headers, parameters) {
 54   this.requestNamespace_ = requestNamespace;
 55   this.responseNamespace_ = responseNamespace;
 56   this.use_ = methodUse;
 57   this.parameterStyle_ = parameterStyle;
 58   this.returnXmlElement_ = returnXmlElement;
 59   this.returnTypeDefinition_ = returnTypeDefinition;
 60   this.parameters_ = parameters || [];
 61   this.headers_ = headers || [];
 62 };
 63 
 64 /**
 65  * Returns the request namespace configured from the service method.
 66  *
 67  * @return {string} The request namespace of the method.
 68  */
 69 google.system.soap.MethodSignature.prototype.getRequestNamespace = function() {
 70   return this.requestNamespace_;
 71 };
 72 
 73 /**
 74  * Returns the response namespace configured from the service method.
 75  *
 76  * @return {string} The response namespace of the method.
 77  */
 78 google.system.soap.MethodSignature.prototype.getResponseNamespace = function() {
 79   return this.responseNamespace_;
 80 };
 81 
 82 /**
 83  * Returns the method use type configuration, either literal or encoded.
 84  *
 85  * @return {string} The method use.
 86  */
 87 google.system.soap.MethodSignature.prototype.getUse = function() {
 88   return this.use_;
 89 };
 90 
 91 /**
 92  * Returns the method parameter style, either wrapped or non-wrapped.

 94  * @return {string} The method parameter style.
 95  */
 96 google.system.soap.MethodSignature.prototype.getParameterStyle = function() {

 98 };
 99 
100 /**
101  * Returns the method parameter style, either wrapped or non-wrapped.
102  *
103  * @return {string} The method parameter style.
104  */
105 google.system.soap.MethodSignature.prototype.getReturnXmlElement = function() {

107 };
108 
109 /**
110  * Returns the method expected return type definition. A return type definition
111  * is of the form {className: className, isArray: [true|false],
112  * isSystem: [true|false]}.
113  *
114  * @return {Object} The return type definition.
115  */
116 google.system.soap.MethodSignature.prototype.getReturnTypeDefinition =
117     function() {
118   return this.returnTypeDefinition_;
119 };
120 
121 /**
122  * Retrieves a cloned array of this method headers.
123  *
124  * @return {Array.<Object>} An array of the service configured header
125  * definitions.
126  */
127 google.system.soap.MethodSignature.prototype.getHeaders = function() {
128   return goog.array.clone(this.headers_);
129 };
130 
131 /**
132  * Retrieves a cloned array of this method parameters.
133  *
134  * @return {Array.<Object>} An array of the method configured parameters.
135  */
136 google.system.soap.MethodSignature.prototype.getParameters = function() {
137   return goog.array.clone(this.parameters_);
138 };
139 
140 /**
141  * Retrieves a parameter by its name.
142  *
143  * @param {string} name Name of the requested parameter.
144  * @return {Array.<Object>} The requested parameter definition or null if does
145  * not exists.
146  */
147 google.system.soap.MethodSignature.prototype.getParameterByName = function(
148     name) {
149   for (var i = 0, parameter; parameter = this.parameters_[i]; i++) {
150     if (parameter.name == name) {
151       return parameter;
152     }
153   }
154 };
155 
156 /**
157  * Adds a header to the list of expected headers for this service method.
158  *
159  * @param {string} propertyName Property within the service that holds the
160  * header object.
161  * @param {?string} direction Direction of the header either 'in' for headers
162  * going to the server or 'out' for headers coming from the server. If not
163  * supplied, 'in' will be used.
164  */
165 google.system.soap.MethodSignature.prototype.addHeader = function(propertyName,
166     direction) {
167   this.headers_.push({direction: (direction || 'in'),
168       propertyName: propertyName});
169 };
170 
171 /**
172  * Add a parameter to the list of parameters expected by this service.
173  *
174  * @param {string} className Full class name of the parameter.
175  * @param {boolean} isArray True if it is expected to be an array, false
176  * otherwise.
177  * @param {boolean} isSystem True if it is system kind of property, false
178  * otherwise.
179  * @param {string} name Parameter name.
180  * @param {string} xmlElementName Name of XML how to serialize this property.
181  */
182 google.system.soap.MethodSignature.prototype.addParameter = function(className,
183     isArray, isSystem, name, xmlElementName) {
184   this.parameters_.push({className: className, isArray: isArray,
185       isSystem: isSystem, name: name, xmlElementName: xmlElementName});
186 };
