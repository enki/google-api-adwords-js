 19 /**
 20  * @fileoverview Defines AuthTokenException, an exception class that represents
 21  * an exception thrown by ClientLogin API.
 22  */
 23 
 24 goog.provide('google.ads.common.AuthTokenException');
 25 
 26 goog.require('google.ads.common.AdsException');
 27 
 28 /**
 29  * An exception class that represents an exception thrown by ClientLogin API.
 30  *
 31  * @extends google.ads.common.AdsException
 32  * @constructor
 33  * @param {number} errorCode The error code associated with this Auth Exception.
 34  * @param {string} errorUrl The url that describes this error.
 35  * @param {string} captchaToken The token associated with the captcha, if a
 36  * Require Captcha error is triggered.
 37  * @param {string} captchaUrl The url for the captcha. If Require Captcha error
 38  * is triggered, then this url should be presented to the users to unlock their
 39  * accounts.
 40  * @param {string?} message An optional error message for the exception.
 41  * @param {Object?} innerException If this exception was thrown in response to
 42  * another exception, then you can wrap the original exception within this
 43  * exception.
 44  */
 45 google.ads.common.AuthTokenException = function(errorCode,
 46     errorUrl, captchaToken, captchaUrl, message, innerException) {
 47   google.ads.common.AdsException.call(this, message, innerException);
 48   this.errorCode_ = errorCode;
 49   this.errorUrl_ = errorUrl;
 50   this.captchaToken_ = captchaToken;
 51   this.captchaUrl_ = captchaUrl;
 52 };
 53 goog.inherits(google.ads.common.AuthTokenException,
 54     google.ads.common.AdsException);
 55
