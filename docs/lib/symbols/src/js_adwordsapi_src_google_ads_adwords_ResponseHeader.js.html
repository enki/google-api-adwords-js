 29 /**
 30  * Soap Response header for AdWords API services.
 31  *
 32  * @param {string} version Version of the API.
 33  * @constructor
 34  */

 36   google.system.soap.Object.call(this);
 37   this.version_ = version;
 38 };
 39 var type = google.system.soap.ObjectType.defineType(null,
 40     'google.ads.adwords.ResponseHeader', false);

 44 type.addProperty('responseTime', 'integer', 'responseTime', false, true);
 45 type.addProperty('units', 'integer', 'units', false, true);
