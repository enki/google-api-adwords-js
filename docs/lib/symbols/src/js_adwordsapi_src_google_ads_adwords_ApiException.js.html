 20 /**
 21  * @fileoverview Defines ApiException, a custom exception class for
 22  *     wrapping AdWords API SOAP exceptions.
 23  */
 24 
 25 goog.provide('google.ads.adwords.ApiException');
 26 
 27 goog.require('google.ads.adwords.Exception');
 28 
 29 /**
 30  * A custom exception class for wrapping AdWords API SOAP exceptions.
 31  *
 32  * @extends google.ads.adwords.Exception
 33  * @constructor
 34  * @param {Object} apiException The original ApiException object from
 35  *     AdWords API.
 36  * @param {string?} message An optional error message for the exception.
 37  * @param {Object?} innerException If this exception was thrown in response to
 38  *     another exception, then you can wrap the original exception within this
 39  *     exception.
 40  */
 41 google.ads.adwords.ApiException = function(apiException, message,
 42     innerException) {
 43   google.ads.adwords.Exception.call(this, message, innerException);
 44 
 45   /**
 46    * Internal API exception.
 47    * @type {Object}
 48    */
 49   this.apiException = apiException;
 50 };
 51 goog.inherits(google.ads.adwords.ApiException, google.ads.adwords.Exception);
 52 
 53 /**
 54  * Gets the ApiException object.
 55  *
 56  * @return {Object} The ApiException object from AdWords API.
 57  */
 58 google.ads.adwords.ApiException.prototype.getApiException = function() {
 59   return this.apiException;
 60 };
