 20 /**
 21  * @fileoverview Defines a SOAP object property class.
 22  */
 23 
 24 goog.provide('google.system.soap.ObjectProperty');
 25 
 26 /**
 27  * Represents a SOAP Object Property.
 28  * Contains that metadata that conforms a SOAP Object property, like its
 29  * name, xml element names mapping, etc.
 30  *
 31  * @param {string} name Name of the property.
 32  * @param {string} fullClassName Full class name of the property.
 33  * @param {(string|Object.<string, string>)} xmlElementNameMapping In general
 34  * this maps to a simple xmlElement name unless, it is a choice type of SOAP
 35  * element where multiple xml element names can be defined depending on the
 36  * type of the value to be serialized, so it can take the form of
 37  * [{elementName: 'element', className: 'fullClassName'}, ...].
 38  * @param {boolean} isArray True if is an array type of property, false
 39  * otherwise.
 40  * @param {boolean} isSystem True if is a system kind of property, false
 41  * otherwise.
 42  * @constructor
 43  */
 44 google.system.soap.ObjectProperty = function(name, fullClassName,
 45     xmlElementNameMapping, isArray, isSystem) {
 46   this.name_ = name;
 47   this.fullClassName_ = fullClassName;
 48   this.xmlElementNameMapping_ = xmlElementNameMapping;
 49   this.isArray_ = isArray;
 50   this.isSystem_ = isSystem;
 51 };
 52 
 53 /**
 54  * Retrieves the name of the property.
 55  *
 56  * @return {string} Name of the property.
 57  */
 58 google.system.soap.ObjectProperty.prototype.getName = function() {
 59   return this.name_;
 60 };
 61 
 62 /**
 63  * Retrieves the full class name of the property.
 64  *
 65  * @return {string} Full class name.
 66  */
 67 google.system.soap.ObjectProperty.prototype.getFullClassName = function() {
 68   return this.fullClassName_;
 69 };
 70 
 71 /**
 72  * Retrieves the {google.system.soap.ObjectType} of the property.
 73  *
 74  * @return {google.system.soap.ObjectType} The property type.
 75  */
 76 google.system.soap.ObjectProperty.prototype.getType = function() {
 77   return goog.getObjectByName(this.fullClassName_).prototype.getType();
 78 };
 79 
 80 /**
 81  * Retrieves the property XML element name mapping.
 82  *
 83  * @return {(string|Array.<Object>)} The property XML element name mapping.
 84  */
 85 google.system.soap.ObjectProperty.prototype.getXmlElementNameMapping =
 86     function() {
 87   return this.xmlElementNameMapping_;
 88 };
 89 
 90 /**
 91  * Retrieves if the property if of array type.
 92  *
 93  * @return {boolean} True if is of an array type, false otherwise.
 94  */
 95 google.system.soap.ObjectProperty.prototype.isArray = function() {
 96   return this.isArray_;
 97 };
 98 
 99 /**
100  * Retrieves if the property if of system type.
101  *
102  * @return {boolean} True if is of an system type, false otherwise.
103  */
104 google.system.soap.ObjectProperty.prototype.isSystem = function() {
105   return this.isSystem_;
106 };
107 
108 /**
109  * Retrieves the getter name of the property. Follows the code style conventions
110  * to construct the getter name, E.g. a property named test will have its getter
111  * named getTest.
112  *
113  * @return {string} Name of the getter function.
114  */
115 google.system.soap.ObjectProperty.prototype.getGetterName = function() {
116   // the property named type is reserved.
117   var variableName = (this.name_ === 'type') ? 'type$' : this.name_;
118 
119   return 'get' + variableName.substring(0, 1).toUpperCase() +
120       variableName.substring(1, variableName.length);
121 };
122 
123 /**
124  * Retrieves the setter name of the property. Follows the code style conventions
125  * to construct the setter name, E.g. a property named test will have its setter
126  * named setTest.
127  *
128  * @return {string} Name of the setter function.
129  */
130 google.system.soap.ObjectProperty.prototype.getSetterName = function() {
131   // the property named type is reserved.
132   var variableName = (this.name_ === 'type') ? 'type$' : this.name_;
133 
134   return 'set' + variableName.substring(0, 1).toUpperCase() +
135       variableName.substring(1, variableName.length);
136 };
