 20 /**
 21  * @fileoverview Defines a SOAP object property class.
 22  */
 23 
 24 goog.provide('google.system.soap.ObjectProperty');
 25 
 26 /**
 27  * Represents a SOAP Object Property.
 28  * Contains that metadata that conforms a SOAP Object property, like its
 29  * name, xml element names mapping, etc.
 30  *
 31  * @param {string} name Name of the property.
 32  * @param {string} fullClassName Full class name of the property.
 33  * @param {(string|Object.<string, string>)} xmlElementNameMapping In general
 34  *     this maps to a simple xmlElement name unless, it is a choice type of SOAP
 35  *     element where multiple xml element names can be defined depending on the
 36  *     type of the value to be serialized, so it can take the form of
 37  *     [{elementName: 'element', className: 'fullClassName'}, ...].
 38  * @param {boolean} isArray True if is an array type of property, false
 39  *     otherwise.
 40  * @param {boolean} isSystem True if is a system kind of property, false
 41  *     otherwise.
 42  * @constructor
 43  */
 44 google.system.soap.ObjectProperty = function(name, fullClassName,
 45     xmlElementNameMapping, isArray, isSystem) {
 46 
 47   /**
 48    * Property name.
 49    * @type {string}
 50    * @private
 51    */
 52   this.name_ = name;
 53 
 54   /**
 55    * Property full class name.
 56    * @type {string}
 57    * @private
 58    */
 59   this.fullClassName_ = fullClassName;
 60 
 61   /**
 62    * XML element <-> property mapping.
 63    * @type {string|Array.<Object>}
 64    * @private
 65    */
 66   this.xmlElementNameMapping_ = xmlElementNameMapping;
 67 
 68   /**
 69    * True if the property if of the type array, false otherwise.
 70    * @type {boolean}
 71    * @private
 72    */
 73   this.isArray_ = isArray;
 74 
 75   /**
 76    * True if the property if of the type system (string, long, etc), false
 77    * otherwise.
 78    * @type {boolean}
 79    * @private
 80    */
 81   this.isSystem_ = isSystem;
 82 };
 83 
 84 /**
 85  * Retrieves the name of the property.
 86  *
 87  * @return {string} Name of the property.
 88  */
 89 google.system.soap.ObjectProperty.prototype.getName = function() {
 90   return this.name_;
 91 };
 92 
 93 /**
 94  * Retrieves the full class name of the property.
 95  *
 96  * @return {string} Full class name.
 97  */
 98 google.system.soap.ObjectProperty.prototype.getFullClassName = function() {
 99   return this.fullClassName_;
100 };
101 
102 /**
103  * Retrieves the {google.system.soap.ObjectType} of the property.
104  *
105  * @return {google.system.soap.ObjectType} The property type.
106  */
107 google.system.soap.ObjectProperty.prototype.getType = function() {
108   return goog.getObjectByName(this.fullClassName_).prototype.getType();
109 };
110 
111 /**
112  * Retrieves the property XML element name mapping.
113  *
114  * @return {(string|Array.<Object>)} The property XML element name mapping.
115  */
116 google.system.soap.ObjectProperty.prototype.getXmlElementNameMapping =
117     function() {
118   return this.xmlElementNameMapping_;
119 };
120 
121 /**
122  * Retrieves if the property if of array type.
123  *
124  * @return {boolean} True if is of an array type, false otherwise.
125  */
126 google.system.soap.ObjectProperty.prototype.isArray = function() {
127   return this.isArray_;
128 };
129 
130 /**
131  * Retrieves if the property if of system type.
132  *
133  * @return {boolean} True if is of an system type, false otherwise.
134  */
135 google.system.soap.ObjectProperty.prototype.isSystem = function() {
136   return this.isSystem_;
137 };
138 
139 /**
140  * Retrieves the getter name of the property. Follows the code style conventions
141  * to construct the getter name, E.g. a property named test will have its getter
142  * named getTest.
143  *
144  * @return {string} Name of the getter function.
145  */
146 google.system.soap.ObjectProperty.prototype.getGetterName = function() {
147   // the property named type is reserved.
148   var variableName = (this.name_ === 'type') ? 'type$' : this.name_;
149 
150   return 'get' + variableName.substring(0, 1).toUpperCase() +
151       variableName.substring(1, variableName.length);
152 };
153 
154 /**
155  * Retrieves the setter name of the property. Follows the code style conventions
156  * to construct the setter name, E.g. a property named test will have its setter
157  * named setTest.
158  *
159  * @return {string} Name of the setter function.
160  */
161 google.system.soap.ObjectProperty.prototype.getSetterName = function() {
162   // the property named type is reserved.
163   var variableName = (this.name_ === 'type') ? 'type$' : this.name_;
164 
165   return 'set' + variableName.substring(0, 1).toUpperCase() +
166       variableName.substring(1, variableName.length);
167 };
