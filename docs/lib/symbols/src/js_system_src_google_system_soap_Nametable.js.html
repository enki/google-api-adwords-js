 20 /**
 21  * @fileoverview Defines a google.system.soap.Nametable class.
 22  */
 23 
 24 goog.provide('google.system.soap.Nametable');
 25 
 26 /**
 27  * Creates a new name table that holds prefixes and namespaces for a given
 28  * services.
 29  * @constructor
 30  */
 31 google.system.soap.Nametable = function() {
 32   this.xmlnt_ = {
 33     'soap': 'http://schemas.xmlsoap.org/soap/envelope/',
 34     'xsi': 'http://www.w3.org/2001/XMLSchema-instance',
 35     'xsd': 'http://www.w3.org/2001/XMLSchema'
 36   };
 37   this.localxmlnt_ = {};
 38 };
 39 
 40 /**
 41  * Retrieves the list of local namespaces.
 42  * @return {Array.<Object>} An array of all registered local namespaces.
 43  */
 44 google.system.soap.Nametable.prototype.getLocalNamespaces = function() {
 45   return this.localxmlnt_;
 46 };
 47 
 48 /**
 49  * Returns a SOAP XML namespace given its prefix.
 50  * @param {string} prefix Prefix of the namespace.
 51  * @return {string} Namespace.
 52  */
 53 google.system.soap.Nametable.prototype.getNamespaceFromPrefix = function(
 54     prefix) {
 55   var retVal = this.xmlnt_[prefix];
 56   if (retVal == null) {
 57     retVal = this.localxmlnt_[prefix];
 58   }
 59   return retVal;
 60 };
 61 
 62 /**
 63  * Retrieves the prefix of a given SOAP XML namespace.
 64  * @param {string} namespace Namespace.
 65  * @return {string|null} Its prefix or null if not found.
 66  */
 67 google.system.soap.Nametable.prototype.getPrefixFromNamespace = function(
 68     namespace) {
 69   for (var prefix in this.xmlnt_) {
 70     if (this.xmlnt_[prefix] == namespace) {
 71       return prefix;
 72     }
 73   }
 74 
 75   for (var prefix in this.localxmlnt_) {
 76     if (this.localxmlnt_[prefix] == namespace) {
 77       return prefix;
 78     }
 79   }
 80   return null;
 81 };
