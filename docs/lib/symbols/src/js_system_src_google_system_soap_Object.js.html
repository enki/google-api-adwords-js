 53 /**
 54  * Method that de-serializes a SOAP object.
 55  * @param {Node} node Node that contains the object data.
 56  * @param {google.system.soap.Nametable} xmlnt Namespace table.
 57  */
 58 google.system.soap.Object.prototype.deserialize = function(node, xmlnt) {
 59   for (var i = 0; i < node.childNodes.length; i++) {
 60     var nodeName = this.getNodeName_(node.childNodes[i]);
 61     var propertyName = this.getDeserializationFieldName_(nodeName);
 62     if (!propertyName) {
 63       continue;
 64     }
 65     var availableType = this.getTypeAsPerXsiType_(xmlnt, node.childNodes[i],
 66         this.getType().getNamespace());
 67     if (!goog.isNull(availableType)) {
 68       availableType = goog.getObjectByName(availableType);
 69       if (!goog.isNull(availableType)) {
 70         availableType = new availableType();
 71       }
 72     } else {
 73       var property = this.getType().getPropertyByName(propertyName);
 74       if (!property.isSystem()) {
 75         var propertyFullClassName = property.getFullClassName();
 76         availableType = goog.getObjectByName(propertyFullClassName);
 77         if (!goog.isNull(availableType)) {
 78           availableType = new availableType();
 79         }
 80       }
 81     }
 82 
 83     var propertyValue = null;
 84     if (!goog.isNull(availableType) && !availableType.getType().isEnum()) {
 85       availableType.deserialize(node.childNodes[i], xmlnt);
 86       propertyValue = availableType;
 87     } else {
 88       // we have two possibilities - either we don't know the type of this
 89       // property, or this is a system type. In either case, trying to parse
 90       // as json with a fallback is a safe bet.
 91       try {
 92         propertyValue = goog.json.parse(goog.dom.getTextContent(
 93             node.childNodes[i]));
 94       } catch (err) {
 95         propertyValue = goog.dom.getTextContent(node.childNodes[i]);
 96       }
 97     }
 98 
 99     var property = this.getType().getPropertyByName(propertyName);
100     var currentPropertyValue = this[property.getGetterName()](this);
101     if (property.isArray()) {
102       if (currentPropertyValue == null) {
103         currentPropertyValue = [];
104       }
105       currentPropertyValue.push(propertyValue);
106     } else {
107       currentPropertyValue = propertyValue;
108     }
109     this[property.getSetterName()](currentPropertyValue);
110   }
111 };

113 /**
114  * Method that serializes a SOAP Object into XML.
115  *
116  * @param {Document} doc Current XML document to append.
117  * @param {google.system.soap.Nametable} xmlnt Namespace table.
118  * @param {Node} parentNode Root of the XML document.
119  * @param {string} parentNamespace Current parent namespace.
120  */
121 google.system.soap.Object.prototype.serialize = function(
122     doc, xmlnt, parentNode, parentNamespace) {
123   var defaultNs = this.getSerializationNamespace_(parentNamespace);
124   var nodePrefix = xmlnt.getPrefixFromNamespace(defaultNs);
125   var propertyName = null;
126   var allProperties = this.getType().getProperties();
127   for (var i = 0; i < allProperties.length; i++) {
128     var property = allProperties[i];
129 
130     var propertyValue = this[property.getGetterName()](this);
131     if (propertyValue == null) {
132       continue;
133     }
134 
135     if (property.isArray()) {
136       for (var j = 0; j < propertyValue.length; j++) {
137         var childNode = this.serializeProperty_(doc, xmlnt, defaultNs,
138             nodePrefix, property.getName(), propertyValue[j]);
139         parentNode.appendChild(childNode);
140       }
141     } else {
142       var childNode = this.serializeProperty_(doc, xmlnt, defaultNs, nodePrefix,
143           property.getName(), propertyValue);
144       parentNode.appendChild(childNode);
145     }
146   }
147 };

151 /**
152  * Retrieves the node name for a given a property within an object.
153  *
154  * @param {string} propertyName Property name.
155  * @param {*} propertyValue Property value.
156  * @return {string} The node name.
157  * @private
158  */
159 google.system.soap.Object.prototype.getSerializationNodeNameForProperty_ =
160     function(propertyName, propertyValue) {
161   var property = this.getType().getPropertyByName(propertyName);
162   var xmlElementsMapping = property.getXmlElementNameMapping();
163 
164   if (!goog.isArray(xmlElementsMapping)) {
165     return xmlElementsMapping;
166   }
167 
168   for (var i = 0, elementMapping;
169       elementMapping = xmlElementsMapping[i]; i++) {
170     if (!goog.isNull(elementMapping.elementName) &&
171         !goog.isNull(elementMapping.className)) {
172       var desiredType = goog.getObjectByName(elementMapping.className);
173       if (!goog.isNull(desiredType)) {
174         if (propertyValue instanceof desiredType) {
175           return elementMapping.elementName;
176         }
177       }
178     } else {
179       return elementMapping.elementName;
180     }
181   }
182   return propertyName;
183 };

185 /**
186  * Serializes a SOAP Object property.
187  *
188  * @param {Document} doc The XML document over which is been serialized.
189  * @param {google.system.soap.SoapNametable} xmlnt XML namespace table.
190  * @param {string} namespace Namespace.
191  * @param {string} prefix Namespace prefix.
192  * @param {string} name Element name.
193  * @param {string} value Element value.
194  * @return {Node} The new node.
195  * @private
196  */
197 google.system.soap.Object.prototype.serializeProperty_ = function(
198     doc, xmlnt, namespace, prefix, name, value) {
199   var nodeName = this.getSerializationNodeNameForProperty_(name, value);
200   var node = doc.createElement(prefix + ':' + nodeName);
201   if (value instanceof google.system.soap.Object) {
202     if (!goog.isNull(value.getType().getBaseType())) {
203       // The prefix for the object may not be the same as prefix of the node.
204       // e.g.
205       // <o:targets xsi:type="cm:CountryTarget">
206       //   <cm:countryCode>US</cm:countryCode>
207       // </o:targets>
208       var typePrefix = xmlnt.getPrefixFromNamespace(
209           value.getSerializationNamespace_(namespace));
210 
211       node.setAttribute('xsi:type', typePrefix + ':' +
212           value.getType().getClassName());
213     }
214     value.serialize(doc, xmlnt, node, namespace);
215   } else {
216     goog.dom.setTextContent(node, value);
217   }
218   return node;
219 };

221 /**
222  * Retrieves the xsi prefix for an XML node.
223  *
224  * @param {Node} childNode The node to get its xsi type.
225  * @param {google.system.soap.Nametable} xmlnt The namespace table.
226  * @return {string} The prefix if any or empty string.
227  * @private
228  */
229 google.system.soap.Object.prototype.getXsiPrefixForNode_ = function(
230     childNode, xmlnt) {
231   var node = childNode;
232   var requiredNs = xmlnt.getNamespaceFromPrefix('xsi');
233 
234   while (this.getNodeName_(node) != null) {
235     var foundNs = false;
236     var xsiPrefix;
237     for (var i = 0; i < node.attributes.length; i++) {
238       if (node.attributes[i].prefix == 'xmlns') {
239         if (node.attributes[i].nodeValue == requiredNs) {
240           xsiPrefix = this.getNodeName_(node.attributes[i]) + ':';
241           foundNs = true;
242           break;
243         }
244       } else if (node.attributes[i].prefix == null &&
245           this.getNodeName_(node.attributes[i]) == 'xmlns') {
246         if (node.attributes[i].nodeValue == requiredNs) {
247           xsiPrefix = '';
248           foundNs = true;
249           break;
250         }
251       }
252     }
253     if (foundNs) {
254       return xsiPrefix;
255     } else {
256       node = node.parentNode;
257     }
258   }
259   return null;
260 };

262 /**
263  * Returns an SOAP Object type given its XML xsi type.
264  *
265  * @param {google.system.soap.SoapNametable} xmlnt XML namespace table.
266  * @param {Node} childNode Child node containing the xsi type.
267  * @param {string} namespace Element namespace.
268  * @return {google.system.code.Type} The type of the object.
269  * @private
270  */
271 google.system.soap.Object.prototype.getTypeAsPerXsiType_ = function(
272     xmlnt, childNode, namespace) {
273   var xsiPrefix = '';
274   var xsiType = null;
275 
276   var xsiPrefix = this.getXsiPrefixForNode_(childNode, xmlnt);
277   if (goog.isNull(xsiPrefix)) {
278     // we walked all the way to the root without finding any xsi namespace.
279     return null;
280   }
281   var xsiTypeNodeName = xsiPrefix + 'type';
282 
283   for (var i = 0; i < childNode.attributes.length; i++) {
284     if (childNode.attributes[i].nodeName == xsiTypeNodeName) {
285       xsiType = childNode.attributes[i].nodeValue;
286       break;
287     }
288   }
289 
290   if (!goog.isNull(xsiType)) {
291     var splits = xsiType.split(':');
292     if (splits.length == 2) {
293       xsiType = (namespace == '') ? splits[1] : namespace + '.' + splits[1];
294     } else {
295       xsiType = (namespace == '') ? splits[0] : namespace + '.' + splits[0];
296     }
297   }
298   return xsiType;
299 };

301 /**
302  * Retrieves the serialization namespace of a given SOAP object.
303  *
304  * @param {string} currentNamespace Current namespace.
305  * @return {string} Object serialization namespace.
306  * @private
307  */
308 google.system.soap.Object.prototype.getSerializationNamespace_ = function(
309     currentNamespace) {
310   var xmlns = this.getType().getXmlNamespace();
311   return xmlns || currentNamespace;
312 };

314 /**
315  * Retrieves the SOAP object property name of a given nodeName.
316  *
317  * @param {string} nodeName Name of the node.
318  * @return {string} The object property name.
319  * @private
320  */
321 google.system.soap.Object.prototype.getDeserializationFieldName_ = function(
322     nodeName) {
323   var allProperties = this.getType().getProperties();
324   for (var i = 0; i < allProperties.length; i++) {
325     var propertyName = allProperties[i].getName();
326     var xmlElementMapping = allProperties[i].getXmlElementNameMapping();
327     if (goog.isArray(xmlElementMapping)) {
328       for (var j = 0; j < xmlElementMapping.length; j++) {
329         if (xmlElementMapping[j].elementName == nodeName) {
330           return propertyName;
331         }
332       }
333     } else if (xmlElementMapping == nodeName || propertyName == nodeName) {
334       return propertyName;
335     }
336   }
337   return null;
338 };

340 /**
341  * Retrieves an xml node's name in a browser-independent manner.
342  *
343  * @param {Node} node An XML node.
344  * @return {string} The name of the node.
345  * @private
346  */
347 google.system.soap.Object.prototype.getNodeName_ = function(node) {
348   // IE stores nodename in node.baseName, whereas Firefox, Chrome and related
349   // browsers store them in node.localName.
350   var nodeName = node.localName || node.baseName;
351   if (nodeName == '') {
352     return null;
353   } else {
354     return nodeName;
355   }
356 };
