 36 /**
 37  * A common base class for all SOAP services.
 38  *
 39  * Contains the serialization logic to transform request into SOAP XML
 40  * envelopes and then send them through the network.
 41  *
 42  * @extends google.system.soap.Object
 43  * @constructor
 44  */
 45 google.system.soap.Service = function() {
 46   google.system.soap.Object.call(this);
 47   this.nametable_ = new google.system.soap.Nametable();
 48   this.url_ = null;
 49   this.httpWebTransport_ = null;
 50   this.listeners_ = new Array();
 51 };
 52 goog.inherits(google.system.soap.Service, google.system.soap.Object);

101 /**
102  * Invokes a method of the service. Kicks of the whole serialization, send and
103  * de-serialization call life-cycle.
104  *
105  * @param {string} methodName Name of the method in the service that is been
106  * invoked.
107  * @param {Object.<String,Object>} parameterValues Parameters passed to the
108  * method.
109  * @param {function(google.system.soap.Object)} successCallback Callback
110  * function to be invoked if the request finalizes successfully, the
111  * de-serialized SOAP Object gets passed to the callback.
112  * @param {function(google.system.soap.Object)} failureCallback Callback
113  * function to be invoked if the request finalizes with an error, the
114  * de-serialized SOAP Exception gets passed to the callback.
115  */
116 google.system.soap.Service.prototype.invoke = function(methodName,
117     parameterValues, successCallback, failureCallback) {
118   var service = this;
119   // Before serialize listener phase.
120   var fnFactory = function(listener) {
121     return goog.partial(listener.beforeSerialize, service, methodName,
122         parameterValues);
123   };
124   var serializeRequestFn = goog.bind(this.serializeRequest_, this, methodName,
125       parameterValues, successCallback, failureCallback);
126   this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
127       serializeRequestFn, failureCallback);
128 };
129 
130 /**
131  * Registers a new SOAP listener to the service.
132  *
133  * @param {google.system.soap.ServiceListener} listener Listener to be
134  * registered.
135  */
136 google.system.soap.Service.prototype.registerListener = function(
137     listener) {
138   if (!(listener instanceof google.system.soap.ServiceListener)) {
139     throw new google.system.core.Exception(
140         'Tried to registered an invalid listener type.');
141   }
142   this.listeners_.push(listener);
143 };

162 /**
163  * Phase when the request is serialized and the listener are called for the
164  * beforeSend event. After all listeners are invoked sendRequest_ phase will
165  * continue.
166  *
167  * @param {string} methodName Name of the method in the service that is been
168  * invoked.
169  * @param {Object.<String,Object>} parameterValues Parameters passed to the
170  * method.
171  * @param {function(google.system.soap.Object)} successCallback Callback
172  * function to be invoked if the request finalizes successfully, the
173  * de-serialized SOAP Object gets passed to the function.
174  * @param {function(google.system.soap.Object)} failureCallback Callback
175  * function to be invoked if the request finalizes with an error, the
176  * de-serialized SOAP Exception gets passed to the function.
177  * @private
178  */
179 google.system.soap.Service.prototype.serializeRequest_ = function(
180     methodName, parameterValues, successCallback, failureCallback) {
181   var service = this;
182 
183   // Object serialization.
184   var methodSignature = this.getType().getMethodByName(methodName);
185   var doc = this.createSoapRequest_(
186       methodName, methodSignature.getParameters(), parameterValues);
187 
188   // Before send listener phase.
189   var fnFactory = function(listener) {
190     return goog.partial(listener.beforeSend, service, doc);
191   };
192   var sendRequestFn = goog.bind(this.sendRequest_, this, methodName, doc,
193       successCallback, failureCallback);
194   this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
195       sendRequestFn, failureCallback);
196 };
197 
198 /**
199  * Phase when the request gets send to the server and the listeners are called
200  * for the beforeDeserialize event. After all listeners are invoked
201  * deserializeResponse_ phase will continue.
202  *
203  * @param {string} methodName Name of the method in the service that is been
204  * invoked.
205  * @param {Document} doc The generated XML gocument.
206  * @param {function(google.system.soap.Object)} successCallback Callback
207  * function to be invoked if the request finalizes successfully, the
208  * de-serialized SOAP Object gets passed to the function.
209  * @param {function(google.system.soap.Object)} failureCallback Callback
210  * function to be invoked if the request finalizes with an error, the
211  * de-serialized SOAP Exception gets passed to the function.
212  * @private
213  */
214 google.system.soap.Service.prototype.sendRequest_ = function(
215     methodName, doc, successCallback, failureCallback) {
216   var service = this;
217 
218   // Serialize to string.
219   var serializedDoc = goog.dom.xml.serialize(doc);
220   this.postXmlToServer_(
221     serializedDoc,
222     function(responseXml) {
223       var doc = goog.dom.xml.loadXml(responseXml);
224       // Before de-serialize listener phase.
225       var fnFactory = function(listener) {
226         return goog.partial(listener.beforeDeserialize, service, doc, false);
227       };
228       var deserializeResponseFn = goog.bind(this.deserializeResponse_, this,
229           methodName, doc, successCallback, failureCallback);
230       this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
231           deserializeResponseFn, failureCallback);
232     },
233     function(errorXml) {
234       var doc = goog.dom.xml.loadXml(errorXml);
235       // Before de-serialize listener phase.
236       var fnFactory = function(listener) {
237         return goog.partial(listener.beforeDeserialize, service, doc, true);
238       };
239       var deserializeErrorFn = goog.bind(this.deserializeError_, this, doc,
240           successCallback, failureCallback);
241       this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
242           deserializeErrorFn, failureCallback);
243     }
244   );
245 };

247 /**
248  * Phase when a successful response gets deserialized and the listeners are
249  * called for the beforeReturn event. After all listeners are invoked the client
250  * sucessCallback gets invoked.
251  *
252  * @param {string} methodName Name of the method in the service that is been
253  * invoked.
254  * @param {Document} responseDocument The response XML document.
255  * @param {function(google.system.soap.Object)} successCallback Callback
256  * function to be invoked if the request finalizes successfully, the
257  * de-serialized SOAP Object gets passed to the function.
258  * @param {function(google.system.soap.Object)} failureCallback Callback
259  * function to be invoked if the request finalizes with an error, the
260  * de-serialized SOAP Exception gets passed to the function.
261  * @private
262  */
263 google.system.soap.Service.prototype.deserializeResponse_ = function(
264     methodName, responseDocument, successCallback, failureCallback) {
265   var service = this;
266   var methodSignature = this.getType().getMethodByName(methodName);
267 
268   // De-serializing the response.
269   var soapObject = this.parseSoapResponse_(responseDocument, methodName,
270       methodSignature.getReturnTypeDefinition());
271   // Function to be called after all listeners are notified.
272   var returnResponseFn = function() {
273     if (successCallback != null) {
274       successCallback.call(service, soapObject);
275     }
276   };
277   // Before return listener phase.
278   var fnFactory = function(listener) {
279     return goog.partial(listener.beforeReturn, service, soapObject,
280         false);
281   };
282   this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
283       returnResponseFn, failureCallback);
284 };
285 
286 /**
287  * Phase when an error response gets deserialized and the listeners are
288  * called for the beforeReturn event. After all listeners are invoked the client
289  * failureCallback gets invoked.
290  *
291  * @param {Document} errorDocument The response XML error document.
292  * @param {function(google.system.soap.Object)} successCallback Callback
293  * function to be invoked if the request finalizes successfully, the
294  * de-serialized SOAP Object gets passed to the function.
295  * @param {function(google.system.soap.Object)} failureCallback Callback
296  * function to be invoked if the request finalizes with an error, the
297  * de-serialized SOAP Exception gets passed to the function.
298  * @private
299  */
300 google.system.soap.Service.prototype.deserializeError_ = function(
301     errorDocument, successCallback, failureCallback) {
302   var service = this;
303   var errorObject = this.parseSoapFault_(errorDocument);
304   var returnResponseFn = function() {
305     if (failureCallback != null) {
306       failureCallback.call(service, errorObject);
307     }
308   };
309   // Before return listener phase.
310   var fnFactory = function(listener) {
311     return goog.partial(listener.beforeReturn, service, errorObject, true);
312   };
313   this.fireListeners_(goog.array.clone(this.listeners_), fnFactory,
314       returnResponseFn, failureCallback);
315 };

319 /**
320  * Generic method in charge of calling all listeners in certain phase of a
321  * service call life-cycle.
322  *
323  * @param {Array.<google.system.soap.ServiceListener>} listeners List of
324  * listeners pending to process.
325  * @param {function(google.system.soap.ServiceListener)} fnFactory Function
326  * factory that generates the function to be called in the listener. E.g.
327  * When on the beforeSend phase this factory generates a function that calls
328  * the beforeSend method of the listener.
329  * @param {function()} continueCallback Function to be called after all listener
330  * have been successfully called.
331  * @param {function(google.system.core.Exception)} failureCallback Function to
332  * be called in case one of the listeners fails.
333  * @private
334  */
335 google.system.soap.Service.prototype.fireListeners_ = function(
336     listeners, fnFactory, continueCallback, failureCallback) {
337   if (listeners.length) {
338     var listener = listeners.pop();
339     var continueFn = goog.bind(this.fireListeners_, this, listeners,
340         fnFactory, continueCallback, failureCallback);
341     var listenerErrorFn = goog.bind(this.listenerError_, this, listener,
342         failureCallback);
343     fnFactory(listener).call(listener, continueFn, listenerErrorFn);
344   } else {
345     continueCallback();
346   }
347 };
348 
349 /**
350  * Function called when a listener fails, it simply throws the exception up
351  * in the stack.
352  *
353  * @param {google.system.soap.ServiceListener} listener Listener that caused
354  * the failure.
355  * @param {function(google.system.core.Exception)} failureCallback Function to
356  * be called in case one of the listeners fails.
357  * @param {google.system.core.Exception} exception Exception returned by the
358  * listener.
359  * @private
360  */
361 google.system.soap.Service.prototype.listenerError_ = function(
362     listener, failureCallback, exception) {
363   if (failureCallback) {
364     failureCallback(exception);
365   }
366   throw exception;
367 };

371 /**
372  * Creates a SOAP request document.
373  * @param {string} methodName Method name.
374  * @param {Array.<Object>} parameters Parameters.
375  * @param {Array.<*>} parameterValues Values.
376  * @return {Document} An XML document.
377  * @private
378  */
379 google.system.soap.Service.prototype.createSoapRequest_ = function(
380     methodName, parameters, parameterValues) {
381   var doc = goog.dom.xml.createDocument();
382   var envelope = this.createEmptyEnvelope_(doc);
383   doc.appendChild(envelope);
384   var soapHeaders = this.createSoapHeader_(doc, methodName);
385   if (soapHeaders != null) {
386     envelope.appendChild(soapHeaders);
387   }
388   var soapBody = this.createSoapBody_(doc, methodName, parameters,
389       parameterValues);
390   envelope.appendChild(soapBody);
391   return doc;
392 };

394 /**
395  * Creates an empty SOAP envelope node.
396  *
397  * @param {Document} doc Document to user for creating the node.
398  * @return {Node} New node.
399  * @private
400  */
401 google.system.soap.Service.prototype.createEmptyEnvelope_ = function(doc) {
402   var envelope = doc.createElement('soap:Envelope');
403   envelope.setAttribute('xmlns:soap',
404       'http://schemas.xmlsoap.org/soap/envelope/');
405   envelope.setAttribute('xmlns:xsi',
406       'http://www.w3.org/2001/XMLSchema-instance');
407   envelope.setAttribute('xmlns:xsd', 'http://www.w3.org/2001/XMLSchema');
408   return envelope;
409 };

411 /**
412  * Retrieves the required SOAP header for a given method.
413  *
414  * @param {string} methodName Method name.
415  * @param {string} direction Headers direction.
416  * @return {Array} An array of required headers.
417  * @private
418  */
419 google.system.soap.Service.prototype.getSoapHeadersForMethod_ = function(
420     methodName, direction) {
421   var methodHeaders = this.getType().getMethodByName(methodName).getHeaders();
422 
423   var retVal = [];
424   for (var i = 0; i < methodHeaders.length; i++) {
425     if (methodHeaders[i].direction == direction) {
426       retVal.push(methodHeaders[i]);
427     }
428   }
429   return retVal;
430 };

432 /**

439 google.system.soap.Service.prototype.addLocalNsDeclarations_ = function(

450 /**
451  * Creates the headers of a given method.
452  *
453  * @param {Document} doc Document to use to create the element.
454  * @param {string} methodName Method name that contains the headers definition.
455  * @return {Node} Node with the headers.
456  * @private
457  */
458 google.system.soap.Service.prototype.createSoapHeader_ = function(
459     doc, methodName) {
460   var headers = this.getSoapHeadersForMethod_(methodName, 'in');
461   var headerNode = null;
462   var defaultNs = this.getServiceNamespace_();
463   var xmlnt = this.getNametable();
464 
465   for (var i = 0; i < headers.length; i++) {
466     var header = headers[i];
467     var property = this.getType().getPropertyByName(header.propertyName);
468 
469     if (property == null) {
470       continue;
471     }
472     var propertyValue = this[property.getGetterName()]();
473 
474     if (propertyValue == null) {
475       continue;
476     }
477 
478     // There is at least one soap header, so we need to create soap:Header.
479     if (headerNode == null) {
480       headerNode = doc.createElement('soap:Header');
481     }
482 
483     var nodePrefix = xmlnt.getPrefixFromNamespace(defaultNs);
484     if (property.isArray()) {
485       for (var j = 0; j < propertyValue.length; j++) {
486         var childNode = this.serializeProperty_(doc, xmlnt, defaultNs,
487             nodePrefix, property.getName(), propertyValue[j]);
488         this.addLocalNsDeclarations_(childNode);
489         headerNode.appendChild(childNode);
490       }
491     } else {
492       var childNode = this.serializeProperty_(doc, xmlnt, defaultNs, nodePrefix,
493           property.getName(), propertyValue);
494       this.addLocalNsDeclarations_(childNode);
495       headerNode.appendChild(childNode);
496     }
497   }
498 
499   return headerNode;
500 };

502 /**
503  * Creates the SOAP body of a given method.
504  *
505  * @param {Document} doc Document to create the elements.
506  * @param {string} methodName Method name.
507  * @param {Array.<Object>} parameters Method parameters.
508  * @param {Array.<*>} parameterValues Method values.
509  * @return {Node} The created body element.
510  * @private
511  */
512 google.system.soap.Service.prototype.createSoapBody_ = function(
513     doc, methodName, parameters, parameterValues) {
514   var body = doc.createElement('soap:Body');
515 
516   var xmlnt = this.getNametable();
517   var methodNs = this.getServiceMethodNamespace_(methodName);
518   var nodePrefix = xmlnt.getPrefixFromNamespace(methodNs);
519   var methodNode = doc.createElement(nodePrefix + ':' + methodName);
520 
521   this.addLocalNsDeclarations_(methodNode);
522 
523   for (var i = 0; i < parameters.length; i++) {
524     parameterName = parameters[i].name;
525     parameterValue = parameterValues[i];
526     var parameterNode = null;
527 
528     if (parameterValue instanceof Array) {
529       for (var j = 0; j < parameterValue.length; j++) {
530         var childNode = this.serializeMethodParameter_(doc, xmlnt, methodNs,
531             nodePrefix, methodName, parameterName, parameterValue[j]);
532         methodNode.appendChild(childNode);
533       }
534     } else {
535       var childNode = this.serializeMethodParameter_(doc, xmlnt, methodNs,
536           nodePrefix, methodName, parameterName, parameterValue);
537       methodNode.appendChild(childNode);
538     }
539   }
540 
541   body.appendChild(methodNode);
542 
543   return body;
544 };

546 /**
547  * Sends the serialized request to the server.
548  *
549  * @param {Document} serializedDoc The XML document to send.
550  * @param {function} successCallback Callback function to be invoked in case
551  * of success.
552  * @param {function} failureCallback Callback function to be invoked in case
553  * of failure.
554  * @private
555  */
556 google.system.soap.Service.prototype.postXmlToServer_ = function(
557     serializedDoc, successCallback, failureCallback) {
558   var thisPtr = this;
559   var request = this.getHttpWebTransport();
560 
561   var postParams = serializedDoc;
562 
563   if (request != null) {
564     request.send(thisPtr.url_, 'POST', postParams, null, 0, goog.bind(
565         function(response) {
566           if (successCallback != null) {
567             successCallback.call(thisPtr, response);
568           }
569         }, this),
570         goog.bind(function(response) {
571           if (failureCallback != null) {
572             failureCallback.call(thisPtr, response);
573           }
574         }, this)
575     );
576   }
577 };

579 /**
580  * Parses a call error.
581  *
582  * @param {Document} errorDocument Response error document.
583  * @return {google.system.soap.Exception} The exception.
584  * @private
585  */
586 google.system.soap.Service.prototype.parseSoapFault_ = function(
587     errorDocument) {
588   var xmlnt = this.getNametable();
589   var rootNode = errorDocument.documentElement;
590   if (this.getNodeName_(rootNode) == 'Envelope' && rootNode.namespaceURI ==
591       xmlnt.getNamespaceFromPrefix('soap')) {
592     for (var i = 0; i < rootNode.childNodes.length; i++) {
593       if (this.getNodeName_(rootNode.childNodes[i]) == 'Body' &&
594           rootNode.childNodes[i].namespaceURI ==
595           xmlnt.getNamespaceFromPrefix('soap')) {
596         var faultNode = rootNode.childNodes[i].childNodes[0];
597 
598         var faultCode = null;
599         var faultString = null;
600         var detail = null;
601 
602         for (var j = 0; j < faultNode.childNodes.length; j++) {
603           if (this.getNodeName_(faultNode.childNodes[j]) == 'faultcode') {
604             faultCode = goog.dom.getTextContent(faultNode.childNodes[j]);
605           }
606           if (this.getNodeName_(faultNode.childNodes[j]) == 'faultstring') {
607             faultString = goog.dom.getTextContent(faultNode.childNodes[j]);
608           }
609           if (this.getNodeName_(faultNode.childNodes[j]) == 'detail') {
610             detail = goog.dom.xml.serialize(faultNode.childNodes[j]);
611           }
612         }
613         return new google.system.soap.Exception(faultCode, faultString, detail);
614       }
615     }
616   }
617 };

619 /**
620  * Parses a SOAP response.
621  *
622  * @param {string} responseDocument Response XML document.
623  * @param {string} methodName Name of the called method.
624  * @param {google.system.soap.ObjectType} returnType Type to return.
625  * @return  {google.system.soap.Object} The parsed object.
626  * @private
627  */
628 google.system.soap.Service.prototype.parseSoapResponse_ = function(
629     responseDocument, methodName, returnType) {
630   var xmlnt = this.getNametable();
631   var rootNode = responseDocument.documentElement;
632 
633   if (this.getNodeName_(rootNode) == 'Envelope' && rootNode.namespaceURI ==
634       xmlnt.getNamespaceFromPrefix('soap')) {
635     // TODO(Anash): Add SOAP Response header parsing here.
636     for (var i = 0; i < rootNode.childNodes.length; i++) {
637       if (this.getNodeName_(rootNode.childNodes[i]) == 'Body' &&
638           rootNode.childNodes[i].namespaceURI ==
639           xmlnt.getNamespaceFromPrefix('soap')) {
640         var responseNode = rootNode.childNodes[i].childNodes[0];
641         return this.parseSoapResponseBody_(responseNode, methodName,
642             returnType);
643       }
644     }
645   }
646 };

648 /**
649  * Parses a response body.
650  *
651  * @param {Node} responseNode Response body node.
652  * @param {string} methodName Method name.
653  * @param {google.system.soap.ObjectType} returnType Type to return.
654  * @return {google.system.soap.Object} The parsed object.
655  * @private
656  */
657 google.system.soap.Service.prototype.parseSoapResponseBody_ = function(
658     responseNode, methodName, returnType) {
659   var retVal = null;
660   if (returnType.isArray == true) {
661     retVal = [];
662     for (var i = 0; i < responseNode.childNodes.length; i++) {
663       retVal.push(this.parseSingleObject_(responseNode.childNodes[i],
664           methodName, returnType));
665     }
666   } else {
667     retVal = this.parseSingleObject_(responseNode.childNodes[0], methodName,
668         returnType);
669   }
670   return retVal;
671 };

673 /**
674  * Parses a single node and returns the mapped SOAP object.
675  *
676  * @param {Object} responseNode The node to parse.
677  * @param {Object} methodName Name of the invoked method.
678  * @param {google.system.soap.ObjectType} returnType Return type definition.
679  * @return {google.system.soap.Object} The parsed object.
680  * @private
681  */
682 google.system.soap.Service.prototype.parseSingleObject_ = function(
683     responseNode, methodName, returnType) {
684   var retval = null;
685   if (responseNode == null) {
686     return retval;
687   }
688   // First, try to identify if an xsi:type is mentioned on the node. If yes,
689   // see if this is a SerializableSoapObject. If so, create a retVal.
690   var availableType = this.getTypeAsPerXsiType_(this.getNametable(),
691       responseNode, this.getType().getNamespace());
692   if (availableType != null) {
693     availableType = goog.getObjectByName(availableType);
694     if (availableType != null) {
695       availableType = new availableType();
696       if (availableType instanceof google.system.soap.Object) {
697         retVal = availableType;
698       }
699     }
700   } else {
701     // See if returnType is a SerializableSoapObject. if so, create a retval.
702     var desiredType = goog.getObjectByName(returnType.className);
703     if (desiredType != null) {
704       desiredType = new desiredType();
705       if (desiredType instanceof google.system.soap.Object) {
706         retVal = desiredType;
707       }
708     }
709   }
710   if (retVal == null) {
711     // This means that the value returned by the method call is not a
712     // SerializableSoapObject.
713     // TODO(Anash): Figure out how to handle this.
714   } else {
715     retVal.deserialize(responseNode, this.getNametable());
716   }
717   return retVal;
718 };

759 /**
760  * Serializes a SOAP Service method parameter.
761  *
762  * @param {Document} doc The XML document over which is been serialized.
763  * @param {google.system.soap.SoapNametable} xmlnt XML namespace table.
764  * @param {string} namespace Current namespace.
765  * @param {string} prefix Namespace prefix.
766  * @param {string} method Method name.
767  * @param {string} name Element name.
768  * @param {string} value Element value.
769  * @return {Node} The new node.
770  * @private
771  */
772 google.system.soap.Service.prototype.serializeMethodParameter_ = function(
773     doc, xmlnt, namespace, prefix, method, name, value) {
774   var nodeName = this.getSerializationNodeNameForMethodParameter_(method, name,
775       value);
776 
777   var node = doc.createElement(prefix + ':' + nodeName);
778   if (value instanceof google.system.soap.Object) {
779     if (value.getType().getBaseType() !=
780         google.system.soap.Object.prototype.getType()) {
781       node.setAttribute('xsi:type', prefix + ':' +
782         value.getType().getClassName());
783     }
784     value.serialize(doc, xmlnt, node, namespace);
785   } else {
786     goog.dom.setTextContent(node, value);
787   }
788   return node;
789 };
