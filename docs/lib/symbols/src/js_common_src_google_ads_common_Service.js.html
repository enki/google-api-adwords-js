 20 /**
 21  * @fileoverview Defines AdWords Service, the base class for all SOAP services
 22  * supported by this library.
 23  */
 24 
 25 goog.provide('google.ads.common.Service');
 26 
 27 goog.require('google.system.soap.Service');
 28 
 29 /**
 30  * The base class for all SOAP services supported by this library. Must be
 31  * subclassed and getCustomException_ implemented.
 32  *
 33  * @extends google.system.soap.Service
 34  * @constructor
 35  */
 36 google.ads.common.Service = function() {
 37   google.system.soap.Service.call(this);
 38 };
 39 goog.inherits(google.ads.common.Service, google.system.soap.Service);
 40 
 41 /**
 42  * Overriden soap.Service method to handle ads specific exceptions.
 43  *
 44  * @param {string} methodName Name of the method in the service that is been
 45  * invoked.
 46  * @param {Object.<String,Object>} parameterValues Parameters passed to the
 47  * method.
 48  * @param {function(google.system.soap.Object)} successCallback Callback
 49  * function to be invoked if the request finalizes successfully, the
 50  * de-serialized SOAP Object gets passed to the callback.
 51  * @param {function(google.system.soap.Object)} failureCallback Callback
 52  * function to be invoked if the request finalizes with an error, the
 53  * de-serialized SOAP Exception gets passed to the callback.
 54  */
 55 google.ads.common.Service.prototype.invoke = function(methodName,
 56     parameterValues, successCallback, failureCallback) {
 57   var service = this;
 58   google.system.soap.Service.prototype.invoke.call(service, methodName,
 59       parameterValues, function(retval) {
 60         successCallback(retval);
 61       },
 62       function(soapException) {
 63         failureCallback(service.getCustomException_(soapException));
 64       });
 65 };
 66 
 67 /**
 68  * Gets a custom exception that wraps the SOAP exception thrown
 69  * by the server. Meant to be overriden by the specific product service class.
 70  *
 71  * @param {google.system.soap.SoapException} ex The SOAP exception.
 72  * @return {google.system.core.Exception} A custom exception object
 73  * that wraps the SOAP exception. The exact type depends on what the overriding
 74  * service deserializes the fault xml into.
 75  * @private
 76  */
 77 google.ads.common.Service.prototype.getCustomException_ =
 78     function(ex) {
 79   return ex;
 80 };
