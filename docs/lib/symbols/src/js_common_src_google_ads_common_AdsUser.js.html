 19 /**
 20  * @fileoverview Defines AdsUser, a class that represents an Ads API user.
 21  */
 22 
 23 goog.provide('google.ads.common.AdsUser');
 24 
 25 goog.require('goog.array');
 26 goog.require('goog.string.format');
 27 goog.require('google.ads.common.ServiceSignature');
 28 goog.require('google.system.core.ArgumentNullException');
 29 
 30 /**
 31  * Represents an Ads API user.
 32  *
 33  * @param {google.ads.adwords.AppConfig} config AdWords configuration
 34  * properties.
 35  * @param {google.ads.adwords.AppConfig} serviceRegistry Service
 36  * registry the user will use to retrieve registered services.
 37  * @constructor
 38  */
 39 google.ads.common.AdsUser = function(config, serviceRegistry) {
 40   this.listeners_ = [];
 41   this.serviceFactoryMap_ = {};
 42   this.config_ = config;
 43   this.serviceRegistry_ = serviceRegistry;
 44 };
 45 
 46 /**
 47  * Checks the registry for the factory register to a given service signature
 48  * and then instanciate the factory in case of necessary and returns it.
 49  *
 50  * @param {google.ads.common.ServiceSignature} serviceSignature The service
 51  * signature to look up.
 52  * @return {google.ads.common.ServiceFactory} The instanciated factory.
 53  * @private
 54  */
 55 google.ads.common.AdsUser.prototype.getServiceFactory_ = function(
 56     serviceSignature) {
 57   var factoryClass =
 58       this.serviceRegistry_.getServiceFactoryClass(serviceSignature);
 59   if (factoryClass && !this.serviceFactoryMap_[factoryClass]) {
 60     this.serviceFactoryMap_[factoryClass] = new factoryClass(this.config_);
 61   }
 62   return this.serviceFactoryMap_[factoryClass];
 63 };
 64 
 65 /**
 66  * Creates an object of the requested type of service.
 67  *
 68  * @param {string} serviceSignature Signature of the service being requested.
 69  * @param {string} serverUrl The server url for Ads service.
 70  * @return {google.system.soap.Service} An object of the requested
 71  * type of service. The caller should cast this object to the desired type.
 72  */
 73 google.ads.common.AdsUser.prototype.getService = function(
 74     serviceSignature, serverUrl) {
 75   if (serviceSignature == null) {
 76     throw new google.system.core.ArgumentNullException('serviceSignature');
 77   }
 78   var factory = this.getServiceFactory_(serviceSignature);
 79   if (!factory) {
 80     throw new ArgumentException(goog.string.format(
 81         Google.Api.Ads.Common.CommonErrorMessages.
 82             UnregisteredServiceTypeRequested,
 83         serviceSignature.getServiceName()));
 84   }
 85   return factory.createService(serviceSignature, serverUrl);
 86 };
 87 
 88 /**
 89  * Registers a new SOAP listener that will chained to every service obtained
 90  * by this user.
 91  *
 92  * @param {google.system.soap.ServiceListener} listener Listener to be
 93  * registered.
 94  */
 95 google.ads.common.AdsUser.prototype.registerListener = function(
 96     listener) {
 97   this.listeners_.push(listener);
 98 };
 99 
100 /**
101  * Removes a listener from this user.
102  *
103  * @param {google.system.soap.ServiceListener} listener Listener to be removed.
104  * @return {boolean} True if the listener was found and removed, false
105  * otherwise.
106  */
107 google.ads.common.AdsUser.prototype.removeListener = function(
108     listener) {
109   return goog.array.remove(this.listeners_, listener);
110 };
111 
112 /**
113  * Retrieves all registered listeners of this user.
114  * @return {Array.<google.system.soap.ServiceListener>} A list of listeners.
115  */
116 google.ads.common.AdsUser.prototype.getListeners = function() {
117   return goog.array.clone(this.listeners_);
118 };
119 
120 /**
121  * Retrieves the user configuration.
122  *
123  * @return {google.ads.common.AppConfigBase} The configuration.
124  */
125 google.ads.common.AdsUser.prototype.getConfig = function() {
126   return this.config_;
127 };
