 20 /**
 21  * @fileoverview Defines AdsClient, the base class for all exceptions thrown
 22  * by the library related to an Ads API call.
 23  */
 24 
 25 goog.provide('google.ads.common.AdsException');
 26 
 27 goog.require('google.system.core.Exception');
 28 
 29 /**
 30  * Base class for all exceptions thrown by the library related to an Ads API
 31  * call.
 32  *
 33  * @extends google.system.core.Exception
 34  * @constructor
 35  * @param {string?} message An optional error message for the exception.
 36  * @param {Object?} innerException If this exception was thrown in response to
 37  * another exception, then you can wrap the original exception within this
 38  * exception.
 39  */
 40 google.ads.common.AdsException = function(message, innerException) {
 41   google.system.core.Exception.call(this, message, innerException);
 42 };
 43 goog.inherits(google.ads.common.AdsException, google.system.core.Exception);
