 33 /**
 34  * This class wraps the ClientLogin API. See
 35  * http://code.google.com/apis/accounts/docs/AuthForInstalledApps.html
 36  * for details.
 37  *
 38  * @param {AppConfigBase} config Configuration.
 39  * @param {string} service Service name.
 40  * @param {string} email Email address.
 41  * @param {string} password Password.
 42  * @constructor
 43  */
 44 google.ads.common.AuthToken = function(config, service, email,
 45     password) {
 46   if (config == null) {
 47     throw new google.system.core.ArgumentNullException('config');
 48   }
 49 
 50   this.service_ = service;
 51   this.email_ = email;
 52   this.password_ = password;
 53   this.config_ = config;
 54   this.SOURCE_ = 'Google-' + this.config_.getSignature();
 55 };
 56 
 57 /**
 58  * Url endpoint for ClientLogin API.
 59  *
 60  * @private
 61  */
 62 google.ads.common.AuthToken.prototype.URL_ =
 63     'https://www.google.com/accounts/ClientLogin';
 64 
 65 /**
 66  * The prefix to be appended for captcha urls.
 67  *
 68  * @private
 69  */
 70 google.ads.common.AuthToken.prototype.CAPTCHA_URL_PREFIX_ =
 71     'http://www.google.com/accounts/';
 72 
 73 /**
 74  * Account type to be used with ClientLogin API.
 75  *
 76  * @private
 77  */
 78 google.ads.common.AuthToken.prototype.ACCOUNT_TYPE_ = 'GOOGLE';
 79 
 80 /**
 81  * Obtains a ClientLogin token for use with various Ads APIs.
 82  *
 83  * @param {function} onSuccess Callback function to be invoked in case of
 84  * success. The authtoken gets passed to the function.
 85  * @param {function} onError Callback function to be invoked in case of
 86  * error. The authentication exception is passed to the function.
 87  */
 88 google.ads.common.AuthToken.prototype.getToken = function(
 89     onSuccess, onError) {
 90   var postParams =
 91     'accountType=' + encodeURIComponent(this.ACCOUNT_TYPE_) +
 92     '&Email=' + encodeURIComponent(this.email_) +
 93     '&Passwd=' + encodeURIComponent(this.password_) +
 94     '&service=' + encodeURIComponent(this.service_) +
 95     '&source=' + encodeURIComponent(this.SOURCE_);
 96 
 97   var transport = google.system.net.HttpWebTransportFactory.create(
 98       this.config_.getHttpWebTransport(),
 99       this.config_.getHttpWebTransportSettings());
100 
101   if (transport != null) {
102     transport.send(this.URL_, 'POST', postParams, null, 0, goog.bind(
103         function(response) {
104           var objResponse = this.parseResponse_(response);
105           if (onSuccess != null) {
106             onSuccess(objResponse['Auth']);
107           }
108         }, this),
109         goog.bind(function(response) {
110           var authException = this.extractException_(response);
111           if (onError != null) {
112             onError(authException);
113           }
114         }, this)
115     );
116   } else {
117     onError(new google.ads.common.AdsException('Can\'t instanciate web ' +
118         'transport ' + this.config_.getHttpWebTransport()));
119   }
120 };
121 
122 /**
123  * Extracts a ClientLogin failure and wraps it into an AuthTokenException.
124  *
125  * @param {string} response The Http web response.
126  * @return {google.ads.common.AuthTokenException} An AuthTokenException
127  * that corresponds to the parsed error response.
128  * @private
129  */
130 google.ads.common.AuthToken.prototype.extractException_ = function(
131     response) {
132   var tblResponse = this.parseResponse_(response);
133   var url = tblResponse['Url'];
134   var errCode = tblResponse['Error'];
135   var captchaToken = tblResponse['CaptchaToken'];
136   var captchaUrl = this.CAPTCHA_URL_PREFIX_ + tblResponse['CaptchaUrl'];
137 
138   return new google.ads.common.AuthTokenException(errCode, url,
139       captchaToken, captchaUrl,
140       google.ads.common.CommonErrorMessages.AuthTokenLoginFailed);
141 };
142 
143 /**
144  * Parses a ClientLogin response into a hashtable.
145  *
146  * @param {string} response The Http web response.
147  * @return {Object} The parsed response as an object.
148  * @private
149  */
150 google.ads.common.AuthToken.prototype.parseResponse_ = function(
151     response) {
152   var tblResponse = {};
153   var splits = response.split('\n');
154   for (var i = 0; i < splits.length; i++) {
155     var subsplits = splits[i].split('=', 2);
156     if (subsplits.length == 2) {
157       tblResponse[subsplits[0]] = subsplits[1];
158     }
159   }
160   return tblResponse;
161 };
