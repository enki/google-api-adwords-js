 20 /**
 21  * @fileoverview Defines SOAPHeaderServiceListener, a service listener in
 22  * charge of injecting AdWords API SOAP headers before a request gets sent
 23  * to the servers.
 24  */
 25 
 26 goog.provide('google.ads.adwords.SOAPHeaderServiceListener');
 27 
 28 goog.require('google.ads.adwords.RequestHeader');
 29 goog.require('google.system.soap.ServiceListener');
 30 
 31 /**
 32  * A service listener in charge of injecting AdWords API SOAP headers
 33  * before a request gets sent to the servers.
 34  *
 35  * @param {google.ads.adwords.AppConfig} config AdWords configuration
 36  * properties.
 37  * @extends google.system.soap.ServiceListener
 38  * @constructor
 39  */
 40 google.ads.adwords.SOAPHeaderServiceListener = function(
 41     config) {
 42   google.system.soap.ServiceListener.call(this);
 43   this.config_ = config;
 44 };
 45 goog.inherits(google.ads.adwords.SOAPHeaderServiceListener,
 46     google.system.soap.ServiceListener);
 47 
 48 /**
 49  * Called on before serialization, sets all AdWords API required headers from
 50  * config and then requests an auth token if required.
 51  *
 52  * @param {google.system.soap.SoapService} service Service object been invoked.
 53  * @param {string} methodName Name of the method.
 54  * @param {Array.<Object>} parameterValues Parameters passed to the method.
 55  * @param {function} successCallback Function to be called in case of sucesss.
 56  * @param {function(google.system.core.Exception)} failureCallback Function to
 57  * be called in case of an error occurred, an exception is expected by the
 58  * function.
 59  */
 60 google.ads.adwords.SOAPHeaderServiceListener.prototype.beforeSerialize =
 61     function(service, methodName, parameterValues, successCallback,
 62         failureCallback) {
 63   this.makeRequestHeader_(service, this.config_, goog.bind(this.setHeader_,
 64       this, service, successCallback, failureCallback), failureCallback);
 65 };
 66 
 67 /**
 68  * Sets the header in the service object. Used as a callback from retrieving
 69  * the auth token.
 70  *
 71  * @param {google.system.soap.SoapService} service Service object been invoked.
 72  * @param {function} successCallback Function to be called in case of sucesss.
 73  * @param {function(google.system.core.Exception)} failureCallback Function to
 74  * call in case of a error, expects an exception to be passed to it.
 75  * @param {google.ads.adwords.RequestHeader} header Header to be set.
 76  * @private
 77  */
 78 google.ads.adwords.SOAPHeaderServiceListener.prototype.setHeader_ = function(
 79     service, successCallback, failureCallback, header) {
 80   if (service.setRequestHeader != null) {
 81     service.setRequestHeader(header);
 82   }
 83   successCallback();
 84 };
 85 
 86 /**
 87  * Make a request header given a config.
 88  *
 89  * @param {google.system.soap.SoapService} service Service object been invoked.
 90  * @param {Object.<string, string>} config A dictionary of configuration,
 91  * parameters.
 92  * @param {function} callback A function to be called with the generated
 93  * RequestHeader object, to be injected in the called service.
 94  * @param {function} failureCallback A function to be called when an error is
 95  * encountered.
 96  * @private
 97  */
 98 google.ads.adwords.SOAPHeaderServiceListener.prototype.makeRequestHeader_ =
 99     function(service, config, callback, failureCallback) {
100   var headers = this.readHeadersFromConfig_(config);
101   var requestHeader = service.requestHeader;
102   if (requestHeader == null) {
103     var fullType = service.getType().getPropertyByName('requestHeader').
104         getFullClassName();
105     availableType = goog.getObjectByName(fullType);
106     if (availableType != null) {
107       requestHeader = new availableType();
108     }
109   }
110 
111   for (var key in headers) {
112     var setter = 'set' + key.substr(0, 1).toUpperCase() + key.substr(1);
113     var getter = 'get' + key.substr(0, 1).toUpperCase() + key.substr(1);
114 
115     // Respect any values set by the user, even if it is a null.
116     if (requestHeader[getter] && typeof requestHeader[getter](headers[key]) !=
117         'undefined') {
118       continue;
119     }
120 
121     if (requestHeader[setter] != null) {
122       requestHeader[setter](headers[key]);
123     }
124   }
125 
126   if (headers['authToken'] == null) {
127     var tokenGen = new google.ads.common.AuthToken(
128         config, 'adwords', config.getEmail(), config.getPassword());
129     tokenGen.getToken(function(token) {
130       config.setConfigValue('authToken', token);
131       requestHeader.setAuthToken(token);
132       callback(requestHeader);
133     }, function (exception) {
134       failureCallback(exception);
135     });
136   } else {
137     callback(requestHeader);
138   }
139 };
140 
141 /**
142  * Reads the headers from Application configuration.
143  *
144  * @param {google.ads.common.AppConfigBase} config The configuration
145  * class.
146  * @return {Object.<string, string>} An object, with key-value pairs as
147  * headername, headervalue.
148  * @private
149  */
150 google.ads.adwords.SOAPHeaderServiceListener.prototype.readHeadersFromConfig_ =
151     function(config) {
152   var configHeaders = {};
153   var headerKeys = ['authToken', 'email', 'password', 'clientCustomerId',
154       'clientEmail', 'developerToken', 'applicationToken', 'userAgent'];
155 
156   for (var i = 0; i < headerKeys.length; i++) {
157     var value = config.getConfigValue(headerKeys[i]);
158     if (!goog.string.isEmptySafe(value)) {
159       configHeaders[headerKeys[i]] = value;
160     }
161   }
162   return configHeaders;
163 };
