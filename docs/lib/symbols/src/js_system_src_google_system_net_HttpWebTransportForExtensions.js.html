 19 /**
 20  * @fileoverview Defines HttpWebTransportForExtensions, a wrapper that uses a
 21  * direct ajax call to the server. Cross domain xhr permissions are defined
 22  * in manifest.json.
 23  */
 24 
 25 goog.provide('google.system.net.HttpWebTransportForExtensions');
 26 
 27 goog.require('goog.net.XhrIo');
 28 goog.require('google.system.net.HttpWebTransport');
 29 
 30 /**
 31  * This class defines a wrapper that uses a server proxy for making HTTP calls.
 32  *
 33  * @param {google.system.net.HttpWebTransportSettings} settings Setting to be
 34  * applied.
 35  * @extends google.system.net.HttpWebTransport
 36  * @constructor
 37  */
 38 google.system.net.HttpWebTransportForExtensions = function(settings) {
 39   google.system.net.HttpWebTransport.call(this, settings);
 40 };
 41 goog.inherits(google.system.net.HttpWebTransportForExtensions,
 42     google.system.net.HttpWebTransport);
 43 
 44 /**
 45  * Sends a request to the requested server.
 46  *
 47  * @param {string} url The url to make the request to.
 48  * @param {string} method The HTTP method.
 49  * @param {Object} content The data to send to servers.
 50  * @param {Object} headers The HTTP headers.
 51  * @param {number} timeoutInterval The timeout interval.
 52  * @param {function} onSuccess The callback for success.
 53  * @param {function} onFailure The callback for failure.
 54  */
 55 google.system.net.HttpWebTransportForExtensions.prototype.send = function(url,
 56     method, content, headers, timeoutInterval, onSuccess, onFailure) {
 57 
 58   goog.net.XhrIo.send(url, goog.bind(
 59     function(event) {
 60       var success = event.target.isSuccess();
 61       var status = event.target.getStatus();
 62       var response = event.target.getResponseText();
 63       if (success && status == 200) {
 64         if (onSuccess != null) {
 65           onSuccess(response);
 66         }
 67       } else {
 68         if (onFailure != null) {
 69           onFailure(response);
 70         }
 71       }
 72     }, this),
 73     method, content, headers, timeoutInterval
 74   );
 75 };
