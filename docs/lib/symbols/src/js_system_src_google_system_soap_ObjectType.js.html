 21  * @fileoverview Defines Type, a class to store the type of a
 22  *     google.system.soap.Object.

 30 /**
 31  * Represents a SOAP Object Type.
 32  * Contains that metadata that conforms a SOAP Object, like its properties,
 33  * XmlNamespace, etc.
 34  *
 35  * @param {string} xmlNamespace XML namespace of the object.
 36  * @param {string} fullClassName Full class name of the this object type.
 37  * @param {string} isEnum True if the object is of type enum.
 38  * @param {string?} baseFullClassName Base class name if this type inherits
 39  *     from other, this constructor takes cares automatically of the object
 40  *     inheritance if this type is provided, if not it makes the class name
 41  *     inherit from google.system.soap.Object.
 42  * @constructor
 43  */
 44 google.system.soap.ObjectType = function(xmlNamespace, fullClassName, isEnum,
 45     baseFullClassName) {
 46 
 47   /**
 48    * XML namespace of the object.
 49    * @type {string}
 50    * @private
 51    */
 52   this.xmlNamespace_ = xmlNamespace;
 53 
 54   /**
 55    * Full object classname (E.g. google.ads.adwords.v201003.Campaign).
 56    * @type {string}
 57    * @private
 58    */
 59   this.fullClassName_ = fullClassName;
 60 
 61   /**
 62    * If it's an enumeration type.
 63    * @type {boolean}
 64    * @private
 65    */
 66   this.isEnum_ = isEnum;
 67 
 68   /**
 69    * List of this type properties.
 70    * @type {Array.<google.system.soap.ObjectProperty>}
 71    * @private
 72    */
 73   this.properties_ = [];
 74 
 75   if (fullClassName != 'google.system.soap.Object') {
 76     this.baseFullClassName_ = baseFullClassName ? baseFullClassName :
 77         'google.system.soap.Object';
 78   }
 79   var classObject = goog.getObjectByName(this.fullClassName_);
 80   if (goog.isNull(classObject)) {
 81     throw new google.system.core.Exception('nonexistent fullClassName: ' +
 82         fullClassName + ' provided');
 83   }
 84   if (fullClassName != 'google.system.soap.Object') {
 85     var baseClassObject = goog.getObjectByName(this.baseFullClassName_);
 86     if (goog.isNull(baseClassObject)) {
 87       throw new google.system.core.Exception('nonexistent baseFullClassName: ' +
 88           this.baseFullClassName_ + ' provided');
 89     }
 90     goog.inherits(classObject, baseClassObject);
 91   }
 92   classObject.prototype['type_'] = this;
 93 };

 95 /**
 96  * Static utility method to define a type in the system.
 97  *
 98  * @param {string} xmlNamespace Namespace of the type.
 99  * @param {string} fullClassName Type full class name
100  *     (E.g. google.ads.adwords.v201008.Campaign).
101  * @param {boolean} isEnum True if it represent an enumeration.
102  * @param {?string} baseFullClassName Base type full name, defaults to
103  *     'google.system.soap.Object'.
104  * @return {google.system.soap.ObjectType} The new object type.
105  */
106 google.system.soap.ObjectType.defineType = function(
107     xmlNamespace, fullClassName, isEnum, baseFullClassName) {
108   return new google.system.soap.ObjectType(xmlNamespace, fullClassName, isEnum,
109       baseFullClassName);
110 };

121 /**
122  * Retrieves the type full class name.
123  *
124  * @return {string} The type full class name.
125  */
126 google.system.soap.ObjectType.prototype.getFullClassName = function() {
127   return this.fullClassName_;
128 };
129 
130 /**
131  * Retrieves the type class name, not including the namespace. So for example
132  * a type with class name 'com.google.ads.ClassA' it will return 'ClassA'.
133  *
134  * @return {string} The type class name.
135  */
136 google.system.soap.ObjectType.prototype.getClassName = function() {
137   var nameArray = this.fullClassName_.split('.');
138   return nameArray[nameArray.length - 1];
139 };
140 
141 /**
142  * Retrieves the type class namespace. So for example
143  * a type with class name 'com.google.ads.ClassA' it will return
144  * 'com.google.ads'.
145  *
146  * @return {string} The class namespace.
147  */
148 google.system.soap.ObjectType.prototype.getNamespace = function() {
149   var i = this.fullClassName_.lastIndexOf('.');
150   i = i > 0 ? i : 0;
151   return this.fullClassName_.substring(0, i);
152 };

172 /**
173  * Retrieves the base {google.system.soap.ObjectType} of this type.
174  *
175  * @return {google.system.soap.ObjectType} The base type.
176  */
177 google.system.soap.ObjectType.prototype.getBaseType = function() {
178   if (this.baseFullClassName_) {
179     return goog.getObjectByName(this.baseFullClassName_).prototype.getType();
180   }
181 };

183 /**
184  * Adds a property to the object type definition.
185  *
186  * @param {string} name Name of the property, getters and setters will be
187  *     generated based on this.
188  * @param {string} propertyFullClassName Full classname that maps the property
189  *     unless is a system property where possible values can be
190  *     [boolean|integer|string].
191  * @param {(string|Object.<string, string>)} xmlElementNameMapping In general
192  *     this maps to a simple xmlElement name unless, it is a choice type of SOAP
193  *     element where multiple xml element names can be defined depending on the
194  *     type of the value to be serialiazed, so it can take the form of
195  *     [{elementName: 'element', className: 'fullClassName'}, ...].
196  * @param {boolean} isArray True if is an array type of property, false
197  *     otherwise.
198  * @param {boolean} isSystem True if is a system kind of property, false
199  *     otherwise.
200  * @return {google.system.soap.ObjectProperty} The new property.
201  */
202 google.system.soap.ObjectType.prototype.addProperty = function(
203     name, propertyFullClassName, xmlElementNameMapping, isArray, isSystem) {
204 
205   var property = new google.system.soap.ObjectProperty(name,
206       propertyFullClassName, xmlElementNameMapping, isArray, isSystem);
207 
208   if (goog.isNull(this.getPropertyByName(name, true))) {
209     this.properties_.push(property);
210     var classType = goog.getObjectByName(this.fullClassName_);
211     if (!goog.isNull(classType)) {
212       classType.prototype[property.getSetterName()] = function(value) {
213         this[name] = value;
214       };
215       classType.prototype[property.getGetterName()] = function() {
216         return this[name];
217       };
218     }
219   }
220 
221   return property;
222 };
223 
224 /**
225  * Retrieves the list of properties tied to this type.
226  *
227  * @param {boolean?} ignoreInheritance If true it will only look at this type
228  *     direct property and not follow the inherentance path.
229  * @return {Array.<google.system.soap.ObjectProperty>} The list of properties.
230  */
231 google.system.soap.ObjectType.prototype.getProperties = function(
232     ignoreInheritance) {
233   var properties = [];
234 
235   if (!ignoreInheritance) {
236     // Anash: We need to enforce order of nodes when there's inheritance in
237     // serialized types. Base class properties should come before derived
238     // class properties.
239     var tempType = this;
240     var types = [];
241     while (tempType != null) {
242       types.push(tempType);
243       tempType = tempType.getBaseType();
244     }
245     types = types.reverse();
246     for(var i = 0; i < types.length; i++) {
247       for (var j = 0; j < types[i].properties_.length; j++) {
248         properties.push(types[i].properties_[j]);
249       }
250     }
251   } else {
252     properties = goog.array.clone(this.properties_);
253   }
254 
255   return properties;
256 };
257 
258 /**
259  * Retrieves a property by its name.
260  *
261  * @param {string} name Name of the property to retrieve.
262  * @param {boolean} ignoreInheritance If true it will only look at this type
263  *     direct property and not follow the inheritance path.
264  * @return {google.system.soap.ObjectProperty} The property or null if not
265  *     property was found.
266  */
267 google.system.soap.ObjectType.prototype.getPropertyByName = function(name,
268     ignoreInheritance) {
269   var properties = this.getProperties(ignoreInheritance);
270   for (var i = 0; i < properties.length; i++) {
271     if (properties[i].getName() == name) {
272       return properties[i];
273     }
274   }
275   return null;
276 };
