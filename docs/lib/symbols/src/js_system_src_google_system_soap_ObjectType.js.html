 28 /**
 29  * Represents a SOAP Object Type.
 30  * Contains that metadata that conforms a SOAP Object, like its properties,
 31  * XmlNamespace, etc.
 32  *
 33  * @param {string} xmlNamespace XML namespace of the object.
 34  * @param {string} fullClassName Full class name of the this object type.
 35  * @param {string} isEnum True if the object is of type enum.
 36  * @param {string?} baseFullClassName Base class name if this type inherits
 37  *     from other, this constructor takes cares automatically of the object
 38  *     inheritance if this type is provided, if not it makes the class name
 39  *     inherit from google.system.soap.Object.
 40  * @constructor
 41  */
 42 google.system.soap.ObjectType = function(xmlNamespace, fullClassName, isEnum,
 43     baseFullClassName) {
 44 
 45   /**
 46    * XML namespace of the object.
 47    * @type {string}
 48    * @private
 49    */
 50   this.xmlNamespace_ = xmlNamespace;
 51 
 52   /**
 53    * Full object classname (E.g. google.ads.adwords.v201003.Campaign).
 54    * @type {string}
 55    * @private
 56    */
 57   this.fullClassName_ = fullClassName;
 58 
 59   /**
 60    * If it's an enumeration type.
 61    * @type {boolean}
 62    * @private
 63    */
 64   this.isEnum_ = isEnum;
 65 
 66   /**
 67    * List of this type properties.
 68    * @type {Array.<google.system.soap.ObjectProperty>}
 69    * @private
 70    */
 71   this.properties_ = [];
 72 
 73   if (fullClassName != 'google.system.soap.Object') {
 74     this.baseFullClassName_ = baseFullClassName ? baseFullClassName :
 75         'google.system.soap.Object';
 76   }
 77   var classObject = goog.getObjectByName(this.fullClassName_);
 78   if (goog.isNull(classObject)) {
 79     throw new google.system.core.Exception('nonexistent fullClassName: ' +
 80         fullClassName + ' provided');
 81   }
 82   if (fullClassName != 'google.system.soap.Object') {
 83     var baseClassObject = goog.getObjectByName(this.baseFullClassName_);
 84     if (goog.isNull(baseClassObject)) {
 85       throw new google.system.core.Exception('nonexistent baseFullClassName: ' +
 86           this.baseFullClassName_ + ' provided');
 87     }
 88     goog.inherits(classObject, baseClassObject);
 89   }
 90   classObject.prototype['type_'] = this;
 91 };

 93 /**
 94  * Static utility method to define a type in the system.
 95  *
 96  * @param {string} xmlNamespace Namespace of the type.
 97  * @param {string} fullClassName Type full class name
 98  *     (E.g. google.ads.adwords.v201008.Campaign).
 99  * @param {boolean} isEnum True if it represent an enumeration.
100  * @param {?string} baseFullClassName Base type full name, defaults to
101  *     'google.system.soap.Object'.
102  * @return {google.system.soap.ObjectType} The new object type.
103  */
104 google.system.soap.ObjectType.defineType = function(
105     xmlNamespace, fullClassName, isEnum, baseFullClassName) {
106   return new google.system.soap.ObjectType(xmlNamespace, fullClassName, isEnum,
107       baseFullClassName);
108 };

119 /**
120  * Retrieves the type full class name.
121  *
122  * @return {string} The type full class name.
123  */
124 google.system.soap.ObjectType.prototype.getFullClassName = function() {
125   return this.fullClassName_;
126 };
127 
128 /**
129  * Retrieves the type class name, not including the namespace. So for example
130  * a type with class name 'com.google.ads.ClassA' it will return 'ClassA'.
131  *
132  * @return {string} The type class name.
133  */
134 google.system.soap.ObjectType.prototype.getClassName = function() {
135   var nameArray = this.fullClassName_.split('.');
136   return nameArray[nameArray.length - 1];
137 };
138 
139 /**
140  * Retrieves the type class namespace. So for example
141  * a type with class name 'com.google.ads.ClassA' it will return
142  * 'com.google.ads'.
143  *
144  * @return {string} The class namespace.
145  */
146 google.system.soap.ObjectType.prototype.getNamespace = function() {
147   var i = this.fullClassName_.lastIndexOf('.');
148   i = i > 0 ? i : 0;
149   return this.fullClassName_.substring(0, i);
150 };

170 /**
171  * Retrieves the base {google.system.soap.ObjectType} of this type.
172  *
173  * @return {google.system.soap.ObjectType} The base type.
174  */
175 google.system.soap.ObjectType.prototype.getBaseType = function() {
176   if (this.baseFullClassName_) {
177     return goog.getObjectByName(this.baseFullClassName_).prototype.getType();
178   }

181 /**
182  * Adds a property to the object type definition.
183  *
184  * @param {string} name Name of the property, getters and setters will be
185  *     generated based on this.
186  * @param {string} propertyFullClassName Full classname that maps the property
187  *     unless is a system property where possible values can be
188  *     [boolean|integer|string].
189  * @param {(string|Object.<string, string>)} xmlElementNameMapping In general
190  *     this maps to a simple xmlElement name unless, it is a choice type of SOAP
191  *     element where multiple xml element names can be defined depending on the
192  *     type of the value to be serialiazed, so it can take the form of
193  *     [{elementName: 'element', className: 'fullClassName'}, ...].
194  * @param {boolean} isArray True if is an array type of property, false
195  *     otherwise.
196  * @param {boolean} isSystem True if is a system kind of property, false
197  *     otherwise.
198  * @return {google.system.soap.ObjectProperty} The new property.
199  */
200 google.system.soap.ObjectType.prototype.addProperty = function(
201     name, propertyFullClassName, xmlElementNameMapping, isArray, isSystem) {
202 
203   var property = new google.system.soap.ObjectProperty(name,
204       propertyFullClassName, xmlElementNameMapping, isArray, isSystem);
205 
206   if (goog.isNull(this.getPropertyByName(name, true))) {
207     this.properties_.push(property);
208     var classType = goog.getObjectByName(this.fullClassName_);
209     if (!goog.isNull(classType)) {
210       classType.prototype[property.getSetterName()] = function(value) {
211         this[name] = value;
212       };
213       classType.prototype[property.getGetterName()] = function() {
214         return this[name];
215       };
216     }
217   }
218 
219   return property;
220 };
221 
222 /**
223  * Retrieves the list of properties tied to this type.
224  *
225  * @param {boolean?} ignoreInheritance If true it will only look at this type
226  *     direct property and not follow the inherentance path.
227  * @return {Array.<google.system.soap.ObjectProperty>} The list of properties.
228  */
229 google.system.soap.ObjectType.prototype.getProperties = function(
230     ignoreInheritance) {
231   var properties = [];
232 
233   if (!ignoreInheritance) {
234     var tempType = this;
235     while (tempType != null) {
236       for (var i = 0; i < tempType.properties_.length; i++) {
237         properties.push(tempType.properties_[i]);
238       }
239       tempType = tempType.getBaseType();
240     }
241   } else {
242     properties = goog.array.clone(this.properties_);
243   }
244 
245   return properties;
246 };
247 
248 /**
249  * Retrieves a property by its name.
250  *
251  * @param {string} name Name of the property to retrieve.
252  * @param {boolean} ignoreInheritance If true it will only look at this type
253  *     direct property and not follow the inheritance path.
254  * @return {google.system.soap.ObjectProperty} The property or null if not
255  * property was found.
256  */
257 google.system.soap.ObjectType.prototype.getPropertyByName = function(name,
258     ignoreInheritance) {
259   var properties = this.getProperties(ignoreInheritance);
260   for (var i = 0; i < properties.length; i++) {
261     if (properties[i].getName() == name) {
262       return properties[i];
263     }
264   }
265   return null;
266 };
