 20 /**
 21  * @fileoverview Defines google.system.soap.ServiceType class.
 22  */
 23 
 24 goog.provide('google.system.soap.ServiceType');
 25 
 26 goog.require('google.system.soap.MethodSignature');
 27 goog.require('google.system.soap.ObjectType');
 28 
 29 /**
 30  * Represents a SOAP Service Type.
 31  * Contains that metadata that conforms a SOAP Object, like its properties,
 32  * XmlNamespace, etc.
 33  *
 34  * @param {string} xmlNamespace Xml namespace of the service.
 35  * @param {string} fullClassName Full class name of the service.
 36  * @param {string} fullBaseClassName Full class name of the base class for the
 37  * service.
 38  * @extends google.system.soap.ObjectType
 39  * @constructor
 40  */
 41 google.system.soap.ServiceType = function(xmlNamespace, fullClassName,
 42     fullBaseClassName) {
 43   google.system.soap.ObjectType.call(this, xmlNamespace, fullClassName, false,
 44       fullBaseClassName);
 45   this.methods_ = [];
 46 };
 47 goog.inherits(google.system.soap.ServiceType, google.system.soap.ObjectType);
 48 
 49 /**
 50  * Utility static method to define a service.
 51  *
 52  * @param {string} xmlNamespace XML namespace.
 53  * @param {string} fullClassName Full class name of the service.
 54  * @param {string} fullBaseClassName Full class name of the base class for
 55  * the service.
 56  * @return {google.system.soap.ServiceType} The new service type.
 57  */
 58 google.system.soap.ServiceType.defineService = function(
 59     xmlNamespace, fullClassName, fullBaseClassName) {
 60   return new google.system.soap.ServiceType(xmlNamespace, fullClassName,
 61       fullBaseClassName);
 62 };
 63 
 64 /**
 65  * Utility static method to retrieve the {google.system.soap.ServiceType}
 66  * associated to a given class name.
 67  *
 68  * @param {string} fullClassName The full class name.
 69  * @return {google.system.soap.ServiceType} Returns the service type.
 70  */
 71 google.system.soap.ServiceType.getServiceTypeByName = function(fullClassName) {
 72   var serviceClass = goog.getObjectByName(fullClassName);
 73   return serviceClass ? serviceClass.prototype.getType() : null;
 74 };
 75 
 76 /**
 77  * Adds a method to the service definition.
 78  *
 79  * @param {string} methodName The method name.
 80  * @param {string} requestNamespace The method request namespace.
 81  * @param {string} responseNamespace The method response namespace.
 82  * @param {string} use The use of the method literal/encoded.
 83  * @param {string} parameterStyle The parameter style wrapped or non-wrapped.
 84  * @param {string} returnXmlElement The XML element to start the deserialization
 85  * with.
 86  * @param {Object} returnTypeDefinition The return definition type.
 87  * @param {Array} headers The Headers configured for the service.
 88  * @param {Array} parameters The parameters that the service takes.
 89  * @return {google.system.soap.MethodSignature} The created method signature.
 90  */
 91 google.system.soap.ServiceType.prototype.addMethod = function(
 92     methodName, requestNamespace, responseNamespace, use, parameterStyle,
 93     returnXmlElement, returnTypeDefinition, headers, parameters) {
 94   var method = new google.system.soap.MethodSignature(requestNamespace,
 95       responseNamespace, use, parameterStyle, returnXmlElement,
 96       returnTypeDefinition, headers, parameters);
 97   this.methods_[methodName] = method;
 98   return method;
 99 };
100 
101 /**
102  * Retrieves all the service methods.
103  *
104  * @return {Array.<google.system.soap.MethodSignature>} The service methods.
105  */
106 google.system.soap.ServiceType.prototype.getMethods = function() {
107   return goog.array.clone(this.methods_);
108 };
109 
110 /**
111  * Retrieves a method given a name.
112  *
113  * @param {string} name The name of the method to find.
114  * @return {google.system.soap.MethodSignature} The service method signature.
115  */
116 google.system.soap.ServiceType.prototype.getMethodByName = function(name) {
117   return this.methods_[name];
118 };
