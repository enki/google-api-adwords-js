 20 /**
 21  * @fileoverview Defines AdWordsAppConfig, a class that reads the configuration
 22  * keys for AdWords.
 23  */
 24 
 25 goog.provide('google.ads.adwords.AppConfig');
 26 
 27 goog.require('google.ads.common.AppConfigBase');
 28 
 29 /**
 30  * This class reads the configuration keys for AdWords.
 31  *
 32  * @extends google.ads.common.AppConfigBase
 33  * @constructor
 34  */
 35 google.ads.adwords.AppConfig = function() {
 36   google.ads.common.AppConfigBase.call(this);
 37 
 38   this.authToken_ = null;
 39   this.email_ = null;
 40   this.password_ = null;
 41   this.clientEmail_ = null;
 42   this.clientCustomerId_ = null;
 43   this.developerToken_ = null;
 44   this.userAgent_ = null;
 45   this.legacyAdWordsApiServer_ = this.DEFAULT_LEGACY_ADWORDSAPI_SERVER_;
 46   this.adWordsApiServer_ = this.DEFAULT_ADWORDSAPI_SERVER_;
 47 
 48   var runtimeConfig = google.system.core.Runtime.getInstance().getConfig();
 49   if (runtimeConfig && runtimeConfig['AdWordsApi']) {
 50     this.readSettings(google.system.core.Runtime.getInstance().
 51         getConfig()['AdWordsApi']);
 52   }
 53 };
 54 goog.inherits(google.ads.adwords.AppConfig, google.ads.common.AppConfigBase);

284 /**
285  * Reads the settings from configuration file.
286  *
287  * @param {string} settings Setting array to read from.
288  */
289 google.ads.adwords.AppConfig.prototype.readSettings = function(
290     settings) {
291   google.ads.common.AppConfigBase.prototype.readSettings.call(this,
292       settings);
293   this.setEmail(settings[this.EMAIL_] || this.getEmail());
294   this.setPassword(settings[this.PASSWORD_] || this.getPassword());
295   this.setClientEmail(settings[this.CLIENT_EMAIL_] || this.getClientEmail());
296   this.setClientCustomerId(settings[this.CLIENT_CUSTOMER_ID_] ||
297       this.getClientCustomerId());
298   this.setDeveloperToken(settings[this.DEVELOPER_TOKEN_] ||
299       this.getDeveloperToken());
300   this.setAuthToken(settings[this.AUTHTOKEN_] || this.getAuthToken());
301   this.setUserAgent(settings[this.USER_AGENT_] || this.getUserAgent());
302   this.setAdWordsApiServer(settings[this.ADWORDSAPI_SERVER_] ||
303       this.getAdWordsApiServer());
304 };
