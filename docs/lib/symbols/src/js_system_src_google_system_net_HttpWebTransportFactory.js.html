 19 /**
 20  * @fileoverview Defines HttpWebTransportFactory, a factory for creating
 21  * HttpWebTransport.
 22  */
 23 
 24 goog.provide('google.system.net.HttpWebTransportFactory');
 25 
 26 /**
 27  * Factory for HttpWebTransport kind of objects.
 28  *
 29  * @constructor
 30  */
 31 google.system.net.HttpWebTransportFactory = function() {
 32 };
 33 
 34 /**
 35  * Constructs a new HttpWebTransport object given its name and settings.
 36  *
 37  * @param {string} transportName Name of the provider.
 38  * @param {google.system.net.HttpWebTransportSettings} settings Settings to be
 39  * applied.
 40  * @return {google.system.net.HttpWebTransport|null} A new transport object
 41  * already configured or null if not provided was found.
 42  */
 43 google.system.net.HttpWebTransportFactory.create = function(transportName,
 44     settings) {
 45   var xhrProvider = goog.getObjectByName(transportName);
 46 
 47   if (xhrProvider != null) {
 48     return new xhrProvider(settings);
 49   } else {
 50     return null;
 51   }
 52 };
