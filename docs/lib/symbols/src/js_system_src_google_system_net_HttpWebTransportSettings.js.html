 19 /**
 20  * @fileoverview Defines HttpWebTransportSettings, settings for creating a
 21  * HttpWebTransport.
 22  */
 23 
 24 goog.provide('google.system.net.HttpWebTransportSettings');
 25 
 26 /**
 27  * settings for creating a HttpWebTransport.
 28  *
 29  * @param {?Object.<string, string>} settings To pre-populate with settings.
 30  * @constructor
 31  */
 32 google.system.net.HttpWebTransportSettings = function(settings) {
 33   this.settings_ = settings || {};
 34 };
 35 
 36 /**
 37  * Sets a setting value.
 38  *
 39  * @param {string} key Key of the value.
 40  * @param {string} value New value to set.
 41  */
 42 google.system.net.HttpWebTransportSettings.prototype.setSetting = function(key,
 43     value) {
 44   this.settings_[key] = value;
 45 };
 46 
 47 /**
 48  * Retrieves the value of the given key.
 49  *
 50  * @param {string} key Key to retrieve.
 51  * @return {string} Value set or null if key is not set.
 52  */
 53 google.system.net.HttpWebTransportSettings.prototype.getSetting =
 54     function(key) {
 55   return this.settings_[key];
 56 };
