 20 /**
 21  * @fileoverview Defines AdWordsUser, a class that represents an AdWords
 22  *     API user.
 23  */
 24 
 25 goog.provide('google.ads.adwords.User');
 26 
 27 goog.require('google.ads.adwords.AppConfig');
 28 goog.require('google.ads.common.AdsUser');
 29 
 30 /**
 31  * Service creation params for AdWords API family of services.
 32  *
 33  * @param {?google.ads.adwords.AppConfig} config AdWords configuration
 34  *     properties.
 35  * @extends google.ads.common.AdsUser
 36  * @constructor
 37  */
 38 google.ads.adwords.User = function(config) {
 39   if (config == null) {
 40     config = new google.ads.adwords.AppConfig();
 41   }
 42   google.ads.common.AdsUser.call(this, config,
 43       google.ads.adwords.ServiceRegistry.getInstance());
 44 };
 45 goog.inherits(google.ads.adwords.User, google.ads.common.AdsUser);
