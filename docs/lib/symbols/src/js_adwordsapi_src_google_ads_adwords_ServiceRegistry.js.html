 20 /**
 21  * @fileoverview Lists all the services available through this library.
 22  */
 23 
 24 goog.provide('google.ads.adwords.ServiceRegistry');
 25 
 26 goog.require('google.ads.adwords.ServiceSignature');
 27 goog.require('google.ads.common.ServiceRegistry');
 28 
 29 /**
 30  * Lists all the services available through this library.
 31  *
 32  * @constructor
 33  * @extends google.ads.common.ServiceRegistry
 34  */
 35 google.ads.adwords.ServiceRegistry = function() {
 36   google.ads.common.ServiceRegistry.call(this);
 37 };
 38 goog.inherits(google.ads.adwords.ServiceRegistry,
 39     google.ads.common.ServiceRegistry);
 40 
 41 /**
 42  * Retrieves a singleton instance of this class.
 43  *
 44  * @return {google.ads.adwords.ServiceRegistry} the instance.
 45  */
 46 google.ads.adwords.ServiceRegistry.getInstance = function() {
 47   if (!google.ads.adwords.ServiceRegistry.instance_) {
 48     google.ads.adwords.ServiceRegistry.instance_ =
 49        new google.ads.adwords.ServiceRegistry();
 50   }
 51   return google.ads.adwords.ServiceRegistry.instance_;
 52 };
 53 
 54 /**
 55  * Creates a service creation parameter for defining an AdWords service.
 56  *
 57  * @param {string} version Service version.
 58  * @param {string} groupName Group name.
 59  * @param {string} serviceName Service name.
 60  * @param {object} factoryClass The factory class.
 61  * @return {google.ads.adwords.ServiceRegistrySignature} A service
 62  * creation parameter defining this service.
 63  */
 64 google.ads.adwords.ServiceRegistry.prototype.registerServiceFactory = function(
 65     version, groupName, serviceName, factoryClass) {
 66 
 67   var signature = new google.ads.adwords.ServiceSignature(version, groupName,
 68       serviceName);
 69   var namespace = 'google.ads.adwords.AdWordsService.' + version;
 70   try {
 71     // Make the closure compiler happy.
 72     var pfnProvides = goog.provide;
 73     pfnProvides(namespace);
 74   } catch (e) { }
 75   var ns = goog.getObjectByName(namespace);
 76   ns[serviceName] = signature;
 77 
 78   google.ads.common.ServiceRegistry.prototype.registerServiceFactory.call(this,
 79       signature, factoryClass);
 80 
 81   return signature;
 82 };
