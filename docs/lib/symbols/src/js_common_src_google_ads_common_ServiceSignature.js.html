 20 /**
 21  * @fileoverview Defines ServiceSignature, an API signature object. This class
 22  *     is used as a support class to assist AdsUser in creating a service
 23  *     object.
 24  */
 25 
 26 goog.provide('google.ads.common.ServiceSignature');
 27 
 28 /**
 29  * Defines an API signature object. This class is used as a support class
 30  * to assist AdsUser in creating a service object.
 31  *
 32  * @constructor
 33  * @param {string} version Service version.
 34  * @param {string} serviceName Service name.
 35  * @param {string} protocol The supported protocol.
 36  */
 37 google.ads.common.ServiceSignature = function(version, serviceName, protocol) {
 38 
 39   /**
 40    * Service version.
 41    * @type {string}
 42    */
 43   this.version = version;
 44 
 45   /**
 46    * Service name.
 47    * @type {string}
 48    */
 49   this.serviceName = serviceName;
 50 
 51   /**
 52    * Service protocol.
 53    * @type {string}
 54    */
 55   this.protocol = protocol;
 56 };
 57 
 58 /**
 59  * The various protocols supported by the library.
 60  */
 61 google.ads.common.ServiceSignature.SupportedProtocols = {
 62   SOAP: 'SOAP'
 63 };
 64 
 65 /**
 66  * Gets a unique id to distinguish the service represented by this signature
 67  * object.
 68  *
 69  * @return {string} The id.
 70  */
 71 google.ads.common.ServiceSignature.prototype.getId = function() {
 72   return this.version + '.' + this.serviceName;
 73 };
 74 
 75 /**
 76  * Gets the service version.
 77  *
 78  * @return {string} The service version.
 79  */
 80 google.ads.common.ServiceSignature.prototype.getVersion = function() {
 81   return this.version;
 82 };
 83 
 84 /**
 85  * Gets the service name.
 86  *
 87  * @return {string} The service name.
 88  */
 89 google.ads.common.ServiceSignature.prototype.getServiceName =
 90     function() {
 91   return this.serviceName;
 92 };
 93 
 94 /**
 95  * Gets the type of service.
 96  *
 97  * @return {google.system.soap.ServiceType} The service type.
 98  */
 99  google.ads.common.ServiceSignature.prototype.getServiceType =
100     function() {
101    return null;
102 };
103 
104 /**
105  * Gets the supported protocol.
106  *
107  * @return {string} The supported protocol.
108  */
109  google.ads.common.ServiceSignature.prototype.getSupportedProtocol =
110     function() {
111   return this.protocol;
112 };
