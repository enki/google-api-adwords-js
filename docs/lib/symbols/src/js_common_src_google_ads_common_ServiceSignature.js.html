 20 /**
 21  * @fileoverview Defines ServiceSignature, an API signature object. This class
 22  * is used as a support class to assist AdsUser in creating a service object.
 23  */
 24 
 25 goog.provide('google.ads.common.ServiceSignature');
 26 
 27 /**
 28  * Defines an API signature object. This class is used as a support class
 29  * to assist AdsUser in creating a service object.
 30  *
 31  * @constructor
 32  * @param {string} version Service version.
 33  * @param {string} serviceName Service name.
 34  * @param {string} protocol The supported protocol.
 35  */
 36 google.ads.common.ServiceSignature = function(version, serviceName, protocol) {
 37   this.version = version;
 38   this.serviceName = serviceName;
 39   this.protocol = protocol;
 40 };
 41 
 42 /**
 43  * The various protocols supported by the library.
 44  */
 45 google.ads.common.ServiceSignature.SupportedProtocols = {
 46   SOAP: 'SOAP'
 47 };
 48 
 49 /**
 50  * Gets a unique id to distinguish the service represented by this signature
 51  * object.
 52  *
 53  * @return {string} The id.
 54  */
 55 google.ads.common.ServiceSignature.prototype.getId = function() {
 56   return this.version + '.' + this.serviceName;
 57 };
 58 
 59 /**
 60  * Gets the service version.
 61  *
 62  * @return {string} The service version.
 63  */
 64 google.ads.common.ServiceSignature.prototype.getVersion = function() {
 65   return this.version;
 66 };
 67 
 68 /**
 69  * Gets the service name.
 70  *
 71  * @return {string} The service name.
 72  */
 73 google.ads.common.ServiceSignature.prototype.getServiceName =
 74     function() {
 75   return this.serviceName;
 76 };
 77 
 78 /**
 79  * Gets the type of service.
 80  *
 81  * @return {google.system.soap.ServiceType} The service type.
 82  */
 83  google.ads.common.ServiceSignature.prototype.getServiceType =
 84     function() {
 85    return null;
 86 };
 87 
 88 /**
 89  * Gets the supported protocol.
 90  *
 91  * @return {string} The supported protocol.
 92  */
 93  google.ads.common.ServiceSignature.prototype.getSupportedProtocol =
 94     function() {
 95   return this.protocol;
 96 };
