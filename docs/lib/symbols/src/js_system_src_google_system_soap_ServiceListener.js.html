 20 /**
 21  * @fileoverview Defines google.system.soap.ServiceListener class.
 22  */
 23 
 24 goog.provide('google.system.soap.ServiceListener');
 25 
 26 /**
 27  * Defines a SOAP Listener contract, basic listener object.
 28  *
 29  * It is meant to be specialized and its method overridden by concrete
 30  * listeners.
 31  *
 32  * @constructor
 33  */
 34 google.system.soap.ServiceListener = function() {};
 35 
 36 /**
 37  * Method to be called before the serialization takes places. Listeners have
 38  * the opportunity at this time to modify the called services and parameters.
 39  *
 40  * Implementation of this method is optional.
 41  *
 42  * @param {google.system.soap.SoapService} service Service been invoked.
 43  * @param {string} methodName Method been invoked.
 44  * @param {Object<string,Object>} parameterValues Parameters been passed to
 45  * the call.
 46  * @param {function()} continueCallback Function to be called in case of
 47  * success.
 48  * @param {function(google.system.soap.Exception)} errorCallback Function
 49  * to be called in case an error has occured a SOAP Exception object is expected
 50  * by this function.
 51  */
 52 google.system.soap.ServiceListener.prototype.beforeSerialize = function(
 53     service, methodName, parameterValues, continueCallback, errorCallback) {
 54   continueCallback();
 55 };
 56 
 57 /**
 58  * Method to be called after the service call gets serialized in to XML Document
 59  * but before gets send to the server. Gives the oportunnity to the listener
 60  * to revise the document or even modify before gets sent.
 61  *
 62  * Implementation of this method is optional.
 63  *
 64  * @param {google.system.soap.SoapService} service Service been invoked.
 65  * @param {Document} document The XML document to be sent to the server.
 66  * @param {function()} continueCallback Function to be called in case of
 67  * success.
 68  * @param {function(google.system.soap.Exception)} errorCallback Function
 69  * to be called in case an error has occurred a SOAP Exception object is
 70  * expected by this function.
 71  */
 72 google.system.soap.ServiceListener.prototype.beforeSend = function(
 73     service, document, continueCallback, errorCallback) {
 74   continueCallback();
 75 };
 76 
 77 /**
 78  * Method to be called after the request has been sent and a an XML response
 79  * has been received. Gives the opportunity for the listener to parse and even
 80  * modify the XML document if desired.
 81  *
 82  * Implementation of this method is optional.
 83  *
 84  * @param {google.system.soap.SoapService} service Service been invoked.
 85  * @param {Document} responseXml The XML response document.
 86  * @param {boolean} isError True if the response XML is SOAP exception returned
 87  * by the server.
 88  * @param {function()} continueCallback Function to be called in case of
 89  * success.
 90  * @param {function(google.system.soap.Exception)} errorCallback Function
 91  * to be called in case an error has occurred a SOAP Exception object is
 92  * expected by this function.
 93  */
 94 google.system.soap.ServiceListener.prototype.beforeDeserialize = function(
 95     service, responseXml, isError, continueCallback, errorCallback) {
 96   continueCallback();
 97 };
 98 
 99 /**
100  * Method to be called after the response have been parsed but before the
101  * resulting object is passed back to the client code. Gives the opportunity
102  * for the listener to modify the resulting object before it gets back to the
103  * client.
104  *
105  * Implementation of this method is optional.
106  *
107  * @param {google.system.soap.SoapService} service Service been invoked.
108  * @param {google.system.soap.SoapObject} soapObject The SOAP Object.
109  * @param {boolean} isError True if the SOAP Object contains a SOAP Exception
110  * returned by the server.
111  * @param {function()} continueCallback Function to be called in case of
112  * success.
113  * @param {function(google.system.soap.Exception)} errorCallback Function
114  * to be called in case an error has occurred a SOAP Exception object is
115  * expected by this function.
116  */
117 google.system.soap.ServiceListener.prototype.beforeReturn = function(
118     service, soapObject, isError, continueCallback, errorCallback) {
119   continueCallback();
120 };
