 19 /**
 20  * @fileoverview Defines GetAllNegativeCampaignCriteria, a code example that
 21  * gets all negative campaign criteria in an account.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201101.GetAllNegativeCampaignCriteria');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201101');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201101.CampaignCriterionService');
 29 goog.require('google.ads.adwords.v201101.Keyword');
 30 goog.require('google.ads.adwords.v201101.Placement');
 31 goog.require('google.ads.adwords.v201101.Selector');
 32 
 33 /**
 34  * This code example retrieves all negative campaign criteria in an account.
 35  * To add a negative campaign criterion, run AddNegativeCampaignCriterion.js.
 36  *
 37  * Tags: CampaignCriterionService.get
 38  *
 39  * @extends google.ads.adwords.examples.ExampleBase
 40  * @constructor
 41  */
 42 google.ads.adwords.examples.v201101.GetAllNegativeCampaignCriteria =
 43     function() {
 44   google.ads.adwords.examples.ExampleBase.call(this);
 45   this.description = 'This code example retrieves all negative campaign ' +
 46       'criteria in an account. To add a negative campaign criterion, run ' +
 47       'AddNegativeCampaignCriterion.js.';
 48 };
 49 goog.inherits(
 50     google.ads.adwords.examples.v201101.GetAllNegativeCampaignCriteria,
 51     google.ads.adwords.examples.ExampleBase);
 52 
 53 /**
 54  * Runs the code example.
 55  *
 56  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 57  * example.
 58  * @param {function} callback the callback method to be called once this example
 59  * is complete.
 60  */
 61 google.ads.adwords.examples.v201101.GetAllNegativeCampaignCriteria.prototype.
 62     run = function(user, callback) {
 63   // Get the CampaignCriterionService.
 64   var campaignCriterionService = user.getService(
 65       google.ads.adwords.AdWordsService.v201101.CampaignCriterionService);
 66 
 67   var selector = new google.ads.adwords.v201101.Selector();
 68   selector.fields = ['Id', 'CampaignId', 'KeywordText', 'PlacementUrl'];
 69 
 70   try {
 71     campaignCriterionService.get(selector,
 72       goog.bind(function(page) {
 73         if (page && page.entries && page.entries.length > 0) {
 74           for (var i = 0, len = page.entries.length; i < len; i++) {
 75             var campaignCriterion = page.entries[i];
 76             var criterion = campaignCriterion.criterion;
 77             if (criterion instanceof google.ads.adwords.v201101.Keyword) {
 78               this.writeOutput('Negative keyword campaign criterion with ' +
 79                   'campaign ID = "%d", criterion ID = "%d", and text = "%s" ' +
 80                   'was found.', campaignCriterion.campaignId, criterion.id,
 81                   criterion.text);
 82             } else if (criterion instanceof
 83                 google.ads.adwords.v201101.Placement) {
 84               this.writeOutput('Negative placement campaign criterion with ' +
 85                   'campaign ID = "%d", criterion ID = "%d" and url = "%s" ' +
 86                   'was found.', campaignCriterion.campaignId, criterion.id,
 87                   criterion.url);
 88             } else {
 89               this.writeOutput('Negative campaign criterion with campaign ' +
 90                   'ID = "%d", criterion ID = "%d" and type = "%s" was found.',
 91                   campaignCriterion.campaignId, criterion.id,
 92                   criterion.criterionType);
 93             }
 94           }
 95         } else {
 96           this.writeOutput('No negative campaign criteria were found.');
 97         }
 98         callback();
 99       }, this),
100       goog.bind(function(soapException) {
101         this.writeOutput('Failed to retrieve all negative campaign criteria. ' +
102             'Soap Fault says "%s"',
103             soapException.getInnerException().getFaultString());
104         callback();
105       }, this)
106     );
107   } catch (ex) {
108     this.writeOutput('An exception occurred while running the code example.');
109     callback();
110   }
111 };
