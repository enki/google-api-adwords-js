 40 /**
 41  * This code example illustrates how to create a logical user list.
 42  *
 43  * Tags: UserListService.mutate
 44  *
 45  * @extends google.ads.adwords.examples.ExampleBase
 46  * @constructor
 47  */
 48 google.ads.adwords.examples.v201101.AddLogicalUserList = function() {
 49   google.ads.adwords.examples.ExampleBase.call(this);
 50   this.description = 'This code example illustrates how to create a logical ' +
 51       'user list.';
 52 };
 53 goog.inherits(google.ads.adwords.examples.v201101.AddLogicalUserList,
 54     google.ads.adwords.examples.ExampleBase);
 55 
 56 /**
 57  * Runs the code example.
 58  *
 59  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 60  * example.
 61  * @param {function} callback the callback method to be called once this example
 62  * is complete.
 63  */
 64 google.ads.adwords.examples.v201101.AddLogicalUserList.prototype.run =
 65     function(user, callback) {
 66   // Get the UserListService.
 67   var userListService = user.getService(
 68       google.ads.adwords.AdWordsService.v201101.UserListService);
 69 
 70   var userList = new google.ads.adwords.v201101.LogicalUserList();
 71   userList.name = 'Mars cruise customers #' + this.getTimeStamp();
 72   userList.description = 'A list of mars cruise customers in the last year.';
 73   userList.status = google.ads.adwords.v201101.UserListMembershipStatus.OPEN;
 74   userList.membershipLifeSpan = 365;
 75 
 76   // Make an UserInterest group for Travel > Cruises & Charters. See
 77   // http://code.google.com/apis/adwords/docs/appendix/verticals.html for
 78   // various verticals and their ids.
 79   var interest = new google.ads.adwords.v201101.UserInterest();
 80   interest.name = 'Mars cruise interest group';
 81   interest.id = 206;
 82 
 83   var userListOperand = new google.ads.adwords.v201101.LogicalUserListOperand();
 84   userListOperand.item = interest;
 85 
 86   var rule = new google.ads.adwords.v201101.UserListLogicalRule();
 87   rule.operator = google.ads.adwords.v201101.UserListLogicalRuleOperator.NONE;
 88   rule.ruleOperands = [userListOperand];
 89   userList.rules = [rule];
 90 
 91   operation = new google.ads.adwords.v201101.UserListOperation();
 92   operation.operand = userList;
 93   operation.operator = google.ads.adwords.v201101.Operator.ADD;
 94 
 95   try {
 96     // Add user list.
 97     userListService.mutate([operation],
 98       goog.bind(function(retval) {
 99         if (retval && retval.value && retval.value.length > 0) {
100           for (var i = 0, len = retval.value.length; i < len; i++) {
101             var valueUserList = retval.value[i];
102             this.writeOutput('Logical user list with name "%s" and id %d ' +
103                 'was added.', valueUserList.name, valueUserList.id);
104           }
105         } else {
106           this.writeOutput('No logical user lists were added.');
107         }
108         callback();
109       }, this),
110       goog.bind(function(soapException) {
111         this.writeOutput('Failed to get add logical user list. Soap Fault ' +
112             'says "%s"', soapException.getInnerException().getFaultString());
113         callback();
114       }, this)
115     );
116   } catch (ex) {
117     this.writeOutput('An exception occurred while running the code example.');
118     callback();
119   }
120 };
