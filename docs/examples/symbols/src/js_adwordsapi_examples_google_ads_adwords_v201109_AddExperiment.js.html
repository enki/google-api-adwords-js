 46 /**
 47  * This code example creates an experiment using a query percentage of 10,
 48  * which defines what fraction of auctions should go to the control split
 49  * (90%) vs. the experiment split (10%), then adds experimental bid changes
 50  * for criteria and ad groups. To get campaigns, run GetAllCampaigns.js.
 51  * To get ad groups, run GetAllAdGroups.js. To get criteria, run
 52  * GetAllAdGroupCriteria.js.
 53  *
 54  * Tags: ExperimentService.mutate
 55  *
 56  * @extends google.ads.adwords.examples.ExampleBase
 57  * @constructor
 58  */
 59 google.ads.adwords.examples.v201109.AddExperiment = function() {
 60   google.ads.adwords.examples.ExampleBase.call(this);
 61   this.description = 'This code example creates an experiment using a query ' +
 62       'percentage of 10, which defines what fraction of auctions should go ' +
 63       'to the control split (90%) vs. the experiment split (10%), then adds ' +
 64       'experimental bid changes for criteria and ad groups. To get ' +
 65       'campaigns, run GetAllCampaigns.js. To get ad groups, run ' +
 66       'GetAllAdGroups.cs. To get criteria, run GetAllAdGroupCriteria.js.';
 67 };
 68 goog.inherits(google.ads.adwords.examples.v201109.AddExperiment,
 69     google.ads.adwords.examples.ExampleBase);

 71 /**
 72  * Runs the code example.
 73  *
 74  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 75  *     example.
 76  * @param {function} callback the callback method to be called once this example
 77  *     is complete.
 78  */
 79 google.ads.adwords.examples.v201109.AddExperiment.prototype.run =
 80     function(user, callback) {
 81   // Get the ExperimentService.
 82   var experimentService = user.getService(
 83       google.ads.adwords.AdWordsService.v201109.ExperimentService);
 84 
 85   // Get the AdGroupService.
 86   var adGroupService = user.getService(
 87       google.ads.adwords.AdWordsService.v201109.AdGroupService);
 88 
 89   // Get the AdGroupCriterionService.
 90   var adGroupCriterionService = user.getService(
 91       google.ads.adwords.AdWordsService.v201109.AdGroupCriterionService);
 92 
 93   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 94   var adGroupId = this._T('INSERT_AD_GROUP_ID_HERE');
 95   var criterionId = this._T('INSERT_CRITERION_ID_HERE');
 96 
 97   // Create experiment.
 98   var experiment = new google.ads.adwords.v201109.Experiment();
 99   experiment.campaignId = campaignId;
100   experiment.name = 'Interplanetary Cruise #' + this.getTimeStamp();
101   experiment.queryPercentage = 10;
102   experiment.startDateTime = this.getToday();
103 
104   // Create operation.
105   var experimentOperation =
106       new google.ads.adwords.v201109.ExperimentOperation();
107   experimentOperation.operator = google.ads.adwords.v201109.Operator.ADD;
108   experimentOperation.operand = experiment;
109 
110   try {
111     // Add experiment.
112     experimentService.mutate([experimentOperation],
113       goog.bind(function(experimentRetVal) {
114         if (experimentRetVal && experimentRetVal.value &&
115             experimentRetVal.value.length > 0) {
116           var tempExperiment = experimentRetVal.value[0];
117           this.writeOutput('Experiment with name = "%s" and id = "%d" was ' +
118               'added.', tempExperiment.name, tempExperiment.id);
119           var experimentId = tempExperiment.id;
120 
121           // Set ad group for the experiment.
122           var adGroup = new google.ads.adwords.v201109.AdGroup();
123           adGroup.id = adGroupId;
124 
125           // Create experiment bid multiplier rule that will modify ad group bid
126           // for the experiment.
127           var adGroupBidMultiplier = new google.ads.adwords.v201109.
128               ManualCPCAdGroupExperimentBidMultipliers();
129           adGroupBidMultiplier.maxCpcMultiplier =
130               new google.ads.adwords.v201109.BidMultiplier();
131           adGroupBidMultiplier.maxCpcMultiplier.multiplier = 1.5;
132 
133           // Set experiment data to the ad group.
134           var adGroupExperimentData =
135               new google.ads.adwords.v201109.AdGroupExperimentData();
136           adGroupExperimentData.experimentId = experimentId;
137           adGroupExperimentData.experimentDeltaStatus =
138               google.ads.adwords.v201109.ExperimentDeltaStatus.MODIFIED;
139           adGroupExperimentData.experimentBidMultipliers = adGroupBidMultiplier;
140 
141           adGroup.experimentData = adGroupExperimentData;
142 
143           // Create operation.
144           var adGroupOperation =
145               new google.ads.adwords.v201109.AdGroupOperation();
146           adGroupOperation.operand = adGroup;
147           adGroupOperation.operator = google.ads.adwords.v201109.Operator.SET;
148 
149           // Update ad group.
150           adGroupService.mutate([adGroupOperation],
151             goog.bind(function(retval) {
152               // Display results.
153               if (retval && retval.value) {
154                 for (var i = 0, len = retval.value.length; i < len; i++) {
155                   var tempAdGroup = retval.value[i];
156                   this.writeOutput('Ad group with name = "%s", id = "%d" ' +
157                       'and status = "%s" was updated for the experiment.',
158                       tempAdGroup.name, tempAdGroup.id, tempAdGroup.status);
159                 }
160               } else {
161                 this.writeOutput('No ad groups were updated.');
162               }
163 
164               // Set ad group criteria for the experiment.
165               var criterion = new google.ads.adwords.v201109.Criterion();
166               criterion.id = criterionId;
167 
168               var adGroupCriterion =
169                   new google.ads.adwords.v201109.BiddableAdGroupCriterion();
170               adGroupCriterion.adGroupId = adGroupId;
171               adGroupCriterion.criterion = criterion;
172 
173               // Create experiment bid multiplier rule that will modify
174               // criterion bid for the experiment.
175               var bidMultiplier = new google.ads.adwords.v201109.
176                   ManualCPCAdGroupCriterionExperimentBidMultiplier();
177               bidMultiplier.maxCpcMultiplier =
178                   new google.ads.adwords.v201109.BidMultiplier();
179               bidMultiplier.maxCpcMultiplier.multiplier = 1.5;
180 
181               // Set experiment data to the criterion.
182               var adGroupCriterionExperimentData = new google.ads.adwords.
183                   v201109.BiddableAdGroupCriterionExperimentData();
184               adGroupCriterionExperimentData.experimentId = experimentId;
185               adGroupCriterionExperimentData.experimentDeltaStatus =
186                   google.ads.adwords.v201109.ExperimentDeltaStatus.MODIFIED;
187               adGroupCriterionExperimentData.experimentBidMultiplier =
188                   bidMultiplier;
189 
190               adGroupCriterion.experimentData = adGroupCriterionExperimentData;
191 
192               // Create operation.
193               var adGroupCriterionOperation =
194                   new google.ads.adwords.v201109.AdGroupCriterionOperation();
195               adGroupCriterionOperation.operand = adGroupCriterion;
196               adGroupCriterionOperation.operator =
197                   google.ads.adwords.v201109.Operator.SET;
198 
199               // Update ad group criteria.
200               adGroupCriterionService.mutate([adGroupCriterionOperation],
201                 goog.bind(function(adGroupCriterionRetVal) {
202                   // Display results.
203                   if (adGroupCriterionRetVal && adGroupCriterionRetVal.value) {
204                     for (var i = 0, len = adGroupCriterionRetVal.value.length;
205                         i < len; i++) {
206                       var tempAgc = adGroupCriterionRetVal.value[i];
207                       this.writeOutput('Ad group criterion with ad group id =' +
208                           ' "%d", criterion id = "%d" and type = "%s" was ' +
209                           'updated for the experiment.',
210                           tempAgc.adGroupId, tempAgc.criterion.id,
211                           tempAgc.criterion.criterionType);
212                     }
213                   } else {
214                     this.writeOutput('No ad group criteria were updated.');
215                   }
216                   callback();
217                 }, this),
218                 goog.bind(function(soapException) {
219                   this.writeOutput('Failed to update ad group criterion. ' +
220                       'Soap Fault says "%s"',
221                       soapException.getInnerException().getFaultString());
222                   callback();
223                 }, this)
224               );
225               callback();
226             }, this),
227             goog.bind(function(soapException) {
228               this.writeOutput('Failed to update ad group.Soap Fault says "%s"',
229                   soapException.getInnerException().getFaultString());
230               callback();
231             }, this)
232           );
233         } else {
234           this.writeOutput('No experiments were added.');
235         }
236         callback();
237       }, this),
238       goog.bind(function(soapException) {
239         this.writeOutput('Failed to add experiment. Soap Fault says "%s"',
240             soapException.getInnerException().getFaultString());
241         callback();
242       }, this)
243     );
244   } catch (ex) {
245     this.writeOutput('An exception occurred while running the code example.');
246     callback();
247   }
248 };

250 /**
251  * Gets today's date as a string of the format yyyymmdd hhmmss.
252  *
253  * @return {string} today's date as a string of the format yyyymmdd hhmmss.
254  */
255 google.ads.adwords.examples.v201109.AddExperiment.prototype.getToday =
256     function() {
257   var currentTime = new Date();
258   var month = currentTime.getMonth() + 1;
259   var day = currentTime.getDate();
260   var year = currentTime.getFullYear();
261   var hours = currentTime.getHours();
262   var minutes = currentTime.getMinutes();
263   var seconds = currentTime.getSeconds();
264 
265   return goog.string.format('%d%02d%02d %02d%02d%02d', year, month, day,
266       hours, minutes, seconds);
267 };
