 19 /**
 20  * @fileoverview Defines SetCampaignTargets, a code example that adds geo,
 21  * language, and network targeting to an existing campaign.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201101.SetCampaignTargets');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201101');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201101.CampaignTargetOperation');
 29 goog.require('google.ads.adwords.v201101.CampaignTargetService');
 30 goog.require('google.ads.adwords.v201101.CountryTarget');
 31 goog.require('google.ads.adwords.v201101.GeoTargetList');
 32 goog.require('google.ads.adwords.v201101.LanguageTarget');
 33 goog.require('google.ads.adwords.v201101.LanguageTargetList');
 34 goog.require('google.ads.adwords.v201101.Operator');
 35 
 36 /**
 37  * This code example adds geo, language, and network targeting to an
 38  * existing campaign. To get a campaign, run GetAllCampaigns.js.
 39  *
 40  * Tags: CampaignTargetService.mutate
 41  *
 42  * @extends google.ads.adwords.examples.ExampleBase
 43  * @constructor
 44  */
 45 google.ads.adwords.examples.v201101.SetCampaignTargets = function() {
 46   google.ads.adwords.examples.ExampleBase.call(this);
 47   this.description = 'This code example adds geo, language, and network ' +
 48       'targeting to an existing campaign. To get a campaign, run ' +
 49       'GetAllCampaigns.js.';
 50 };
 51 goog.inherits(google.ads.adwords.examples.v201101.SetCampaignTargets,
 52     google.ads.adwords.examples.ExampleBase);
 53 
 54 /**
 55  * Runs the code example.
 56  *
 57  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 58  * example.
 59  * @param {function} callback the callback method to be called once this example
 60  * is complete.
 61  */
 62 google.ads.adwords.examples.v201101.SetCampaignTargets.prototype.run =
 63     function(user, callback) {
 64   // Get the CampaignTargetService.
 65 
 66   var campaignTargetService = user.getService(
 67       google.ads.adwords.AdWordsService.v201101.CampaignTargetService);
 68 
 69   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 70 
 71   // Create language targets.
 72   var langTargetList = new google.ads.adwords.v201101.LanguageTargetList();
 73   langTargetList.campaignId = campaignId;
 74 
 75   var langTarget1 = new google.ads.adwords.v201101.LanguageTarget();
 76   langTarget1.languageCode = 'en';
 77 
 78   var langTarget2 = new google.ads.adwords.v201101.LanguageTarget();
 79   langTarget2.languageCode = 'ja';
 80 
 81   langTargetList.targets = [langTarget1, langTarget2];
 82 
 83   // Create language target set operation.
 84   var langTargetOperation =
 85       new google.ads.adwords.v201101.CampaignTargetOperation();
 86   langTargetOperation.operator = google.ads.adwords.v201101.Operator.SET;
 87   langTargetOperation.operand = langTargetList;
 88 
 89   // Create geo targets.
 90   var geoTargetList = new google.ads.adwords.v201101.GeoTargetList();
 91   geoTargetList.campaignId = campaignId;
 92 
 93   var geoTarget1 = new google.ads.adwords.v201101.CountryTarget();
 94   geoTarget1.countryCode = 'US';
 95 
 96   var geoTarget2 = new google.ads.adwords.v201101.CountryTarget();
 97   geoTarget2.countryCode = 'JP';
 98 
 99   // Create geo target set operation.
100   var geoTargetOperation =
101       new google.ads.adwords.v201101.CampaignTargetOperation();
102   geoTargetOperation.operator = google.ads.adwords.v201101.Operator.SET;
103   geoTargetOperation.operand = geoTargetList;
104 
105   try {
106     campaignTargetService.mutate(
107       [geoTargetOperation, langTargetOperation],
108       goog.bind(function(retval) {
109         // Display campaign targets.
110         if (retval && retval.value && retval.value.length > 0) {
111           for (var i = 0, len = retval.value.length; i < len; i++) {
112             var targetList = retval.value[i];
113             this.writeOutput('Campaign target of type "%s" was set to ' +
114                 'Campaign with id = "%s".', targetList.TargetListType,
115                 targetList.campaignId);
116           }
117         } else {
118           this.writeOutput('No campaign target(s) were set.');
119         }
120         callback();
121       }, this),
122       goog.bind(function(soapException) {
123         this.writeOutput('Failed to set campaign target(s). Soap Fault says ' +
124             '"%s"', soapException.getInnerException().getFaultString());
125         callback();
126       }, this)
127     );
128   } catch (ex) {
129     this.writeOutput('An exception occurred while running the code example.');
130     callback();
131   }
132 };
