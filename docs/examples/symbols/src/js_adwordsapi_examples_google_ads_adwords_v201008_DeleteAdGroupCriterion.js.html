 19 /**
 20  * @fileoverview Defines DeleteAdGroupCriterion, a code example that deletes an
 21  * ad group criterion using the 'REMOVE operator.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.DeleteAdGroupCriterion');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.AdGroupCriterionOperation');
 29 goog.require('google.ads.adwords.v201008.AdGroupCriterionService');
 30 goog.require('google.ads.adwords.v201008.BiddableAdGroupCriterion');
 31 goog.require('google.ads.adwords.v201008.Keyword');
 32 goog.require('google.ads.adwords.v201008.KeywordMatchType');
 33 goog.require('google.ads.adwords.v201008.Operator');

 35 /**
 36  * This code example deletes an ad group criterion using the 'REMOVE'
 37  * operator. To get ad group criteria, run GetAllAdGroupCriteria.js.
 38  *
 39  * Tags: AdGroupCriterionService.mutate
 40  *
 41  * @extends google.ads.adwords.examples.ExampleBase
 42  * @constructor
 43  */
 44 
 45 google.ads.adwords.examples.v201008.DeleteAdGroupCriterion = function() {
 46   google.ads.adwords.examples.ExampleBase.call(this);
 47   this.description = 'This code example deletes an ad group criterion using ' +
 48       'the "REMOVE" operator. To get ad group criteria, run ' +
 49       'GetAllAdGroupCriteria.js.';
 50 };
 51 goog.inherits(google.ads.adwords.examples.v201008.DeleteAdGroupCriterion,
 52     google.ads.adwords.examples.ExampleBase);
 53 
 54 /**
 55  * Runs the code example.
 56  *
 57  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 58  * example.
 59  * @param {function} callback the callback method to be called once this example
 60  * is complete.
 61  */
 62 google.ads.adwords.examples.v201008.DeleteAdGroupCriterion.prototype.run =
 63     function(user, callback) {
 64   // Get the AdGroupCriterionService.
 65   var adGroupCriterionService = user.getService(
 66       google.ads.adwords.AdWordsService.v201008.AdGroupCriterionService);
 67 
 68   var adGroupId = this._T('INSERT_AD_GROUP_ID_HERE');
 69   var criterionId = this._T('INSERT_CRITERION_ID_HERE');
 70 
 71   // Create base class criterion to avoid setting keyword and placement specific
 72   // fields.
 73   var criterion = new google.ads.adwords.v201008.Criterion();
 74   criterion.id = criterionId;
 75 
 76   // Create ad group criterion.
 77   var adGroupCriterion =
 78       new google.ads.adwords.v201008.BiddableAdGroupCriterion();
 79   adGroupCriterion.adGroupId = adGroupId;
 80   adGroupCriterion.criterion = criterion;
 81 
 82   // Create operations.
 83   var operation = new google.ads.adwords.v201008.AdGroupCriterionOperation();
 84   operation.operand = adGroupCriterion;
 85   operation.operator = google.ads.adwords.v201008.Operator.REMOVE;
 86 
 87   try {
 88     adGroupCriterionService.mutate([operation],
 89       goog.bind(function(retval) {
 90         if (retval && retval.value) {
 91           for (var i = 0, len = retval.value.length; i < len; i++) {
 92             var adGroupCriterion = retval.value[i];
 93             this.writeOutput('Ad group criterion with ad group id = "%s", ' +
 94                 'criterion id = "%s" and type = "%s" was deleted.',
 95                 adGroupCriterion.adGroupId, adGroupCriterion.criterion.id,
 96                 adGroupCriterion.criterion.CriterionType);
 97           }
 98         } else {
 99           this.writeOutput('No ad group criteria were deleted.\n');
100         }
101         callback();
102       }, this),
103       goog.bind(function(soapException) {
104         this.writeOutput('Failed to delete ad group criteria. Soap Fault ' +
105             'says "%s"', soapException.getInnerException().getFaultString());
106         callback();
107       }, this)
108     );
109   } catch (ex) {
110     this.writeOutput('An exception occurred while running the code example.');
111     callback();
112   }
113 };
