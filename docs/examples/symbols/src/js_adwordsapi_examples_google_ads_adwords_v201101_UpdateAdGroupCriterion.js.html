 19 /**
 20  * @fileoverview Defines UpdateAdGroupCriterion, a code example that updates
 21  * the bid of an ad group criterion.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201101.UpdateAdGroupCriterion');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201101');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201101.AdGroupCriterionOperation');
 29 goog.require('google.ads.adwords.v201101.AdGroupCriterionService');
 30 goog.require('google.ads.adwords.v201101.Bid');
 31 goog.require('google.ads.adwords.v201101.BiddableAdGroupCriterion');
 32 goog.require('google.ads.adwords.v201101.Criterion');
 33 goog.require('google.ads.adwords.v201101.ManualCPCAdGroupCriterionBids');
 34 goog.require('google.ads.adwords.v201101.Money');

 37 /**
 38  * This code example updates the bid of an ad group criterion. To get
 39  * ad group criteria, run GetAllAdGroupCriteria.js.
 40  *
 41  * Tags: AdGroupCriterionService.mutate
 42  *
 43  * @extends google.ads.adwords.examples.ExampleBase
 44  * @constructor
 45  */
 46 google.ads.adwords.examples.v201101.UpdateAdGroupCriterion = function() {
 47   google.ads.adwords.examples.ExampleBase.call(this);
 48   this.description = 'This code example updates the bid of an ad group ' +
 49       'criterion. To get ad group criteria, run GetAllAdGroupCriteria.js.';
 50 };
 51 goog.inherits(google.ads.adwords.examples.v201101.UpdateAdGroupCriterion,
 52     google.ads.adwords.examples.ExampleBase);
 53 
 54 /**
 55  * Runs the code example.
 56  *
 57  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 58  * example.
 59  * @param {function} callback the callback method to be called once this example
 60  * is complete.
 61  */
 62 google.ads.adwords.examples.v201101.UpdateAdGroupCriterion.prototype.run =
 63     function(user, callback) {
 64   // Get the AdGroupCriterionService.
 65   var adGroupCriterionService = user.getService(
 66       google.ads.adwords.AdWordsService.v201101.AdGroupCriterionService);
 67 
 68   var adGroupId = this._T('INSERT_AD_GROUP_ID_HERE');
 69   var criterionId = this._T('INSERT_CRITERION_ID_HERE');
 70 
 71   var criterion = new google.ads.adwords.v201101.Criterion();
 72   criterion.id = criterionId;
 73 
 74   // Create ad group criterion.
 75   var biddableAdGroupCriterion =
 76       new google.ads.adwords.v201101.BiddableAdGroupCriterion();
 77   biddableAdGroupCriterion.adGroupId = adGroupId;
 78   biddableAdGroupCriterion.criterion = criterion;
 79 
 80   // Create bids.
 81   var bids = new google.ads.adwords.v201101.ManualCPCAdGroupCriterionBids();
 82   bids.maxCpc = new google.ads.adwords.v201101.Bid();
 83   bids.maxCpc.amount = new google.ads.adwords.v201101.Money();
 84   bids.maxCpc.amount.microAmount = 10000;
 85 
 86   biddableAdGroupCriterion.bids = bids;
 87 
 88   // Create operations.
 89   var operation = new google.ads.adwords.v201101.AdGroupCriterionOperation();
 90   operation.operator = google.ads.adwords.v201101.Operator.SET;
 91   operation.operand = biddableAdGroupCriterion;
 92 
 93   try {
 94     adGroupCriterionService.mutate([operation],
 95       goog.bind(function(retval) {
 96         if (retval && retval.value) {
 97           for (var i = 0, len = retval.value.length; i < len; i++) {
 98             var adGroupCriterion = retval.value[i];
 99             var bidAmount = 0;
100             // This check is not really necessary here, but it shows how you
101             // can write code to check the type of a field and take action
102             // accordingly.
103             if (adGroupCriterion instanceof google.ads.adwords.v201101.
104                 BiddableAdGroupCriterion) {
105               bidAmount = adGroupCriterion.bids.maxCpc.amount.microAmount;
106             }
107 
108             this.writeOutput('Ad group criterion with ad group id = "%s", ' +
109                 'criterion id = "%s" and type = "%s" was updated with bid ' +
110                 'amount = "%s" micros.',
111                 adGroupCriterion.adGroupId, adGroupCriterion.criterion.id,
112                 adGroupCriterion.criterion.CriterionType, bidAmount);
113           }
114         } else {
115           this.writeOutput('No ad group criteria were updated.\n');
116         }
117         callback();
118       }, this),
119       goog.bind(function(soapException) {
120         this.writeOutput('Failed to update ad group criteria. Soap Fault ' +
121             'says "%s"', soapException.getInnerException().getFaultString());
122         callback();
123       }, this)
124     );
125   } catch (ex) {
126     this.writeOutput('An exception occurred while running the code example.');
127     callback();
128   }
129 };
