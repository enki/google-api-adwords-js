 19 /**
 20  * @fileoverview Defines DeleteSiteLinks, a code example that shows how to
 21  *     delete site links from an existing campaign.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201101.DeleteSiteLinks');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201101');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201101.AdExtension');
 29 goog.require('google.ads.adwords.v201101.CampaignAdExtension');
 30 goog.require('google.ads.adwords.v201101.CampaignAdExtensionOperation');
 31 goog.require('google.ads.adwords.v201101.CampaignAdExtensionService');
 32 goog.require('google.ads.adwords.v201101.CampaignAdExtensionStatus');
 33 goog.require('google.ads.adwords.v201101.Operator');
 34 goog.require('google.ads.adwords.v201101.Predicate');
 35 goog.require('google.ads.adwords.v201101.PredicateOperator');
 36 goog.require('google.ads.adwords.v201101.Selector');
 37 goog.require('google.ads.adwords.v201101.Sitelink');
 38 goog.require('google.ads.adwords.v201101.SitelinksExtension');

 40 /**
 41  * This code example shows how to delete site links from an existing
 42  * campaign. To add site links to an existing campaign, run AddSiteLinks.js.
 43  * To get existing campaigns, run GetAllCampaigns.js.
 44  *
 45  * Tags: CampaignAdExtensionService.mutate
 46  *
 47  * @extends google.ads.adwords.examples.ExampleBase
 48  * @constructor
 49  */
 50 
 51 google.ads.adwords.examples.v201101.DeleteSiteLinks = function() {
 52   google.ads.adwords.examples.ExampleBase.call(this);
 53   this.description = 'This code example shows how to delete site links from ' +
 54       'an existing campaign. To add site links to an existing campaign, ' +
 55       'run AddSiteLinks.js. To get existing campaigns, run GetAllCampaigns.js.';
 56 };
 57 goog.inherits(google.ads.adwords.examples.v201101.DeleteSiteLinks,
 58     google.ads.adwords.examples.ExampleBase);
 59 
 60 /**
 61  * Runs the code example.
 62  *
 63  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 64  *     example.
 65  * @param {function} callback the callback method to be called once this example
 66  *     is complete.
 67  */
 68 google.ads.adwords.examples.v201101.DeleteSiteLinks.prototype.run = function(
 69     user, callback) {
 70   // Get the CampaignAdExtensionService.
 71   var campaignAdExtensionService = user.getService(
 72       google.ads.adwords.AdWordsService.v201101.CampaignAdExtensionService);
 73 
 74   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 75   var siteLinkExtensionId = -1;
 76 
 77   // Get the campaign ad extension containing sitelinks.
 78   var selector = new google.ads.adwords.v201101.Selector();
 79 
 80   var campaignPredicate = new google.ads.adwords.v201101.Predicate();
 81   campaignPredicate.operator =
 82       google.ads.adwords.v201101.PredicateOperator.EQUALS;
 83   campaignPredicate.field = 'CampaignId';
 84   campaignPredicate.values = [campaignId];
 85 
 86   var statusPredicate = new google.ads.adwords.v201101.Predicate();
 87   statusPredicate.operator =
 88       google.ads.adwords.v201101.PredicateOperator.EQUALS;
 89   statusPredicate.field = 'Status';
 90   statusPredicate.values = [
 91       google.ads.adwords.v201101.CampaignAdExtensionStatus.ACTIVE
 92   ];
 93 
 94   selector.predicates = [campaignPredicate, statusPredicate];
 95   selector.fields = ['AdExtensionId', 'Status'];
 96 
 97   try {
 98     campaignAdExtensionService.get(selector,
 99       goog.bind(function(page) {
100         if (page && page.entries) {
101           for (var i = 0, len = page.entries.length; i < len; i++) {
102             var extension = page.entries[i];
103             if (extension.adExtension instanceof
104                 google.ads.adwords.v201101.SitelinksExtension) {
105               siteLinkExtensionId = extension.adExtension.id;
106               break;
107             }
108           }
109         }
110 
111         if (siteLinkExtensionId == -1) {
112           this.writeOutput('No sitelinks extension for the specified ' +
113               'campaignId');
114           callback();
115           return;
116         }
117 
118         var campaignAdExtension =
119             new google.ads.adwords.v201101.CampaignAdExtension();
120         campaignAdExtension.campaignId = campaignId;
121         campaignAdExtension.adExtension =
122             new google.ads.adwords.v201101.AdExtension();
123         campaignAdExtension.adExtension.id = siteLinkExtensionId;
124 
125         var operation = new google.ads.adwords.v201101.
126             CampaignAdExtensionOperation();
127         operation.operator = google.ads.adwords.v201101.Operator.REMOVE;
128         operation.operand = campaignAdExtension;
129 
130         campaignAdExtensionService.mutate([operation],
131           goog.bind(function(retval) {
132             if (retval && retval.value && retval.value.length > 0) {
133               for (var i = 0, len = retval.value.length; i < len; i++) {
134                 var campaignExtension = retval.value[i];
135                 this.writeOutput('Deleted a campaign ad extension with id = ' +
136                     '"%d" and status = "%s"', campaignExtension.adExtension.id,
137                     campaignExtension.status);
138                 for (var j = 0, len1 = campaignExtension.adExtension.sitelinks.
139                     length; j < len1; j++) {
140                   var siteLink = campaignExtension.adExtension.sitelinks[j];
141                   this.writeOutput('-- Site link text is "%s" and destination' +
142                       ' url is %s', siteLink.displayText,
143                       siteLink.destinationUrl);
144                 }
145               }
146             }
147             callback();
148           }, this),
149           goog.bind(function(soapException) {
150             this.writeOutput('Failed to delete site links. Soap Fault says' +
151                 ' "%s"', soapException.getInnerException().getFaultString());
152             callback();
153           }, this)
154         );
155       }, this),
156       goog.bind(function(soapException) {
157         this.writeOutput('Failed to delete experiment. Soap Fault says "%s"',
158             soapException.getInnerException().getFaultString());
159         callback();
160       }, this)
161     );
162   } catch (ex) {
163     this.writeOutput('An exception occurred while running the code example.');
164     callback();
165   }
166 };
