 20  * @fileoverview Defines GetRelatedPlacements, a code example that retrieves
 21  * urls that have content keywords related to a given website.

 36 /**
 37  * This code example retrieves urls that have content keywords related to a
 38  * given website.
 39  *
 40  * Tags: TargetingIdeaService.get
 41  *
 42  * @extends google.ads.adwords.examples.ExampleBase
 43  * @constructor
 44  */
 45 google.ads.adwords.examples.v201008.GetRelatedPlacements = function() {
 46   google.ads.adwords.examples.ExampleBase.call(this);
 47   this.description = 'This code example retrieves urls that have content ' +
 48       'keywords related to a given website.';
 49 };
 50 goog.inherits(google.ads.adwords.examples.v201008.GetRelatedPlacements,
 51     google.ads.adwords.examples.ExampleBase);
 52 
 53 /**
 54  * Runs the code example.
 55  *
 56  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 57  * example.
 58  * @param {function} callback the callback method to be called once this example
 59  * is complete.
 60  */
 61 google.ads.adwords.examples.v201008.GetRelatedPlacements.prototype.run =
 62     function(user, callback) {
 63   // Get the TargetingIdeaService.
 64   var targetingIdeaService = user.getService(
 65       google.ads.adwords.AdWordsService.v201008.TargetingIdeaService);
 66 
 67   var urlText = 'mars.google.com';
 68 
 69   var searchParameter =
 70       new google.ads.adwords.v201008.RelatedToUrlSearchParameter();
 71   searchParameter.urls = [urlText];
 72   searchParameter.includeSubUrls = false;
 73 
 74   var selector = new google.ads.adwords.v201008.TargetingIdeaSelector();
 75   selector.searchParameters = [searchParameter];
 76   selector.ideaType = google.ads.adwords.v201008.IdeaType.PLACEMENT;
 77   selector.requestType = google.ads.adwords.v201008.RequestType.IDEAS;
 78 
 79   var paging = new google.ads.adwords.v201008.Paging();
 80   paging.startIndex = 0;
 81   paging.numberResults = 10;
 82 
 83   selector.paging = paging;
 84 
 85   try {
 86     targetingIdeaService.get(selector,
 87       goog.bind(function(page) {
 88         if (page && page.entries) {
 89           this.writeOutput('There are a total of %s urls with content ' +
 90               'keywords related to "%s". The first %d entries are displayed ' +
 91               'below:', page.totalNumEntries, urlText, page.entries.length);
 92 
 93           for (var i = 0, len = page.entries.length; i < len; i++) {
 94             var idea = page.entries[i];
 95             for (var j = 0, len1 = idea.data.length; j < len1; j++) {
 96               var entry = idea.data[j];
 97               if (entry.key == google.ads.adwords.v201008.AttributeType.
 98                   PLACEMENT) {
 99                 var placementAttribute = entry.value;
100                 this.writeOutput('Related content keywords were found at ' +
101                     '"%s".', placementAttribute.value.url);
102               }
103             }
104           }
105         } else {
106           this.writeOutput('No urls with content keywords related to your ' +
107               'url were found.');
108         }
109         callback();
110       }, this),
111       goog.bind(function(soapException) {
112         this.writeOutput('Failed to get related placements. Soap Fault ' +
113             'says "%s"', soapException.getInnerException().getFaultString());
114         callback();
115       }, this)
116     );
117   } catch (ex) {
118     this.writeOutput('An exception occurred while running the code example.');
119     callback();
120   }
121 };
