 19 /**
 20  * @fileoverview Defines GetGeoLocationInfo, a code example that gets geo
 21  * location information for addresses.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetGeoLocationInfo');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.Address');
 29 goog.require('google.ads.adwords.v201008.GeoLocationSelector');
 30 goog.require('google.ads.adwords.v201008.GeoLocationService');
 31 goog.require('google.ads.adwords.v201008.InvalidGeoLocation');
 32 
 33 /**
 34  * This code example gets geo location information for addresses.
 35  *
 36  * Tags: GeoLocationService.get
 37  *
 38  * @extends google.ads.adwords.examples.ExampleBase
 39  * @constructor
 40  */
 41 google.ads.adwords.examples.v201008.GetGeoLocationInfo = function() {
 42   google.ads.adwords.examples.ExampleBase.call(this);
 43   this.description = 'This code example gets geo location information for' +
 44       ' addresses.';
 45 };
 46 goog.inherits(google.ads.adwords.examples.v201008.GetGeoLocationInfo,
 47     google.ads.adwords.examples.ExampleBase);
 48 
 49 /**
 50  * Runs the code example.
 51  *
 52  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 53  * example.
 54  * @param {function} callback the callback method to be called once this example
 55  * is complete.
 56  */
 57 google.ads.adwords.examples.v201008.GetGeoLocationInfo.prototype.run =
 58     function(user, callback) {
 59   // Get the GeoLocationService.
 60   var geoLocationService = user.getService(
 61       google.ads.adwords.AdWordsService.v201008.GeoLocationService);
 62 
 63   var address1 = new google.ads.adwords.v201008.Address();
 64   address1.streetAddress = '1600 Amphitheatre Parkway';
 65   address1.cityName = 'Mountain View';
 66   address1.provinceCode = 'CA';
 67   address1.postalCode = '94043';
 68   address1.countryCode = 'US';
 69 
 70   var address2 = new google.ads.adwords.v201008.Address();
 71   address2.streetAddress = '76 Ninth Avenue';
 72   address2.cityName = 'New York';
 73   address2.provinceCode = 'NY';
 74   address2.postalCode = '10011';
 75   address2.countryCode = 'US';
 76 
 77   // In order to type unicode addresses like these, make sure your js file is
 78   // also encoded in UTF-8 without BOM, since calcdeps.py support UTT-8 encoding
 79   // with BOM.
 80   var address3 = new google.ads.adwords.v201008.Address();
 81   address3.streetAddress = '五四大街1号, Beijing东城区';
 82   address3.countryCode = 'CN';
 83 
 84   // Create selector.
 85   var selector = new google.ads.adwords.v201008.GeoLocationSelector();
 86   selector.addresses = [address1, address2, address3];
 87 
 88   try {
 89     geoLocationService.get(selector,
 90       goog.bind(function(geoLocations) {
 91         // Get geo locations.
 92         if (geoLocations) {
 93           // Display geo locations.
 94           for (var i = 0, len = geoLocations.length; i < len; i++) {
 95             var geoLocation = geoLocations[i];
 96             if (!(geoLocation instanceof google.ads.adwords.v201008.
 97                 InvalidGeoLocation)) {
 98               this.writeOutput('Address %s has latitude %d and longitude %d.',
 99                   geoLocation.address.streetAddress,
100                   geoLocation.geoPoint.latitudeInMicroDegrees,
101                   geoLocation.geoPoint.longitudeInMicroDegrees);
102             } else {
103               this.writeOutput('Invalid geo location returned.');
104             }
105           }
106         } else {
107           this.writeOutput('No geo locations were returned.');
108         }
109       callback();
110       }, this),
111       goog.bind(function(soapException) {
112         this.writeOutput('Failed to get geo locations. Soap Fault ' +
113             'says "%s"', soapException.getInnerException().getFaultString());
114         callback();
115       }, this)
116     );
117   } catch (ex) {
118     this.writeOutput('An exception occurred while running the code example.');
119     callback();
120   }
121 };
