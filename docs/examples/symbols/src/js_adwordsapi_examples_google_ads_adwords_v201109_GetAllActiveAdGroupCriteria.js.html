 19 /**
 20  * @fileoverview Defines GetAllActiveAdGroupCriteria, a code example that gets
 21  *     all active ad group criteria in an ad group.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201109.GetAllActiveAdGroupCriteria');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201109');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201109.AdGroupCriterionService');
 29 goog.require('google.ads.adwords.v201109.Keyword');
 30 goog.require('google.ads.adwords.v201109.Placement');
 31 goog.require('google.ads.adwords.v201109.Predicate');
 32 goog.require('google.ads.adwords.v201109.Selector');
 33 goog.require('google.ads.adwords.v201109.UserStatus');

 35 /**
 36  * This code example gets all active ad group criteria in an ad group. To
 37  * add ad group criteria, run AddAdGroupCriteria.js. To get ad groups in an
 38  * account, run GetAllAdGroups.js.
 39  *
 40  * Tags: AdGroupCriterionService.get
 41  *
 42  * @extends google.ads.adwords.examples.ExampleBase
 43  * @constructor
 44  */
 45 
 46 google.ads.adwords.examples.v201109.GetAllActiveAdGroupCriteria = function() {
 47   google.ads.adwords.examples.ExampleBase.call(this);
 48   this.description = 'This code example gets all active ad group criteria ' +
 49       'in an ad group. To add ad group criteria, run AddAdGroupCriteria.js. ' +
 50       'To get ad groups in an account, run GetAllAdGroups.js.';
 51 };
 52 goog.inherits(google.ads.adwords.examples.v201109.GetAllActiveAdGroupCriteria,
 53     google.ads.adwords.examples.ExampleBase);
 54 
 55 /**
 56  * Runs the code example.
 57  *
 58  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 59  *     example.
 60  * @param {function} callback the callback method to be called once this example
 61  *     is complete.
 62  */
 63 google.ads.adwords.examples.v201109.GetAllActiveAdGroupCriteria.prototype.run =
 64     function(user, callback) {
 65   // Get the AdGroupCriterionService.
 66   var adGroupCriterionService = user.getService(
 67       google.ads.adwords.AdWordsService.v201109.AdGroupCriterionService);
 68 
 69   var adGroupId = this._T('INSERT_ADGROUP_ID_HERE');
 70 
 71   // Create a selector.
 72   var selector = new google.ads.adwords.v201109.Selector();
 73   selector.fields = ['Id', 'AdGroupId', 'KeywordText', 'Status',
 74       'KeywordMatchType', 'PlacementUrl'];
 75 
 76   // Set filter conditions.
 77   var adGroupPredicate = new google.ads.adwords.v201109.Predicate();
 78   adGroupPredicate.field = 'AdGroupId';
 79   adGroupPredicate.operator = google.ads.adwords.v201109.PredicateOperator.IN;
 80   adGroupPredicate.values = [adGroupId];
 81 
 82   var statusPredicate = new google.ads.adwords.v201109.Predicate();
 83   statusPredicate.field = 'Status';
 84   statusPredicate.operator =
 85       google.ads.adwords.v201109.PredicateOperator.EQUALS;
 86   statusPredicate.values = [google.ads.adwords.v201109.UserStatus.ACTIVE];
 87 
 88   selector.predicates = [adGroupPredicate, statusPredicate];
 89 
 90   try {
 91     adGroupCriterionService.get(selector,
 92       goog.bind(function(page) {
 93         if (page && page.entries && page.entries.length > 0) {
 94           // Display ad group criteria.
 95           for (var i = 0, len = page.entries.length; i < len; i++) {
 96             var adGroupCriterion = page.entries[i];
 97             var criterion = adGroupCriterion.criterion;
 98             if (criterion instanceof google.ads.adwords.v201109.Keyword) {
 99               this.writeOutput('Keyword ad group criterion with ad group ' +
100                   'ID = "%s", criterion ID = "%s", text = "%s" and ' +
101                   'matchType = "%s" was found.', adGroupCriterion.adGroupId,
102                   criterion.id, criterion.text, criterion.matchType);
103             } else if (criterion instanceof google.ads.adwords.v201109.
104                 Placement) {
105               this.writeOutput('Placement ad group criterion with ad ' +
106                   'group ID = "%s", criterion ID = "%s" and url = "%s"' +
107                   ' was found.', adGroupCriterion.adGroupId, criterion.id,
108                   criterion.url);
109             }
110           }
111         } else {
112           this.writeOutput('No ad group criteria found.');
113         }
114         callback();
115       }, this),
116       goog.bind(function(soapException) {
117         this.writeOutput('Failed to get ad group criteria. Soap Fault ' +
118             'says "%s"', soapException.getInnerException().getFaultString());
119         callback();
120       }, this)
121     );
122   } catch (ex) {
123     this.writeOutput('An exception occurred while running the code example.');
124     callback();
125   }
126 };
