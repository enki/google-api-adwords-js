 19 /**
 20  * @fileoverview Defines GetAllVideos, a code example that gets all videos.
 21  */
 22 
 23 goog.provide('google.ads.adwords.examples.v201109.GetAllVideos');
 24 
 25 goog.require('google.ads.adwords.AdWordsService.v201109');
 26 goog.require('google.ads.adwords.examples.ExampleBase');
 27 goog.require('google.ads.adwords.v201109.MediaMediaType');
 28 goog.require('google.ads.adwords.v201109.MediaService');
 29 goog.require('google.ads.adwords.v201109.Predicate');
 30 goog.require('google.ads.adwords.v201109.PredicateOperator');
 31 goog.require('google.ads.adwords.v201109.Selector');
 32 goog.require('google.ads.adwords.v201109.Video');

 34 /**
 35  * This code example gets all videos. To upload video, see
 36  * http://adwords.google.com/support/aw/bin/answer.py?hl=en&answer=39454.
 37  *
 38  * Tags: MediaService.get
 39  *
 40  * @extends google.ads.adwords.examples.ExampleBase
 41  * @constructor
 42  */
 43 google.ads.adwords.examples.v201109.GetAllVideos = function() {
 44   google.ads.adwords.examples.ExampleBase.call(this);
 45   this.description = 'This code example gets all videos. To upload video, ' +
 46       'see http://adwords.google.com/support/aw/bin/answer.py?hl=en&' +
 47       'answer=39454.';
 48 };
 49 goog.inherits(google.ads.adwords.examples.v201109.GetAllVideos,
 50     google.ads.adwords.examples.ExampleBase);
 51 
 52 /**
 53  * Runs the code example.
 54  *
 55  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 56  *     example.
 57  * @param {function} callback the callback method to be called once this example
 58  *     is complete.
 59  */
 60 google.ads.adwords.examples.v201109.GetAllVideos.prototype.run =
 61     function(user, callback) {
 62   // Get the MediaService.
 63   var mediaService = user.getService(
 64       google.ads.adwords.AdWordsService.v201109.MediaService);
 65 
 66   // Create selector.
 67   var selector = new google.ads.adwords.v201109.Selector();
 68   selector.fields = ['MediaId', 'Type'];
 69 
 70   // Create a filter.
 71   var predicate = new google.ads.adwords.v201109.Predicate();
 72   predicate.operator = google.ads.adwords.v201109.PredicateOperator.EQUALS;
 73   predicate.field = 'Type';
 74   predicate.values = [google.ads.adwords.v201109.MediaMediaType.VIDEO];
 75 
 76   selector.predicates = [predicate];
 77 
 78   try {
 79     // Get all videos.
 80     mediaService.get(selector,
 81       goog.bind(function(page) {
 82         // Display videos.
 83         if (page && page.entries && page.entries.length > 0) {
 84           for (var i = 0, len = page.entries.length; i < len; i++) {
 85             var video = page.entries[i];
 86             this.writeOutput('Video with id = "%d" and name = "%s" was found.',
 87                 video.mediaId, video.name || '');
 88           }
 89         } else {
 90           this.writeOutput('No videos were found.');
 91         }
 92         callback();
 93       }, this),
 94       goog.bind(function(soapException) {
 95         this.writeOutput('Failed to retrieve all videos. Soap Fault says "%s"',
 96             soapException.getInnerException().getFaultString());
 97         callback();
 98       }, this)
 99     );
100   } catch (ex) {
101     this.writeOutput('An exception occurred while running the code example.');
102     callback();
103   }
104 };
