 19 /**
 20  * @fileoverview Defines GetTrafficEstimates, a code example that gets
 21  * keyword traffic estimates.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetTrafficEstimates');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.AdGroupEstimateRequest');
 29 goog.require('google.ads.adwords.v201008.CampaignEstimateRequest');
 30 goog.require('google.ads.adwords.v201008.CountryTarget');
 31 goog.require('google.ads.adwords.v201008.Keyword');
 32 goog.require('google.ads.adwords.v201008.KeywordEstimateRequest');
 33 goog.require('google.ads.adwords.v201008.KeywordMatchType');
 34 goog.require('google.ads.adwords.v201008.LanguageTarget');
 35 goog.require('google.ads.adwords.v201008.Money');
 36 goog.require('google.ads.adwords.v201008.TrafficEstimatorSelector');
 37 goog.require('google.ads.adwords.v201008.TrafficEstimatorService');
 38 
 39 /**
 40  * This code example gets keyword traffic estimates.
 41  *
 42  * Tags: TrafficEstimatorService.get
 43  *
 44  * @extends google.ads.adwords.examples.ExampleBase
 45  * @constructor
 46  */
 47 google.ads.adwords.examples.v201008.GetTrafficEstimates = function() {
 48   google.ads.adwords.examples.ExampleBase.call(this);
 49   this.description = 'This code example gets keyword traffic estimates.';
 50 };
 51 goog.inherits(google.ads.adwords.examples.v201008.GetTrafficEstimates,
 52     google.ads.adwords.examples.ExampleBase);
 53 
 54 /**
 55  * Runs the code example.
 56  *
 57  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 58  * example.
 59  * @param {function} callback the callback method to be called once this example
 60  * is complete.
 61  */
 62 google.ads.adwords.examples.v201008.GetTrafficEstimates.prototype.run =
 63     function(user, callback) {
 64   // Get the TrafficEstimatorService.
 65   var trafficEstimatorService = user.getService(
 66       google.ads.adwords.AdWordsService.v201008.TrafficEstimatorService);
 67 
 68   // Create keywords. Up to 2000 keywords can be passed in a single request.
 69   var keyword1 = new google.ads.adwords.v201008.Keyword();
 70   keyword1.text = 'mars cruise';
 71   keyword1.matchType = google.ads.adwords.v201008.KeywordMatchType.BROAD;
 72 
 73   var keyword2 = new google.ads.adwords.v201008.Keyword();
 74   keyword2.text = 'cheap cruise';
 75   keyword2.matchType = google.ads.adwords.v201008.KeywordMatchType.PHRASE;
 76 
 77   var keyword3 = new google.ads.adwords.v201008.Keyword();
 78   keyword3.text = 'cruise';
 79   keyword3.matchType = google.ads.adwords.v201008.KeywordMatchType.EXACT;
 80 
 81   var keywords = [keyword1, keyword2, keyword3];
 82 
 83   // Create a keyword estimate request for each keyword.
 84   var keywordEstimateRequests = [];
 85 
 86   for (var i = 0, len = keywords.length; i < len; i++) {
 87     var keywordEstimateRequest =
 88         new google.ads.adwords.v201008.KeywordEstimateRequest();
 89     keywordEstimateRequest.keyword = keywords[i];
 90     keywordEstimateRequests.push(keywordEstimateRequest);
 91   }
 92 
 93   // Create ad group estimate requests.
 94   var adGroupEstimateRequest =
 95       new google.ads.adwords.v201008.AdGroupEstimateRequest();
 96   adGroupEstimateRequest.keywordEstimateRequests = keywordEstimateRequests;
 97   adGroupEstimateRequest.maxCpc = new google.ads.adwords.v201008.Money();
 98   adGroupEstimateRequest.maxCpc.microAmount = 1000000;
 99 
100   // Create campaign estimate requests.
101   var campaignEstimateRequest =
102       new google.ads.adwords.v201008.CampaignEstimateRequest();
103   campaignEstimateRequest.adGroupEstimateRequests = [adGroupEstimateRequest];
104 
105   var countryTarget = new google.ads.adwords.v201008.CountryTarget();
106   countryTarget.countryCode = 'US';
107 
108   var languageTarget = new google.ads.adwords.v201008.LanguageTarget();
109   languageTarget.languageCode = 'en';
110 
111   campaignEstimateRequest.targets = [countryTarget, languageTarget];
112 
113   // Create selector.
114   var selector = new google.ads.adwords.v201008.TrafficEstimatorSelector();
115   selector.campaignEstimateRequests = [campaignEstimateRequest];
116 
117   try {
118     // Get traffic estimates.
119     trafficEstimatorService.get(selector,
120       goog.bind(function(result) {
121         // Display traffic estimates.
122         if (result && result.campaignEstimates &&
123             result.campaignEstimates.length > 0) {
124           var campaignEstimate = result.campaignEstimates[0];
125           if (campaignEstimate.adGroupEstimates &&
126               campaignEstimate.adGroupEstimates.length > 0) {
127             var adGroupEstimate = campaignEstimate.adGroupEstimates[0];
128             if (adGroupEstimate.keywordEstimates) {
129               for (var i = 0, len = adGroupEstimate.keywordEstimates.length;
130                   i < len; i++) {
131                 var keyword = keywordEstimateRequests[i].keyword;
132                 var keywordEstimate = adGroupEstimate.keywordEstimates[i];
133 
134                 // Find the mean of the min and max values.
135                 var meanAverageCpc =
136                     (keywordEstimate.min.averageCpc.microAmount +
137                         keywordEstimate.max.averageCpc.microAmount) / 2;
138                 var meanAveragePosition =
139                     (keywordEstimate.min.averagePosition +
140                         keywordEstimate.max.averagePosition) / 2;
141                 var meanClicks = (keywordEstimate.min.clicks +
142                     keywordEstimate.max.clicks) / 2;
143                 var meanTotalCost =
144                     (keywordEstimate.min.totalCost.microAmount +
145                         keywordEstimate.max.totalCost.microAmount) / 2;
146 
147                 this.writeOutput('Results for the keyword with text = "%s" ' +
148                     'and match type = "%s":', keyword.text, keyword.matchType);
149                 this.writeOutput('  Estimated average CPC: %d', meanAverageCpc);
150                 this.writeOutput('  Estimated ad position: %f',
151                     meanAveragePosition);
152                 this.writeOutput('  Estimated daily clicks: %d', meanClicks);
153                 this.writeOutput('  Estimated daily cost: %d', meanTotalCost);
154               }
155             }
156           }
157         } else {
158           this.writeOutput('No traffic estimates were returned.');
159         }
160         callback();
161       }, this),
162       goog.bind(function(soapException) {
163         this.writeOutput('Failed to get traffic estimates. Soap Fault ' +
164             'says "%s"', soapException.getInnerException().getFaultString());
165         callback();
166       }, this)
167     );
168   } catch (ex) {
169     this.writeOutput('An exception occurred while running the code example.');
170     callback();
171   }
172 };
