 19 /**
 20  * @fileoverview Defines GetTrafficEstimates, a code example that gets
 21  *     keyword traffic estimates.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetTrafficEstimates');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.AdGroupEstimateRequest');
 29 goog.require('google.ads.adwords.v201008.CampaignEstimateRequest');
 30 goog.require('google.ads.adwords.v201008.CountryTarget');
 31 goog.require('google.ads.adwords.v201008.Keyword');
 32 goog.require('google.ads.adwords.v201008.KeywordEstimateRequest');
 33 goog.require('google.ads.adwords.v201008.KeywordMatchType');
 34 goog.require('google.ads.adwords.v201008.LanguageTarget');
 35 goog.require('google.ads.adwords.v201008.Money');
 36 goog.require('google.ads.adwords.v201008.TrafficEstimatorSelector');
 37 goog.require('google.ads.adwords.v201008.TrafficEstimatorService');
 38 
 39 /**
 40  * This code example gets keyword traffic estimates.
 41  *
 42  * Tags: TrafficEstimatorService.get
 43  *
 44  * @extends google.ads.adwords.examples.ExampleBase
 45  * @constructor
 46  */
 47 google.ads.adwords.examples.v201008.GetTrafficEstimates = function() {
 48   google.ads.adwords.examples.ExampleBase.call(this);
 49   /**
 50   * Example description.
 51   * @type {string}
 52   */
 53   this.description = 'This code example gets keyword traffic estimates.';
 54 };
 55 goog.inherits(google.ads.adwords.examples.v201008.GetTrafficEstimates,
 56     google.ads.adwords.examples.ExampleBase);
 57 
 58 /**
 59  * Runs the code example.
 60  *
 61  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 62  *     example.
 63  * @param {function} callback the callback method to be called once this example
 64  *     is complete.
 65  */
 66 google.ads.adwords.examples.v201008.GetTrafficEstimates.prototype.run =
 67     function(user, callback) {
 68   // Get the TrafficEstimatorService.
 69   var trafficEstimatorService = user.getService(
 70       google.ads.adwords.AdWordsService.v201008.TrafficEstimatorService);
 71 
 72   // Create keywords. Up to 2000 keywords can be passed in a single request.
 73   var keyword1 = new google.ads.adwords.v201008.Keyword();
 74   keyword1.text = 'mars cruise';
 75   keyword1.matchType = google.ads.adwords.v201008.KeywordMatchType.BROAD;
 76 
 77   var keyword2 = new google.ads.adwords.v201008.Keyword();
 78   keyword2.text = 'cheap cruise';
 79   keyword2.matchType = google.ads.adwords.v201008.KeywordMatchType.PHRASE;
 80 
 81   var keyword3 = new google.ads.adwords.v201008.Keyword();
 82   keyword3.text = 'cruise';
 83   keyword3.matchType = google.ads.adwords.v201008.KeywordMatchType.EXACT;
 84 
 85   var keywords = [keyword1, keyword2, keyword3];
 86 
 87   // Create a keyword estimate request for each keyword.
 88   var keywordEstimateRequests = [];
 89 
 90   for (var i = 0, len = keywords.length; i < len; i++) {
 91     var keywordEstimateRequest =
 92         new google.ads.adwords.v201008.KeywordEstimateRequest();
 93     keywordEstimateRequest.keyword = keywords[i];
 94     keywordEstimateRequests.push(keywordEstimateRequest);
 95   }
 96 
 97   // Create ad group estimate requests.
 98   var adGroupEstimateRequest =
 99       new google.ads.adwords.v201008.AdGroupEstimateRequest();
100   adGroupEstimateRequest.keywordEstimateRequests = keywordEstimateRequests;
101   adGroupEstimateRequest.maxCpc = new google.ads.adwords.v201008.Money();
102   adGroupEstimateRequest.maxCpc.microAmount = 1000000;
103 
104   // Create campaign estimate requests.
105   var campaignEstimateRequest =
106       new google.ads.adwords.v201008.CampaignEstimateRequest();
107   campaignEstimateRequest.adGroupEstimateRequests = [adGroupEstimateRequest];
108 
109   var countryTarget = new google.ads.adwords.v201008.CountryTarget();
110   countryTarget.countryCode = 'US';
111 
112   var languageTarget = new google.ads.adwords.v201008.LanguageTarget();
113   languageTarget.languageCode = 'en';
114 
115   campaignEstimateRequest.targets = [countryTarget, languageTarget];
116 
117   // Create selector.
118   var selector = new google.ads.adwords.v201008.TrafficEstimatorSelector();
119   selector.campaignEstimateRequests = [campaignEstimateRequest];
120 
121   try {
122     // Get traffic estimates.
123     trafficEstimatorService.get(selector,
124       goog.bind(function(result) {
125         // Display traffic estimates.
126         if (result && result.campaignEstimates &&
127             result.campaignEstimates.length > 0) {
128           var campaignEstimate = result.campaignEstimates[0];
129           if (campaignEstimate.adGroupEstimates &&
130               campaignEstimate.adGroupEstimates.length > 0) {
131             var adGroupEstimate = campaignEstimate.adGroupEstimates[0];
132             if (adGroupEstimate.keywordEstimates) {
133               for (var i = 0, len = adGroupEstimate.keywordEstimates.length;
134                   i < len; i++) {
135                 var keyword = keywordEstimateRequests[i].keyword;
136                 var keywordEstimate = adGroupEstimate.keywordEstimates[i];
137 
138                 // Find the mean of the min and max values.
139                 var meanAverageCpc =
140                     (keywordEstimate.min.averageCpc.microAmount +
141                         keywordEstimate.max.averageCpc.microAmount) / 2;
142                 var meanAveragePosition =
143                     (keywordEstimate.min.averagePosition +
144                         keywordEstimate.max.averagePosition) / 2;
145                 var meanClicks = (keywordEstimate.min.clicks +
146                     keywordEstimate.max.clicks) / 2;
147                 var meanTotalCost =
148                     (keywordEstimate.min.totalCost.microAmount +
149                         keywordEstimate.max.totalCost.microAmount) / 2;
150 
151                 this.writeOutput('Results for the keyword with text = "%s" ' +
152                     'and match type = "%s":', keyword.text, keyword.matchType);
153                 this.writeOutput('  Estimated average CPC: %d', meanAverageCpc);
154                 this.writeOutput('  Estimated ad position: %f',
155                     meanAveragePosition);
156                 this.writeOutput('  Estimated daily clicks: %d', meanClicks);
157                 this.writeOutput('  Estimated daily cost: %d', meanTotalCost);
158               }
159             }
160           }
161         } else {
162           this.writeOutput('No traffic estimates were returned.');
163         }
164         callback();
165       }, this),
166       goog.bind(function(soapException) {
167         this.writeOutput('Failed to get traffic estimates. Soap Fault ' +
168             'says "%s"', soapException.getInnerException().getFaultString());
169         callback();
170       }, this)
171     );
172   } catch (ex) {
173     this.writeOutput('An exception occurred while running the code example.');
174     callback();
175   }
176 };
