 19 /**
 20  * @fileoverview Defines UpdateAdGroup, a code example that updates an ad group,
 21  * setting its status to 'PAUSED'.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.UpdateAdGroup');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.AdGroup');
 29 goog.require('google.ads.adwords.v201008.AdGroupOperation');
 30 goog.require('google.ads.adwords.v201008.AdGroupService');
 31 goog.require('google.ads.adwords.v201008.AdGroupStatus');
 32 goog.require('google.ads.adwords.v201008.Operator');
 33 
 34 /**
 35  * This code example illustrates how to update an ad group, setting its
 36  * status to 'PAUSED'. To create an ad group, run AddAdGroup.js.
 37  *
 38  * Tags: AdGroupService.mutate
 39  *
 40  * @extends google.ads.adwords.examples.ExampleBase
 41  * @constructor
 42  */
 43 google.ads.adwords.examples.v201008.UpdateAdGroup = function() {
 44   google.ads.adwords.examples.ExampleBase.call(this);
 45   this.description = 'This code example illustrates how to update an ad ' +
 46     'group, setting its status to "PAUSED". To create an ad group, run ' +
 47     'AddAdGroup.js.';
 48 };
 49 goog.inherits(google.ads.adwords.examples.v201008.UpdateAdGroup,
 50     google.ads.adwords.examples.ExampleBase);
 51 
 52 /**
 53  * Runs the code example.
 54  *
 55  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 56  * example.
 57  * @param {function} callback the callback method to be called once this example
 58  * is complete.
 59  */
 60 google.ads.adwords.examples.v201008.UpdateAdGroup.prototype.run =
 61     function(user, callback) {
 62   // Get the AdGroupService.
 63   var adGroupService = user.getService(
 64       google.ads.adwords.AdWordsService.v201008.AdGroupService);
 65 
 66   var adGroupId = this._T('INSERT_AD_GROUP_ID_HERE');
 67 
 68   var adGroup = new google.ads.adwords.v201008.AdGroup();
 69   adGroup.status = google.ads.adwords.v201008.AdGroupStatus.PAUSED;
 70   adGroup.id = adGroupId;
 71 
 72   var operation = new google.ads.adwords.v201008.AdGroupOperation();
 73   operation.operator = google.ads.adwords.v201008.Operator.SET;
 74   operation.operand = adGroup;
 75 
 76   try {
 77     adGroupService.mutate([operation],
 78       goog.bind(function(retval) {
 79         if (retval && retval.value) {
 80           for (var i = 0, len = retval.value.length; i < len; i++) {
 81             var adGroupValue = retval.value[i];
 82             this.writeOutput('Ad group with id = "%s" was successfully paused.',
 83                 adGroupValue.id);
 84           }
 85         }
 86         callback();
 87       }, this),
 88       goog.bind(function(soapException) {
 89         this.writeOutput('Failed to update ad group(s). Soap Fault says "%s"',
 90             soapException.getInnerException().getFaultString());
 91         callback();
 92       }, this)
 93     );
 94   } catch (ex) {
 95     this.writeOutput('An exception occurred while running the code example.');
 96     callback();
 97   }
 98 };
