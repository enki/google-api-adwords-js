 19 /**
 20  * @fileoverview Defines GetAllReportDefinitions, a code example that all report
 21  * definitions.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetAllReportDefinitions');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.ReportDefinitionSelector');
 29 goog.require('google.ads.adwords.v201008.ReportDefinitionService');
 30 
 31 /**
 32  * This code example gets all report definitions. To add a report
 33  * definition, run AddKeywordsPerformanceReportDefinition.js.
 34  *
 35  * Tags: ReportDefinitionService.get
 36  *
 37  * @extends google.ads.adwords.examples.ExampleBase
 38  * @constructor
 39  */
 40 google.ads.adwords.examples.v201008.GetAllReportDefinitions = function() {
 41   google.ads.adwords.examples.ExampleBase.call(this);
 42   this.description = 'This code example gets all report definitions. To add ' +
 43       'a report definition, run AddKeywordsPerformanceReportDefinition.js.';
 44 };
 45 goog.inherits(google.ads.adwords.examples.v201008.GetAllReportDefinitions,
 46     google.ads.adwords.examples.ExampleBase);
 47 
 48 /**
 49  * Runs the code example.
 50  *
 51  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 52  * example.
 53  * @param {function} callback the callback method to be called once this example
 54  * is complete.
 55  */
 56 google.ads.adwords.examples.v201008.GetAllReportDefinitions.prototype.run =
 57     function(user, callback) {
 58   // Get the ReportDefinitionService.
 59   var reportDefinitionService = user.getService(
 60       google.ads.adwords.AdWordsService.v201008.ReportDefinitionService);
 61 
 62   var selector = new google.ads.adwords.v201008.ReportDefinitionSelector();
 63 
 64   try {
 65     // Get all report definitions.
 66     reportDefinitionService.get(selector,
 67       goog.bind(function(page) {
 68         // Display report definitions.
 69         if (page && page.entries && page.entries.length > 0) {
 70           for (var i = 0, len = page.entries.length; i < len; i++) {
 71             var reportDefinition = page.entries[i];
 72             this.writeOutput('ReportDefinition with name "%s" and id "%d" ' +
 73                 'was found.', reportDefinition.reportName, reportDefinition.id);
 74           }
 75         } else {
 76           this.writeOutput('No report definitions were found.');
 77         }
 78         callback();
 79       }, this),
 80       goog.bind(function(soapException) {
 81         this.writeOutput('Failed to get all report definitions. Soap Fault ' +
 82             'says "%s"', soapException.getInnerException().getFaultString());
 83         callback();
 84       }, this)
 85     );
 86   } catch (ex) {
 87     this.writeOutput('An exception occurred while running the code example.');
 88     callback();
 89   }
 90 };
