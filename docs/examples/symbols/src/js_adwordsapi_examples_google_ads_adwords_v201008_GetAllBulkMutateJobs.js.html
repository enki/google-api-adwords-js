 19 /**
 20  * @fileoverview Defines GetAllBulkMutateJobs, a code example that
 21  *     gets all bulk mutate jobs in the account.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetAllBulkMutateJobs');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.ApiUsageType');
 29 goog.require('google.ads.adwords.v201008.BulkMutateJobSelector');
 30 goog.require('google.ads.adwords.v201008.BulkMutateJobService');
 31 
 32 /**
 33  * This code example gets all bulk mutate jobs in the account. To add a
 34  * bulk mutate job, run PerformBulkMutateJob.js.
 35  *
 36  * Tags: BulkMutateJobService.get
 37  *
 38  * @extends google.ads.adwords.examples.ExampleBase
 39  * @constructor
 40  */
 41 
 42 google.ads.adwords.examples.v201008.GetAllBulkMutateJobs =
 43     function() {
 44   google.ads.adwords.examples.ExampleBase.call(this);
 45   /**
 46   * Example description.
 47   * @type {string}
 48   */
 49   this.description = 'This code example gets all bulk mutate jobs in the ' +
 50       'account. To add a bulk mutate job, run PerformBulkMutateJob.js.';
 51 };
 52 goog.inherits(
 53     google.ads.adwords.examples.v201008.GetAllBulkMutateJobs,
 54     google.ads.adwords.examples.ExampleBase);
 55 
 56 /**
 57  * Runs the code example.
 58  *
 59  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 60  *     example.
 61  * @param {function} callback the callback method to be called once this example
 62  *     is complete.
 63  */
 64 google.ads.adwords.examples.v201008.GetAllBulkMutateJobs.prototype.
 65     run = function(user, callback) {
 66   // Get the BulkMutateJobService.
 67   var bulkMutateJobService = user.getService(
 68       google.ads.adwords.AdWordsService.v201008.BulkMutateJobService);
 69 
 70   // Create selector.
 71   var selector = new google.ads.adwords.v201008.BulkMutateJobSelector();
 72   selector.includeStats = true;
 73 
 74   try {
 75     // Get all bulk mutate jobs.
 76     bulkMutateJobService.get(selector,
 77       goog.bind(function(bulkMutateJobs) {
 78         // Display bulk mutate jobs.
 79         if (bulkMutateJobs) {
 80           for (var i = 0, len = bulkMutateJobs.length; i < len; i++) {
 81             var bulkMutateJob = bulkMutateJobs[i];
 82             this.writeOutput('Bulk mutate job with id "%d" and status "%s" ' +
 83                 'was found.', bulkMutateJob.id, bulkMutateJob.status);
 84 
 85             if (bulkMutateJob.status == google.ads.adwords.v201008.
 86                 BasicJobStatus.PENDING) {
 87               this.writeOutput('  Total parts: %d, parts received: %d.',
 88                   bulkMutateJob.numRequestParts,
 89                   bulkMutateJob.numRequestPartsReceived);
 90             } else if (bulkMutateJob.status == google.ads.adwords.v201008.
 91                 BasicJobStatus.PROCESSING) {
 92               this.writeOutput('  Percent complete: %d.',
 93                   bulkMutateJob.stats.progressPercent);
 94             } else if (bulkMutateJob.status == google.ads.adwords.v201008.
 95                 BasicJobStatus.COMPLETED) {
 96               this.writeOutput('  Total operations: %d, failed: %d, ' +
 97                   'unprocessed: %d.', bulkMutateJob.stats.numOperations,
 98                   bulkMutateJob.stats.numFailedOperations,
 99                   bulkMutateJob.stats.numUnprocessedOperations);
100             } else if (bulkMutateJob.status == google.ads.adwords.v201008.
101                 BasicJobStatus.FAILED) {
102                   this.writeOutput('  Failure reason: %s.',
103                       bulkMutateJob.failureReason.Item);
104             }
105           }
106         } else {
107           this.writeOutput('No bulk mutate jobs were found.');
108         }
109         callback();
110       }, this),
111       goog.bind(function(soapException) {
112         this.writeOutput('Failed to get all bulk mutate jobs. Soap Fault ' +
113             'says "%s"', soapException.getInnerException().getFaultString());
114         callback();
115       }, this)
116     );
117   } catch (ex) {
118     this.writeOutput('An exception occurred while running the code example.');
119     callback();
120   }
121 };
