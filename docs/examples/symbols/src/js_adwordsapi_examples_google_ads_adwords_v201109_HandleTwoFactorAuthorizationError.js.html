 21  * illustrates how to handle 2 factor authorization errors.

 24 goog.provide('google.ads.adwords.examples.v201109.HandleTwoFactorAuthorizationError');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201109');
 27 goog.require('google.ads.adwords.AppConfig');
 28 goog.require('google.ads.adwords.examples.ExampleBase');
 29 goog.require('google.system.net.HttpWebTransportFactory');
 30 
 31 /**
 32  * This code example illustrates how to handle 2 factor authorization errors.
 33  *
 34  * @extends google.ads.adwords.examples.ExampleBase
 35  * @constructor
 36  */
 37 google.ads.adwords.examples.v201109.HandleTwoFactorAuthorizationError =
 38     function() {
 39   google.ads.adwords.examples.ExampleBase.call(this);
 40   this.description = 'This code example illustrates how to handle 2 factor ' +
 41       'authorization errors.';
 42 };
 43 goog.inherits(
 44     google.ads.adwords.examples.v201109.HandleTwoFactorAuthorizationError,
 45     google.ads.adwords.examples.ExampleBase);
 46 
 47 /**
 48  * Runs the code example.
 49  *
 50  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 51  *     example.
 52  * @param {function} callback the callback method to be called once this example
 53  *     is complete.
 54  */
 55 google.ads.adwords.examples.v201109.HandleTwoFactorAuthorizationError.
 56     prototype.run = function(user, callback) {
 57 
 58   var config = new google.ads.adwords.AppConfig();
 59   config.setEmail('2steptester@gmail.com');
 60   config.setPassword('testaccount');
 61   var authToken = new google.ads.common.AuthToken(config, 'adwords',
 62       config.getEmail(), config.getPassword());
 63 
 64   authToken.getToken(
 65     goog.bind(function(token) {
 66       this.writeOutput('Retrieved auth token = %s for user %s.', token,
 67           config.getEmail());
 68     }, this),
 69     goog.bind(function(error) {
 70       if (error.getInfo() == 'InvalidSecondFactor') {
 71         this.writeOutput('The user has enabled two factor authentication in ' +
 72             'this account. Have the user generate an application-specific ' +
 73             'password to make calls against the AdWords API. See ' +
 74              'http://adwordsapi.blogspot.com/2011/02/authentication-changes-' +
 75              'with-2-step.html for more details');
 76       } else {
 77         this.writeOutput('Invalid credentials');
 78       }
 79       callback();
 80     }, this)
 81   );
 82 };
