 19 /**
 20  * @fileoverview Defines DeleteSiteLinks, a code example that shows how to
 21  * delete site links from an existing campaign.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.DeleteSiteLinks');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.AdExtension');
 29 goog.require('google.ads.adwords.v201008.CampaignAdExtension');
 30 goog.require('google.ads.adwords.v201008.CampaignAdExtensionOperation');
 31 goog.require('google.ads.adwords.v201008.CampaignAdExtensionSelector');
 32 goog.require('google.ads.adwords.v201008.CampaignAdExtensionService');
 33 goog.require('google.ads.adwords.v201008.Operator');
 34 goog.require('google.ads.adwords.v201008.Sitelink');
 35 goog.require('google.ads.adwords.v201008.SitelinksExtension');

 37 /**
 38  * This code example shows how to delete site links from an existing
 39  * campaign. To add site links to an existing campaign, run AddSiteLinks.js.
 40  * To get existing campaigns, run GetAllCampaigns.js.
 41  *
 42  * Tags: CampaignAdExtensionService.mutate
 43  *
 44  * @extends google.ads.adwords.examples.ExampleBase
 45  * @constructor
 46  */
 47 
 48 google.ads.adwords.examples.v201008.DeleteSiteLinks = function() {
 49   google.ads.adwords.examples.ExampleBase.call(this);
 50   this.description = 'This code example shows how to delete site links from ' +
 51       'an existing campaign. To add site links to an existing campaign, ' +
 52       'run AddSiteLinks.js. To get existing campaigns, run GetAllCampaigns.js.';
 53 };
 54 goog.inherits(google.ads.adwords.examples.v201008.DeleteSiteLinks,
 55     google.ads.adwords.examples.ExampleBase);
 56 
 57 /**
 58  * Runs the code example.
 59  *
 60  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 61  * example.
 62  * @param {function} callback the callback method to be called once this example
 63  * is complete.
 64  */
 65 google.ads.adwords.examples.v201008.DeleteSiteLinks.prototype.run = function(
 66     user, callback) {
 67   // Get the CampaignAdExtensionService.
 68   var campaignAdExtensionService = user.getService(
 69       google.ads.adwords.AdWordsService.v201008.CampaignAdExtensionService);
 70 
 71   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 72   var siteLinkExtensionId = -1;
 73 
 74   // Get the campaign ad extension containing sitelinks.
 75   var selector = new google.ads.adwords.v201008.CampaignAdExtensionSelector();
 76   selector.campaignIds = [campaignId];
 77   selector.statuses = [google.ads.adwords.v201008.CampaignAdExtensionStatus.
 78       ACTIVE];
 79 
 80   try {
 81     campaignAdExtensionService.get(selector,
 82       goog.bind(function(page) {
 83         if (page && page.entries) {
 84           for (var i = 0, len = page.entries.length; i < len; i++) {
 85             var extension = page.entries[i];
 86             if (extension.adExtension instanceof
 87                 google.ads.adwords.v201008.SitelinksExtension) {
 88               siteLinkExtensionId = extension.adExtension.id;
 89               break;
 90             }
 91           }
 92         }
 93 
 94         if (siteLinkExtensionId == -1) {
 95           this.writeOutput('No sitelinks extension for the specified ' +
 96               'campaignId');
 97           callback();
 98           return;
 99         }
100 
101         var campaignAdExtension =
102             new google.ads.adwords.v201008.CampaignAdExtension();
103         campaignAdExtension.campaignId = campaignId;
104         campaignAdExtension.adExtension =
105             new google.ads.adwords.v201008.AdExtension();
106         campaignAdExtension.adExtension.id = siteLinkExtensionId;
107 
108         var operation = new google.ads.adwords.v201008.
109             CampaignAdExtensionOperation();
110         operation.operator = google.ads.adwords.v201008.Operator.REMOVE;
111         operation.operand = campaignAdExtension;
112 
113         campaignAdExtensionService.mutate([operation],
114           goog.bind(function(retval) {
115             if (retval && retval.value && retval.value.length > 0) {
116               for (var i = 0, len = retval.value.length; i < len; i++) {
117                 var campaignExtension = retval.value[i];
118                 this.writeOutput('Deleted a campaign ad extension with id = ' +
119                     '"%d" and status = "%s"', campaignExtension.adExtension.id,
120                     campaignExtension.status);
121                 for (var j = 0, len1 = campaignExtension.adExtension.sitelinks.
122                     length; j < len1; j++) {
123                   var siteLink = campaignExtension.adExtension.sitelinks[j];
124                   this.writeOutput('-- Site link text is "%s" and destination' +
125                       ' url is %s', siteLink.displayText,
126                       siteLink.destinationUrl);
127                 }
128               }
129             }
130             callback();
131           }, this),
132           goog.bind(function(soapException) {
133             this.writeOutput('Failed to delete site links. Soap Fault says' +
134                 ' "%s"', soapException.getInnerException().getFaultString());
135             callback();
136           }, this)
137         );
138       }, this),
139       goog.bind(function(soapException) {
140         this.writeOutput('Failed to delete experiment. Soap Fault says "%s"',
141             soapException.getInnerException().getFaultString());
142         callback();
143       }, this)
144     );
145   } catch (ex) {
146     this.writeOutput('An exception occurred while running the code example.');
147     callback();
148   }
149 };
