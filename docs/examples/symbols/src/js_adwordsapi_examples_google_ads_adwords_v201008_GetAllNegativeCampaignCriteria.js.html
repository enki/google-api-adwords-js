 19 /**
 20  * @fileoverview Defines GetAllNegativeCampaignCriteria, a code example that
 21  * gets all negative campaign criteria in an account.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetAllNegativeCampaignCriteria');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.CampaignCriterionSelector');
 29 goog.require('google.ads.adwords.v201008.CampaignCriterionService');
 30 goog.require('google.ads.adwords.v201008.Keyword');
 31 goog.require('google.ads.adwords.v201008.Placement');
 32 
 33 /**
 34  * This code example retrieves all negative campaign criteria in an account.
 35  * To add a negative campaign criterion, run AddNegativeCampaignCriterion.js.
 36  *
 37  * Tags: CampaignCriterionService.get
 38  *
 39  * @extends google.ads.adwords.examples.ExampleBase
 40  * @constructor
 41  */
 42 google.ads.adwords.examples.v201008.GetAllNegativeCampaignCriteria =
 43     function() {
 44   google.ads.adwords.examples.ExampleBase.call(this);
 45   this.description = 'This code example retrieves all negative campaign ' +
 46       'criteria in an account. To add a negative campaign criterion, run ' +
 47       'AddNegativeCampaignCriterion.js.';
 48 };
 49 goog.inherits(
 50     google.ads.adwords.examples.v201008.GetAllNegativeCampaignCriteria,
 51     google.ads.adwords.examples.ExampleBase);
 52 
 53 /**
 54  * Runs the code example.
 55  *
 56  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 57  * example.
 58  * @param {function} callback the callback method to be called once this example
 59  * is complete.
 60  */
 61 google.ads.adwords.examples.v201008.GetAllNegativeCampaignCriteria.prototype.
 62     run = function(user, callback) {
 63   // Get the CampaignCriterionService.
 64   var campaignCriterionService = user.getService(
 65       google.ads.adwords.AdWordsService.v201008.CampaignCriterionService);
 66 
 67   var selector = new google.ads.adwords.v201008.CampaignCriterionSelector();
 68 
 69   try {
 70     campaignCriterionService.get(selector,
 71       goog.bind(function(page) {
 72         if (page && page.entries && page.entries.length > 0) {
 73           for (var i = 0, len = page.entries.length; i < len; i++) {
 74             var campaignCriterion = page.entries[i];
 75             var criterion = campaignCriterion.criterion;
 76             if (criterion instanceof google.ads.adwords.v201008.Keyword) {
 77               this.writeOutput('Negative keyword campaign criterion with ' +
 78                   'campaign ID = "%d", criterion ID = "%d", and text = "%s" ' +
 79                   'was found.', campaignCriterion.campaignId, criterion.id,
 80                   criterion.text);
 81             } else if (criterion instanceof
 82                 google.ads.adwords.v201008.Placement) {
 83               this.writeOutput('Negative placement campaign criterion with ' +
 84                   'campaign ID = "%d", criterion ID = "%d" and url = "%s" ' +
 85                   'was found.', campaignCriterion.campaignId, criterion.id,
 86                   criterion.url);
 87             } else {
 88               this.writeOutput('Negative campaign criterion with campaign ' +
 89                   'ID = "%d", criterion ID = "%d" and type = "%s" was found.',
 90                   campaignCriterion.campaignId, criterion.id,
 91                   criterion.criterionType);
 92             }
 93           }
 94         } else {
 95           this.writeOutput('No negative campaign criteria were found.');
 96         }
 97         callback();
 98       }, this),
 99       goog.bind(function(soapException) {
100         this.writeOutput('Failed to retrieve all negative campaign criteria. ' +
101             'Soap Fault says "%s"',
102             soapException.getInnerException().getFaultString());
103         callback();
104       }, this)
105     );
106   } catch (ex) {
107     this.writeOutput('An exception occurred while running the code example.');
108     callback();
109   }
110 };
