 36 /**
 37  * This code example retrieves the number of operations recorded for the
 38  * developer token being used to make this call over the given date range.
 39  *
 40  * Tags: InfoService.get
 41  *
 42  * @extends google.ads.adwords.examples.ExampleBase
 43  * @constructor
 44  */
 45 google.ads.adwords.examples.v201008.GetOperationCount = function() {
 46   google.ads.adwords.examples.ExampleBase.call(this);
 47   /**
 48   * Example description.
 49   * @type {string}
 50   */
 51   this.description = 'This code example retrieves the number of operations ' +
 52       'recorded for the developer token being used to make this call over ' +
 53       'the given date range.';
 54 };
 55 goog.inherits(google.ads.adwords.examples.v201008.GetOperationCount,
 56     google.ads.adwords.examples.ExampleBase);
 57 
 58 /**
 59  * Runs the code example.
 60  *
 61  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 62  *     example.
 63  * @param {function} callback the callback method to be called once this example
 64  *     is complete.
 65  */
 66 google.ads.adwords.examples.v201008.GetOperationCount.prototype.run =
 67     function(user, callback) {
 68   // Get the InfoService.
 69   var infoService = user.getService(
 70       google.ads.adwords.AdWordsService.v201008.InfoService);
 71   infoService.requestHeader = new google.ads.adwords.v201008.RequestHeader();
 72   infoService.requestHeader.clientEmail = null;
 73 
 74   // To get results for a different date range, set startDate and endDates to
 75   // appropriate values.
 76   var startDate = this.getStartOfMonth();
 77   var endDate = this.getYesterday();
 78 
 79   var selector = new google.ads.adwords.v201008.InfoSelector();
 80   selector.apiUsageType =
 81       google.ads.adwords.v201008.ApiUsageType.OPERATION_COUNT;
 82   selector.dateRange = new google.ads.adwords.v201008.DateRange();
 83   selector.dateRange.min = startDate;
 84   selector.dateRange.max = endDate;
 85 
 86   try {
 87     infoService.get(selector,
 88       goog.bind(function(info) {
 89         if (info) {
 90           this.writeOutput('The total number of operations made during "%s"' +
 91               ' - "%s" is "%d".', startDate, endDate, info.cost);
 92         }
 93         callback();
 94       }, this),
 95       goog.bind(function(soapException) {
 96         this.writeOutput('Failed to get operation count. Soap Fault ' +
 97             'says "%s"', soapException.getInnerException().getFaultString());
 98         callback();
 99       }, this)
100     );
101   } catch (ex) {
102     this.writeOutput('An exception occurred while running the code example.');
103     callback();
104   }
105 };

107 /**
108  * Gets yesterday's date as a string of the format yyyymmdd.
109  *
110  * @return {string} yesterday's date as a string of the format yyyymmdd.
111  */
112 google.ads.adwords.examples.v201008.GetOperationCount.prototype.getYesterday =
113     function() {
114   var currentTime = new Date();
115   currentTime.setDate(currentTime.getDate() - 1);
116   var month = currentTime.getMonth() + 1;
117   var day = currentTime.getDate();
118   var year = currentTime.getFullYear();
119 
120   return goog.string.format('%d%02d%02d', year, month, day);
121 };
122 
123 /**
124  * Gets first day of the current month as a string of the format yyyymmdd.
125  *
126  * @return {string} first day of current month as a string of the format
127  * yyyymmdd.
128  */
129 google.ads.adwords.examples.v201008.GetOperationCount.prototype.
130     getStartOfMonth = function() {
131   var currentTime = new Date();
132   currentTime.setDate(1);
133   var month = currentTime.getMonth() + 1;
134   var day = currentTime.getDate();
135   var year = currentTime.getFullYear();
136   return goog.string.format('%d%02d%02d', year, month, day);
137 };
