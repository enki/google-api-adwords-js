 36 /**
 37  * This code example retrieves the number of API units recorded for the
 38  * developer token being used to make this call over the given date range.
 39  *
 40  * Tags: InfoService.get
 41  *
 42  * @extends google.ads.adwords.examples.ExampleBase
 43  * @constructor
 44  */
 45 google.ads.adwords.examples.v201008.GetUnitCount = function() {
 46   google.ads.adwords.examples.ExampleBase.call(this);
 47   this.description = 'This code example retrieves the number of API units ' +
 48       'recorded for the developer token being used to make this call over ' +
 49       'the given date range.';
 50 };
 51 goog.inherits(google.ads.adwords.examples.v201008.GetUnitCount,
 52     google.ads.adwords.examples.ExampleBase);
 53 
 54 /**
 55  * Runs the code example.
 56  *
 57  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 58  * example.
 59  * @param {function} callback the callback method to be called once this example
 60  * is complete.
 61  */
 62 google.ads.adwords.examples.v201008.GetUnitCount.prototype.run =
 63     function(user, callback) {
 64   // Get the InfoService.
 65   var infoService = user.getService(
 66       google.ads.adwords.AdWordsService.v201008.InfoService);
 67 
 68   infoService.requestHeader = new google.ads.adwords.v201008.RequestHeader();
 69   infoService.requestHeader.clientEmail = null;
 70 
 71   // To get results for a different date range, set startDate and endDates to
 72   // appropriate values.
 73   var startDate = this.getStartOfMonth();
 74   var endDate = this.getYesterday();
 75 
 76   var selector = new google.ads.adwords.v201008.InfoSelector();
 77   selector.apiUsageType =
 78       google.ads.adwords.v201008.ApiUsageType.UNIT_COUNT;
 79   selector.dateRange = new google.ads.adwords.v201008.DateRange();
 80   selector.dateRange.min = startDate;
 81   selector.dateRange.max = endDate;
 82 
 83   try {
 84     infoService.get(selector,
 85       goog.bind(function(info) {
 86         if (info) {
 87           this.writeOutput('The total number of API units consumed during ' +
 88               '"%s" to "%s" is %d".', startDate, endDate, info.cost);
 89         }
 90         callback();
 91       }, this),
 92       goog.bind(function(soapException) {
 93         this.writeOutput('Failed to get unit cost. Soap Fault ' +
 94             'says "%s"', soapException.getInnerException().getFaultString());
 95         callback();
 96       }, this)
 97     );
 98   } catch (ex) {
 99     this.writeOutput('An exception occurred while running the code example.');
100     callback();
101   }
102 };

104 /**
105  * Gets yesterday's date as a string of the format yyyymmdd.
106  *
107  * @return {string} yesterday's date as a string of the format yyyymmdd.
108  */
109 google.ads.adwords.examples.v201008.GetUnitCount.prototype.getYesterday =
110     function() {
111   var currentTime = new Date();
112   currentTime.setDate(currentTime.getDate() - 1);
113   var month = currentTime.getMonth() + 1;
114   var day = currentTime.getDate();
115   var year = currentTime.getFullYear();
116 
117   return goog.string.format('%d%02d%02d', year, month, day);
118 };
119 
120 /**
121  * Gets first day of the current month as a string of the format yyyymmdd.
122  *
123  * @return {string} first day of current month as a string of the format
124  * yyyymmdd.
125  */
126 google.ads.adwords.examples.v201008.GetUnitCount.prototype.
127     getStartOfMonth = function() {
128   var currentTime = new Date();
129   currentTime.setDate(1);
130   var month = currentTime.getMonth() + 1;
131   var day = currentTime.getDate();
132   var year = currentTime.getFullYear();
133   return goog.string.format('%d%02d%02d', year, month, day);
134 };
