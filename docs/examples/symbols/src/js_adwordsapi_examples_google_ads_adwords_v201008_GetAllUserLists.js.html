 19 /**
 20  * @fileoverview Defines GetAllUserLists, a code example that gets all the user
 21  * lists for an account.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetAllUserLists');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.UserListSelector');
 29 goog.require('google.ads.adwords.v201008.UserListService');
 30 
 31 /**
 32  * This code example illustrates how to retrieve all the user lists for
 33  * an account. To add a user list run AddUserList.js.
 34  *
 35  * Tags: UserListService.get
 36  *
 37  * @extends google.ads.adwords.examples.ExampleBase
 38  * @constructor
 39  */
 40 google.ads.adwords.examples.v201008.GetAllUserLists = function() {
 41   google.ads.adwords.examples.ExampleBase.call(this);
 42   this.description = 'This code example illustrates how to retrieve all the ' +
 43       'user lists for an account. To add a user list run AddUserList.js.';
 44 };
 45 goog.inherits(google.ads.adwords.examples.v201008.GetAllUserLists,
 46     google.ads.adwords.examples.ExampleBase);
 47 
 48 /**
 49  * Runs the code example.
 50  *
 51  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 52  * example.
 53  * @param {function} callback the callback method to be called once this example
 54  * is complete.
 55  */
 56 google.ads.adwords.examples.v201008.GetAllUserLists.prototype.run =
 57     function(user, callback) {
 58   // Get the UserListService.
 59   var userListService = user.getService(
 60       google.ads.adwords.AdWordsService.v201008.UserListService);
 61 
 62   var selector = new google.ads.adwords.v201008.UserListSelector();
 63 
 64   try {
 65     userListService.get(selector,
 66       goog.bind(function(page) {
 67         if (page && page.entries && page.entries.length > 0) {
 68           for (var i = 0, len = page.entries.length; i < len; i++) {
 69             var userList = page.entries[i];
 70             this.writeOutput('User list name is "%s", id is "%d", status is ' +
 71                 '"%s" and number of users is %d.', userList.name, userList.id,
 72                 userList.status, userList.size);
 73           }
 74         } else {
 75           this.writeOutput('No user lists were found.');
 76         }
 77       callback();
 78       }, this),
 79       goog.bind(function(soapException) {
 80         this.writeOutput('Failed to get user lists. Soap Fault ' +
 81             'says "%s"', soapException.getInnerException().getFaultString());
 82         callback();
 83       }, this)
 84     );
 85   } catch (ex) {
 86     this.writeOutput('An exception occurred while running the code example.');
 87     callback();
 88   }
 89 };
