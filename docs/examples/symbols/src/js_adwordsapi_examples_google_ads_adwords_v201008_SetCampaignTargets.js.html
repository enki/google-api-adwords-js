 19 /**
 20  * @fileoverview Defines SetCampaignTargets, a code example that adds geo,
 21  * language, and network targeting to an existing campaign.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.SetCampaignTargets');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.CampaignTargetOperation');
 29 goog.require('google.ads.adwords.v201008.CampaignTargetService');
 30 goog.require('google.ads.adwords.v201008.CountryTarget');
 31 goog.require('google.ads.adwords.v201008.GeoTargetList');
 32 goog.require('google.ads.adwords.v201008.LanguageTarget');
 33 goog.require('google.ads.adwords.v201008.LanguageTargetList');
 34 goog.require('google.ads.adwords.v201008.NetworkCoverageType');
 35 goog.require('google.ads.adwords.v201008.NetworkTarget');
 36 goog.require('google.ads.adwords.v201008.NetworkTargetList');
 37 goog.require('google.ads.adwords.v201008.Operator');
 38 
 39 /**
 40  * This code example adds geo, language, and network targeting to an
 41  * existing campaign. To get a campaign, run GetAllCampaigns.js.
 42  *
 43  * Tags: CampaignTargetService.mutate
 44  *
 45  * @extends google.ads.adwords.examples.ExampleBase
 46  * @constructor
 47  */
 48 google.ads.adwords.examples.v201008.SetCampaignTargets = function() {
 49   google.ads.adwords.examples.ExampleBase.call(this);
 50   this.description = 'This code example adds geo, language, and network ' +
 51       'targeting to an existing campaign. To get a campaign, run ' +
 52       'GetAllCampaigns.js.';
 53 };
 54 goog.inherits(google.ads.adwords.examples.v201008.SetCampaignTargets,
 55     google.ads.adwords.examples.ExampleBase);
 56 
 57 /**
 58  * Runs the code example.
 59  *
 60  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 61  * example.
 62  * @param {function} callback the callback method to be called once this example
 63  * is complete.
 64  */
 65 google.ads.adwords.examples.v201008.SetCampaignTargets.prototype.run =
 66     function(user, callback) {
 67   // Get the CampaignTargetService.
 68 
 69   var campaignTargetService = user.getService(
 70       google.ads.adwords.AdWordsService.v201008.CampaignTargetService);
 71 
 72   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 73 
 74   // Create language targets.
 75   var langTargetList = new google.ads.adwords.v201008.LanguageTargetList();
 76   langTargetList.campaignId = campaignId;
 77 
 78   var langTarget1 = new google.ads.adwords.v201008.LanguageTarget();
 79   langTarget1.languageCode = 'en';
 80 
 81   var langTarget2 = new google.ads.adwords.v201008.LanguageTarget();
 82   langTarget2.languageCode = 'ja';
 83 
 84   langTargetList.targets = [langTarget1, langTarget2];
 85 
 86   // Create language target set operation.
 87   var langTargetOperation =
 88       new google.ads.adwords.v201008.CampaignTargetOperation();
 89   langTargetOperation.operator = google.ads.adwords.v201008.Operator.SET;
 90   langTargetOperation.operand = langTargetList;
 91 
 92   // Create geo targets.
 93   var geoTargetList = new google.ads.adwords.v201008.GeoTargetList();
 94   geoTargetList.campaignId = campaignId;
 95 
 96   var geoTarget1 = new google.ads.adwords.v201008.CountryTarget();
 97   geoTarget1.countryCode = 'US';
 98 
 99   var geoTarget2 = new google.ads.adwords.v201008.CountryTarget();
100   geoTarget2.countryCode = 'JP';
101 
102   // Create geo target set operation.
103   var geoTargetOperation =
104       new google.ads.adwords.v201008.CampaignTargetOperation();
105   geoTargetOperation.operator = google.ads.adwords.v201008.Operator.SET;
106   geoTargetOperation.operand = geoTargetList;
107 
108   // Create network targets.
109   var networkTargetList = new google.ads.adwords.v201008.NetworkTargetList();
110   networkTargetList.campaignId = campaignId;
111 
112   // Specifying GOOGLE_SEARCH is necessary if you want to target
113   // SEARCH_NETWORK.
114   var networkTarget1 = new google.ads.adwords.v201008.NetworkTarget();
115   networkTarget1.networkCoverageType =
116       google.ads.adwords.v201008.NetworkCoverageType.GOOGLE_SEARCH;
117 
118   var networkTarget2 = new google.ads.adwords.v201008.NetworkTarget();
119   networkTarget2.networkCoverageType =
120       google.ads.adwords.v201008.NetworkCoverageType.SEARCH_NETWORK;
121 
122   networkTargetList.targets = [networkTarget1, networkTarget2];
123 
124   // Create network target set operation.
125   var networkTargetOperation =
126       new google.ads.adwords.v201008.CampaignTargetOperation();
127   networkTargetOperation.operator = google.ads.adwords.v201008.Operator.SET;
128   networkTargetOperation.operand = networkTargetList;
129 
130   try {
131     campaignTargetService.mutate(
132       [geoTargetOperation, langTargetOperation, networkTargetOperation],
133       goog.bind(function(retval) {
134         // Display campaign targets.
135         if (retval && retval.value && retval.value.length > 0) {
136           for (var i = 0, len = retval.value.length; i < len; i++) {
137             var targetList = retval.value[i];
138             this.writeOutput('Campaign target of type "%s" was set to ' +
139                 'Campaign with id = "%s".', targetList.TargetListType,
140                 targetList.campaignId);
141           }
142         } else {
143           this.writeOutput('No campaign target(s) were set.');
144         }
145         callback();
146       }, this),
147       goog.bind(function(soapException) {
148         this.writeOutput('Failed to set campaign target(s). Soap Fault says ' +
149             '"%s"', soapException.getInnerException().getFaultString());
150         callback();
151       }, this)
152     );
153   } catch (ex) {
154     this.writeOutput('An exception occurred while running the code example.');
155     callback();
156   }
157 };
