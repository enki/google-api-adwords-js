 19 /**
 20  * @fileoverview Defines UpdateAd, a code example that updates an ad's
 21  * status given an existing ad group and ad.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.UpdateAd');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.Ad');
 29 goog.require('google.ads.adwords.v201008.AdGroupAd');
 30 goog.require('google.ads.adwords.v201008.AdGroupAdOperation');
 31 goog.require('google.ads.adwords.v201008.AdGroupAdService');
 32 goog.require('google.ads.adwords.v201008.Operator');
 33 
 34 /**
 35  * This code example updates an ad's status given an existing ad group and ad.
 36  *
 37  * Tags: AdGroupAdService.mutate
 38  *
 39  * @extends google.ads.adwords.examples.ExampleBase
 40  * @constructor
 41  */
 42 google.ads.adwords.examples.v201008.UpdateAd = function() {
 43   google.ads.adwords.examples.ExampleBase.call(this);
 44   this.description = 'This code example updates an ad\'s status given an ' +
 45       'existing ad group and ad.';
 46 };
 47 goog.inherits(google.ads.adwords.examples.v201008.UpdateAd,
 48     google.ads.adwords.examples.ExampleBase);
 49 
 50 /**
 51  * Runs the code example.
 52  *
 53  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 54  * example.
 55  * @param {function} callback the callback method to be called once this example
 56  * is complete.
 57  */
 58 google.ads.adwords.examples.v201008.UpdateAd.prototype.run =
 59     function(user, callback) {
 60   // Get the AdGroupAdService.
 61   var adGroupAdService = user.getService(
 62       google.ads.adwords.AdWordsService.v201008.AdGroupAdService);
 63 
 64   var adGroupId = this._T('INSERT_ADGROUP_ID_HERE');
 65   var adId = this._T('INSERT_AD_ID_HERE');
 66   var status = this._T('INSERT_AD_GROUP_AD_STATUS_HERE');
 67 
 68   // Update your Ad.
 69   var adGroupAd = new google.ads.adwords.v201008.AdGroupAd();
 70   adGroupAd.status = status;
 71   adGroupAd.adGroupId = adGroupId;
 72 
 73   adGroupAd.ad = new google.ads.adwords.v201008.Ad();
 74   adGroupAd.ad.id = adId;
 75 
 76   var operation = new google.ads.adwords.v201008.AdGroupAdOperation();
 77   operation.operator = google.ads.adwords.v201008.Operator.SET;
 78   operation.operand = adGroupAd;
 79 
 80   try {
 81     adGroupAdService.mutate([operation],
 82       goog.bind(function(retval) {
 83         if (retval && retval.value && retval.value.length > 0) {
 84           for (var i = 0, len = retval.value.length; i < len; i++) {
 85             var adGroupAdValue = retval.value[i];
 86             this.writeOutput('Status of ad with id "%s" was set to "%s"',
 87                 adGroupAdValue.ad.id, adGroupAdValue.status);
 88           }
 89         } else {
 90           this.writeOutput('No ads were updated.');
 91         }
 92         callback();
 93       }, this),
 94       goog.bind(function(soapException) {
 95         this.writeOutput('Failed to update ads(s). Soap Fault says "%s"',
 96             soapException.getInnerException().getFaultString());
 97         callback();
 98       }, this)
 99     );
100   } catch (ex) {
101     this.writeOutput('An exception occurred while running the code example.');
102     callback();
103   }
104 };
