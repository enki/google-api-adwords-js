 19 /**
 20  * @fileoverview Defines ExampleBase, the base class for all code examples.
 21  */
 22 
 23 goog.provide('google.ads.adwords.examples.ExampleBase');
 24 
 25 goog.require('goog.string');
 26 goog.require('goog.string.format');
 27 
 28 /**
 29  * Base class for all code examples.
 30  *
 31  * @constructor
 32  */
 33 google.ads.adwords.examples.ExampleBase = function() {
 34   this.description = '';
 35   this.output = '';
 36 };
 37 
 38 /**
 39  * Runs the code example.
 40  *
 41  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 42  *     example.
 43  * @param {function} callback the callback method to be called once this example
 44  *     is complete.
 45  */
 46 google.ads.adwords.examples.ExampleBase.prototype.run =
 47     function(user, callback) {
 48 };
 49 
 50 /**
 51  * Reads an input from the user.
 52  *
 53  * @param {string} message The prompt to display.
 54  * @return {string} The user input.
 55  */
 56 google.ads.adwords.examples.ExampleBase.prototype._T = function(message) {
 57   return prompt(message, '');
 58 };
 59 
 60 /**
 61  * Generates a unique number.
 62  *
 63  * @return {string} The current timestamp as a string.
 64  */
 65 google.ads.adwords.examples.ExampleBase.prototype.getTimeStamp = function() {
 66   return goog.string.createUniqueString();
 67 };
 68 
 69 /**
 70  * Performs sprintf-like conversion, ie. puts the values in a template.
 71  * DO NOT use it instead of built-in conversions in simple cases such as
 72  * 'Cost: %.2f' as it would introduce unneccessary latency oposed to
 73  * 'Cost: ' + cost.toFixed(2).
 74  *
 75  * @param {string} formatString Template string containing % specifiers.
 76  * @param {...string|number} var_args Values formatString is to be filled with.
 77  */
 78 google.ads.adwords.examples.ExampleBase.prototype.writeOutput = function(
 79     formatString, var_args) {
 80   this.output += goog.string.format.apply(this, arguments) + '\n';
 81 };
