 19 /**
 20  * @fileoverview Defines GetAllExperiments, a code example that gets all
 21  *     experiments in a campaign.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201109.GetAllExperiments');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201109');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201109.ExperimentService');
 29 goog.require('google.ads.adwords.v201109.Predicate');
 30 goog.require('google.ads.adwords.v201109.PredicateOperator');
 31 goog.require('google.ads.adwords.v201109.Selector');
 32 
 33 /**
 34  * This code example gets all experiments in a campaign. To add an experiment,
 35  * run AddExperiment.js. To get campaigns, run GetAllCampaigns.js.
 36  *
 37  * Tags: ExperimentService.get
 38  *
 39  * @extends google.ads.adwords.examples.ExampleBase
 40  * @constructor
 41  */
 42 google.ads.adwords.examples.v201109.GetAllExperiments = function() {
 43   google.ads.adwords.examples.ExampleBase.call(this);
 44   this.description = 'This code example gets all experiments in a campaign. ' +
 45       'To add an experiment, run AddExperiment.js. To get campaigns, run ' +
 46       'GetAllCampaigns.js.';
 47 };
 48 goog.inherits(google.ads.adwords.examples.v201109.GetAllExperiments,
 49     google.ads.adwords.examples.ExampleBase);
 50 
 51 /**
 52  * Runs the code example.
 53  *
 54  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 55  *     example.
 56  * @param {function} callback the callback method to be called once this example
 57  *     is complete.
 58  */
 59 google.ads.adwords.examples.v201109.GetAllExperiments.prototype.run =
 60     function(user, callback) {
 61   // Get the ExperimentService.
 62   var experimentService = user.getService(
 63       google.ads.adwords.AdWordsService.v201109.ExperimentService);
 64 
 65   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 66 
 67   // Create selector.
 68   var selector = new google.ads.adwords.v201109.Selector();
 69   selector.fields = ['Name', 'Id', 'CampaignId', 'ControlId', 'AdGroupsCount',
 70       'AdGroupCriteriaCount', 'AdGroupAdsCount'];
 71 
 72   // Create a selector.
 73   var predicate = new google.ads.adwords.v201109.Predicate();
 74   predicate.field = "CampaignId";
 75   predicate.operator = google.ads.adwords.v201109.PredicateOperator.EQUALS;
 76   predicate.values = [campaignId];
 77 
 78   try {
 79     // Get all experiments.
 80     experimentService.get(selector,
 81       goog.bind(function(page) {
 82         // Display results.
 83         if (page && page.entries) {
 84           for (var i = 0, len = page.entries.length; i < len; i++) {
 85             var experiment = page.entries[i];
 86             var stats = experiment.experimentSummaryStats;
 87             this.writeOutput('Experiment with name = "%s", id = "%d" and ' +
 88                 'control id = "%d" was found and it includes %d ad group(s), ' +
 89                 '%d criteria, and %d ads.', experiment.name, experiment.id,
 90                 experiment.controlId, stats.adGroupsCount,
 91                 stats.adGroupCriteriaCount, stats.adGroupAdsCount);
 92           }
 93         } else {
 94           this.writeOutput('No experiments were found.');
 95         }
 96         callback();
 97       }, this),
 98       goog.bind(function(soapException) {
 99         this.writeOutput('Failed to get add user list. Soap Fault ' +
100             'says "%s"', soapException.getInnerException().getFaultString());
101         callback();
102       }, this)
103     );
104   } catch (ex) {
105     this.writeOutput('An exception occurred while running the code example.');
106     callback();
107   }
108 };
