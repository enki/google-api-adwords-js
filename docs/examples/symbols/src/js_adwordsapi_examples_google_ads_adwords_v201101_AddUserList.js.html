 21  * a user list.
 22  */

 24 goog.provide('google.ads.adwords.examples.v201101.AddUserList');

 26 goog.require('google.ads.adwords.AdWordsService.v201101');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201101.Operator');
 29 goog.require('google.ads.adwords.v201101.Predicate');
 30 goog.require('google.ads.adwords.v201101.PredicateOperator');
 31 goog.require('google.ads.adwords.v201101.RemarketingUserList');
 32 goog.require('google.ads.adwords.v201101.Selector');
 33 goog.require('google.ads.adwords.v201101.UserListConversionType');
 34 goog.require('google.ads.adwords.v201101.UserListMembershipStatus');
 35 goog.require('google.ads.adwords.v201101.UserListOperation');
 36 goog.require('google.ads.adwords.v201101.UserListService');

 38 /**
 39  * This code example illustrates how to create a user list.
 40  *
 41  * Tags: UserListService.mutate
 42  *
 43  * @extends google.ads.adwords.examples.ExampleBase
 44  * @constructor
 45  */
 46 google.ads.adwords.examples.v201101.AddUserList = function() {
 47   google.ads.adwords.examples.ExampleBase.call(this);
 48   this.description = 'This code example illustrates how to create a user list.';
 49 };
 50 goog.inherits(google.ads.adwords.examples.v201101.AddUserList,
 51     google.ads.adwords.examples.ExampleBase);
 52 
 53 /**
 54  * Runs the code example.
 55  *
 56  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 57  * example.
 58  * @param {function} callback the callback method to be called once this example
 59  * is complete.
 60  */
 61 google.ads.adwords.examples.v201101.AddUserList.prototype.run =
 62     function(user, callback) {
 63   // Get the UserListService.
 64   var userListService = user.getService(
 65       google.ads.adwords.AdWordsService.v201101.UserListService);
 66 
 67   // Get the ConversionTrackerService.
 68   var conversionTrackerService = user.getService(
 69       google.ads.adwords.AdWordsService.v201101.ConversionTrackerService);
 70 
 71 
 72   var userList = new google.ads.adwords.v201101.RemarketingUserList();
 73   userList.name = 'Mars cruise customers #' + this.getTimeStamp();
 74   userList.description = 'A list of mars cruise customers in the last year.';
 75   userList.status = google.ads.adwords.v201101.UserListMembershipStatus.OPEN;
 76   userList.membershipLifeSpan = 365;
 77 
 78   var conversionType = new google.ads.adwords.v201101.UserListConversionType();
 79   conversionType.name = userList.name;
 80   userList.conversionTypes = [conversionType];
 81 
 82   var operation = new google.ads.adwords.v201101.UserListOperation();
 83   operation.operand = userList;
 84   operation.operator = google.ads.adwords.v201101.Operator.ADD;
 85 
 86   try {
 87     // Add user list.
 88     userListService.mutate([operation],
 89       goog.bind(function(retval) {
 90         if (retval && retval.value && retval.value.length > 0) {
 91           userLists = retval.value;
 92           // Get all conversion snippets
 93           var conversionIds = [];
 94           for (var i = 0; i < userLists.length; i++) {
 95             if (userLists[i].conversionTypes != null) {
 96               for (var j = 0; j < userLists[i].conversionTypes.length; j++) {
 97                 conversionIds.push("" + userLists[i].conversionTypes[j].id);
 98               }
 99             }
100           }
101 
102           var conversionsMap = {};
103 
104           if (conversionIds.length > 0) {
105             // Create selector.
106             var conversionTypePredicate =
107                 new google.ads.adwords.v201101.Predicate();
108             conversionTypePredicate.field = "Id";
109             conversionTypePredicate.operator =
110                 google.ads.adwords.v201101.PredicateOperator.IN;
111             conversionTypePredicate.values = conversionIds;
112 
113             var selector = new google.ads.adwords.v201101.Selector();
114             selector.fields = ["Id"];
115             selector.predicates = [conversionTypePredicate];
116 
117             // Get all conversion trackers.
118             conversionTrackerService.get(selector, goog.bind(function(page) {
119               if (page != null && page.entries != null) {
120                 for (var i = 0; i < page.entries.length; i++) {
121                   var tracker = page.entries[i];
122                   conversionsMap[tracker.id] = tracker;
123                 }
124               }
125 
126               // Display results.
127               for (var i = 0; i < userLists.length; i++) {
128                 var tempUserList = userLists[i];
129                 this.writeOutput("User list with name '%s' and id '%s' was " +
130                     "added.", tempUserList.name, tempUserList.id);
131 
132                 // Display user list associated conversion code snippets.
133                 if (tempUserList.conversionTypes != null) {
134                   for (var j = 0; j < tempUserList.conversionTypes.length;
135                       j++) {
136                     var tempConversionType = tempUserList.conversionTypes[j];
137                     var conversionTracker =
138                         conversionsMap[tempConversionType.id];
139                     this.writeOutput("Conversion type code snippet " +
140                         "associated to the list:\n%s\n",
141                         conversionTracker.snippet);
142                   }
143                 }
144               }
145               callback();
146             }, this), goog.bind(function(soapException) {
147               this.writeOutput('Failed to get conversion tracker snippet. ' +
148                   'Soap Fault says "%s"', soapException.getInnerException().
149                     getFaultString());
150               callback();
151             }, this));
152           }
153         } else {
154           this.writeOutput('No user lists were added.');
155         }
156         callback();
157       }, this),
158       goog.bind(function(soapException) {
159         this.writeOutput('Failed to get add user list. Soap Fault ' +
160             'says "%s"', soapException.getInnerException().getFaultString());
161         callback();
162       }, this)
163     );
164   } catch (ex) {
165     this.writeOutput('An exception occurred while running the code example.');
166     callback();
167   }
168 };
