 40 /**
 41  * This code example adds a keywords performance report. To get ad groups,
 42  * run GetAllAdGroups.js. To get report fields, run GetReportFields.js.
 43  *
 44  * Tags: ReportDefinitionService.mutate
 45  *
 46  * @extends google.ads.adwords.examples.ExampleBase
 47  * @constructor
 48  */
 49 google.ads.adwords.examples.v201101.AddKeywordsPerformanceReportDefinition =
 50     function() {
 51   google.ads.adwords.examples.ExampleBase.call(this);
 52   this.description = 'This code example adds a keywords performance report. ' +
 53       'To get ad groups, run GetAllAdGroups.js. To get report fields, run ' +
 54       'GetReportFields.js.';
 55 };
 56 goog.inherits(
 57     google.ads.adwords.examples.v201101.AddKeywordsPerformanceReportDefinition,
 58     google.ads.adwords.examples.ExampleBase);
 59 
 60 /**
 61  * Runs the code example.
 62  *
 63  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 64  * example.
 65  * @param {function} callback the callback method to be called once this example
 66  * is complete.
 67  */
 68 google.ads.adwords.examples.v201101.AddKeywordsPerformanceReportDefinition.
 69     prototype.run = function(user, callback) {
 70   // Get the ReportDefinitionService.
 71   var reportDefinitionService = user.getService(
 72       google.ads.adwords.AdWordsService.v201101.ReportDefinitionService);
 73 
 74   var adGroupId = this._T('INSERT_AD_GROUP_ID_HERE');
 75   var startDate = this._T('INSERT_START_DATE_HERE');
 76   var endDate = this._T('INSERT_END_DATE_HERE');
 77 
 78   // Create ad group predicate.
 79   var adGroupPredicate = new google.ads.adwords.v201101.Predicate();
 80   adGroupPredicate.field = 'AdGroupId';
 81   adGroupPredicate.operator =
 82       google.ads.adwords.v201101.PredicateOperator.EQUALS;
 83   adGroupPredicate.values = [adGroupId];
 84 
 85   // Create selector.
 86   var selector = new google.ads.adwords.v201101.Selector();
 87   selector.fields = ['AdGroupId', 'Id', 'KeywordText', 'KeywordMatchType',
 88       'Impressions', 'Clicks', 'Cost'];
 89   selector.predicates = [adGroupPredicate];
 90   selector.dateRange = new google.ads.adwords.v201101.DateRange();
 91   selector.dateRange.min = startDate;
 92   selector.dateRange.max = endDate;
 93 
 94   // Create report definition.
 95   var reportDefinition = new google.ads.adwords.v201101.ReportDefinition();
 96   reportDefinition.reportName = 'Keywords performance report #' +
 97       this.getTimeStamp();
 98   reportDefinition.dateRangeType =
 99       google.ads.adwords.v201101.ReportDefinitionDateRangeType.CUSTOM_DATE;
100   reportDefinition.reportType = google.ads.adwords.v201101.
101       ReportDefinitionReportType.KEYWORDS_PERFORMANCE_REPORT;
102   reportDefinition.downloadFormat =
103       google.ads.adwords.v201101.DownloadFormat.XML;
104   reportDefinition.selector = selector;
105 
106   // Create operations.
107   var operation = new google.ads.adwords.v201101.ReportDefinitionOperation();
108   operation.operand = reportDefinition;
109   operation.operator = google.ads.adwords.v201101.Operator.ADD;
110 
111   try {
112     // Add report definition.
113     reportDefinitionService.mutate([operation],
114       goog.bind(function(result) {
115         // Display report definitions.
116         if (result && result.length > 0) {
117           this.writeOutput('Report definition with name "%s" and id "%d" ' +
118               'was added.', result[0].reportName, result[0].id);
119         } else {
120           this.writeOutput('No report definitions were added.');
121         }
122         callback();
123       }, this),
124       goog.bind(function(soapException) {
125         this.writeOutput('Failed to create keyword performance report. Soap ' +
126             'Fault says "%s"', soapException.getInnerException().
127             getFaultString());
128         callback();
129       }, this)
130     );
131   } catch (ex) {
132     this.writeOutput('An exception occurred while running the code example.');
133     callback();
134   }
135 };
