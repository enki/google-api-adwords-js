 19 /**
 20  * @fileoverview Defines GetCampaign, a code example that gets a campaign
 21  * by id.
 22  */
 23 
 24 goog.provide('google.ads.adwords.examples.v201008.GetCampaign');
 25 
 26 goog.require('google.ads.adwords.AdWordsService.v201008');
 27 goog.require('google.ads.adwords.examples.ExampleBase');
 28 goog.require('google.ads.adwords.v201008.CampaignSelector');
 29 goog.require('google.ads.adwords.v201008.CampaignService');
 30 
 31 /**
 32  * This code example gets a campaign by id. To add a campaign, run
 33  * AddCampaign.js.
 34  *
 35  * Tags: CampaignService.get
 36  *
 37  * @extends google.ads.adwords.examples.ExampleBase
 38  * @constructor
 39  */
 40 google.ads.adwords.examples.v201008.GetCampaign = function() {
 41   google.ads.adwords.examples.ExampleBase.call(this);
 42   this.description = 'This code example gets a campaign by id. To add a ' +
 43       'campaign, run AddCampaign.js.';
 44 };
 45 goog.inherits(google.ads.adwords.examples.v201008.GetCampaign,
 46     google.ads.adwords.examples.ExampleBase);
 47 
 48 /**
 49  * Runs the code example.
 50  *
 51  * @param {google.ads.adwords.AdWordsUser} user AdWords user running the code
 52  * example.
 53  * @param {function} callback the callback method to be called once this example
 54  * is complete.
 55  */
 56 google.ads.adwords.examples.v201008.GetCampaign.prototype.run =
 57     function(user, callback) {
 58   // Get the CampaignService.
 59   var campaignService = user.getService(
 60       google.ads.adwords.AdWordsService.v201008.CampaignService);
 61 
 62   var campaignId = this._T('INSERT_CAMPAIGN_ID_HERE');
 63 
 64   var selector = new google.ads.adwords.v201008.CampaignSelector();
 65   selector.ids = [campaignId];
 66 
 67   try {
 68     campaignService.get(selector,
 69       goog.bind(function(page) {
 70         if (page && page.entries && page.entries.length > 0) {
 71           for (var i = 0, len = page.entries.length; i < len; i++) {
 72             var campaignValue = page.entries[i];
 73             this.writeOutput('Campaign with id = "%s", name = "%s" and ' +
 74                 'status = "%s" was found.', campaignValue.id,
 75                 campaignValue.name, campaignValue.status);
 76           }
 77         } else {
 78           this.writeOutput('No campaigns were found.');
 79         }
 80         callback();
 81       }, this),
 82       goog.bind(function(soapException) {
 83         this.writeOutput('Failed to get campaign(s). Soap Fault says "%s"',
 84             soapException.getInnerException().getFaultString());
 85         callback();
 86       }, this)
 87     );
 88   } catch (ex) {
 89     this.writeOutput('An exception occurred while running the code example.');
 90     callback();
 91   }
 92 };
